110                 + "try { System.out.println(); } catch (RuntimeException | IOException e) {} } }";

115             assertEquals(
116                     "Line 1, Column 94: Cannot catch multiple exceptions when running in JDK inferior to 1.7 mode!",
117                     e.getMessage());
118         }

121             parseJava17(code);

123         } catch (ParseException e) {

147                 "    public static void main(String[] args) {\n" +

155     private String readAsString(String resource) {
156         InputStream in = ParserCornersTest.class.getResourceAsStream(resource);
157         try {
158             return IOUtils.toString(in);
159         } catch (IOException e) {
160             throw new RuntimeException(e);
161         } finally {
162             IOUtils.closeQuietly(in);
163         }
164     }
