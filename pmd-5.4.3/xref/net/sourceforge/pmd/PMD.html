71      /**
72       * Parses the given string as a database uri and returns a list of datasources.
73       * @param uriString the URI to parse
74       * @return list of data sources
75       * @throws PMDException if the URI couldn't be parsed
76       * @see DBURI
77       */
78      public static List<DataSource> getURIDataSources(String uriString) throws PMDException {
79          List<DataSource> dataSources = new ArrayList<>();
80  
81          try {
82              DBURI dbUri = new DBURI(uriString);
83              DBMSMetadata dbmsMetadata = new DBMSMetadata(dbUri);
84              LOG.log(Level.FINE, "DBMSMetadata retrieved");
85              List<SourceObject> sourceObjectList = dbmsMetadata.getSourceObjectList();
86              LOG.log(Level.FINE, "Located {0} database source objects", sourceObjectList.size());
87              for (SourceObject sourceObject : sourceObjectList) {
88                  String falseFilePath = sourceObject.getPseudoFileName();
89                  LOG.log(Level.FINEST, "Adding database source object {0}", falseFilePath);
90  
91                  try {
92                      dataSources.add(new ReaderDataSource(dbmsMetadata.getSourceCode(sourceObject), falseFilePath));
93                  } catch (SQLException ex) {
94                      if (LOG.isLoggable(Level.WARNING)) {
95                          LOG.log(Level.WARNING, "Cannot get SourceCode for " + falseFilePath + "  - skipping ...", ex);
96                      }
97                  }
98              }
99          } catch (URISyntaxException e) {
100             throw new PMDException("Cannot get DataSources from DBURI - \"" + uriString + "\"", e);
101         } catch (SQLException e) {
102             throw new PMDException("Cannot get DataSources from DBURI, couldn't access the database - \"" + uriString
103                     + "\"", e);
104         } catch (ClassNotFoundException e) {
105             throw new PMDException("Cannot get DataSources from DBURI, probably missing database jdbc driver - \""
106                     + uriString + "\"", e);
107         } catch (Exception e) {
108             throw new PMDException("Encountered unexpected problem with URI \""
109                     + uriString + "\"", e);
110         }
111         return dataSources;
112     }

119     /**
120      * Helper method to get a configured parser for the requested language. The parser is
121      * configured based on the given {@link PMDConfiguration}.
122      * @param languageVersion the requested language
123      * @param configuration the given configuration
124      * @return the pre-configured parser
125      */
126     public static Parser parserFor(LanguageVersion languageVersion, PMDConfiguration configuration) {
127 
128         // TODO Handle Rules having different parser options.
129         LanguageVersionHandler languageVersionHandler = languageVersion.getLanguageVersionHandler();
130         ParserOptions options = languageVersionHandler.getDefaultParserOptions();
131         if (configuration != null) {
132             options.setSuppressMarker(configuration.getSuppressMarker());
133         }
134         return languageVersionHandler.getParser(options);
135     }

137     /**
138      * Create a report, filter out any defective rules, and keep a record of
139      * them.
140      * 
141      * @param rs the rules
142      * @param ctx the rule context
143      * @param fileName the filename of the source file, which should appear in the report
144      * @return the Report
145      */
146     public static Report setupReport(RuleSets rs, RuleContext ctx, String fileName) {
147 
148         Set<Rule> brokenRules = removeBrokenRules(rs);
149         Report report = Report.createReport(ctx, fileName);
150 
151         for (Rule rule : brokenRules) {
152             report.addConfigError(new Report.RuleConfigurationError(rule, rule.dysfunctionReason()));
153         }
154 
155         return report;
156     }

158     /**
159      * Remove and return the misconfigured rules from the rulesets and log them
160      * for good measure.
161      * 
162      * @param ruleSets
163      *            RuleSets
164      * @return Set<Rule>
165      */
166     private static Set<Rule> removeBrokenRules(RuleSets ruleSets) {
167 
168         Set<Rule> brokenRules = new HashSet<>();
169         ruleSets.removeDysfunctionalRules(brokenRules);
170 
171         for (Rule rule : brokenRules) {
172             if (LOG.isLoggable(Level.WARNING)) {
173                 LOG.log(Level.WARNING,
174                     "Removed misconfigured rule: " + rule.getName() + "  cause: " + rule.dysfunctionReason());
175             }
176         }
177 
178         return brokenRules;
179     }

219     /**
220      * This method is the main entry point for command line usage.
221      * 
222      * @param configuration the configure to use
223      * @return number of violations found.
224      */
225     public static int doPMD(PMDConfiguration configuration) {
226 
227         // Load the RuleSets
228         RuleSetFactory ruleSetFactory = RulesetsFactoryUtils.getRulesetFactory(configuration);
229         RuleSets ruleSets = RulesetsFactoryUtils.getRuleSetsWithBenchmark(configuration.getRuleSets(), ruleSetFactory);
230         if (ruleSets == null) {
231             return 0;
232         }
233 
234         Set<Language> languages = getApplicableLanguages(configuration, ruleSets);
235         List<DataSource> files = getApplicableFiles(configuration, languages);
236 
237         long reportStart = System.nanoTime();
238         try {
239             Renderer renderer = configuration.createRenderer();
240             List<Renderer> renderers = new LinkedList<>();
241             renderers.add(renderer);
242 
243             renderer.setWriter(IOUtil.createWriter(configuration.getReportFile()));
244             renderer.start();
245 
246             Benchmarker.mark(Benchmark.Reporting, System.nanoTime() - reportStart, 0);
247 
248             RuleContext ctx = new RuleContext();
249             final AtomicInteger violations = new AtomicInteger(0);
250             ctx.getReport().addListener(new ReportListener() {
251                 @Override
252                 public void ruleViolationAdded(RuleViolation ruleViolation) {
253                     violations.incrementAndGet();
254                 }
255                 @Override
256                 public void metricAdded(Metric metric) {
257                 }
258             });
259 
260             processFiles(configuration, ruleSetFactory, files, ctx, renderers);
261 
262             reportStart = System.nanoTime();
263             renderer.end();
264             renderer.flush();
265             return violations.get();
266         } catch (Exception e) {
267             String message = e.getMessage();
268             if (message != null) {
269                 LOG.severe(message);
270             } else {
271                 LOG.log(Level.SEVERE, "Exception during processing", e);
272             }
273             LOG.log(Level.FINE, "Exception during processing", e);
274             LOG.info(PMDCommandLineInterface.buildUsageText());
275             return 0;
276         } finally {
277             Benchmarker.mark(Benchmark.Reporting, System.nanoTime() - reportStart, 0);
278         }
279     }

335     /**
336      * Run PMD on a list of files using multiple threads - if more than one is
337      * available
338      * 
339      * @param configuration
340      *            Configuration
341      * @param ruleSetFactory
342      *            RuleSetFactory
343      * @param files
344      *            List<DataSource>
345      * @param ctx
346      *            RuleContext
347      * @param renderers
348      *            List<Renderer>
349      */
350     public static void processFiles(final PMDConfiguration configuration, final RuleSetFactory ruleSetFactory,
351             final List<DataSource> files, final RuleContext ctx, final List<Renderer> renderers) {
352 
353         sortFiles(configuration, files);
354 
355         /*
356          * Check if multithreaded support is available. ExecutorService can also
357          * be disabled if threadCount is not positive, e.g. using the
358          * "-threads 0" command line option.
359          */
360         if (SystemUtils.MT_SUPPORTED && configuration.getThreads() > 0) {
361             new MultiThreadProcessor(configuration).processFiles(ruleSetFactory, files, ctx, renderers);
362         } else {
363             new MonoThreadProcessor(configuration).processFiles(ruleSetFactory, files, ctx, renderers);
364         }
365 
366         if (configuration.getClassLoader() instanceof ClasspathClassLoader) {
367             IOUtil.tryCloseClassLoader(configuration.getClassLoader());
368         }
369     }

371     private static void sortFiles(final PMDConfiguration configuration, final List<DataSource> files) {
372         if (configuration.isStressTest()) {
373             // randomize processing order
374             Collections.shuffle(files);
375         } else {
376             final boolean useShortNames = configuration.isReportShortNames();
377             final String inputPaths = configuration.getInputPaths();
378             Collections.sort(files, new Comparator<DataSource>() {
379                 public int compare(DataSource left, DataSource right) {
380                     String leftString = left.getNiceFileName(useShortNames, inputPaths);
381                     String rightString = right.getNiceFileName(useShortNames, inputPaths);
382                     return leftString.compareTo(rightString);
383                 }
384             });
385         }
386     }

388     /**
389      * Determines all the files, that should be analyzed by PMD.
390      * @param configuration contains either the file path or the DB URI, from where to load the files
391      * @param languages used to filter by file extension
392      * @return List<DataSource> of files
393      */
394     public static List<DataSource> getApplicableFiles(PMDConfiguration configuration, Set<Language> languages) {
395         long startFiles = System.nanoTime();
396         List<DataSource> files = internalGetApplicableFiles(configuration, languages);
397         long endFiles = System.nanoTime();
398         Benchmarker.mark(Benchmark.CollectFiles, endFiles - startFiles, 0);
399         return files;
400     }

402     private static List<DataSource> internalGetApplicableFiles(PMDConfiguration configuration, Set<Language> languages) {
403         LanguageFilenameFilter fileSelector = new LanguageFilenameFilter(languages);
404         List<DataSource> files = new ArrayList<>();
405 
406         if (null != configuration.getInputPaths()) {
407             files.addAll(FileUtil.collectFiles(configuration.getInputPaths(), fileSelector));
408         }
409 
410         if (null != configuration.getInputUri()) {
411             String uriString = configuration.getInputUri();
412             try {
413                 List<DataSource> dataSources = getURIDataSources(uriString);
414 
415                 files.addAll(dataSources);
416             } catch (PMDException ex) {
417                 LOG.log(Level.SEVERE, "Problem with Input URI", ex);
418                 throw new RuntimeException("Problem with DBURI: " + uriString, ex);
419             }
420         }
421         return files;
422     }

424     private static Set<Language> getApplicableLanguages(PMDConfiguration configuration, RuleSets ruleSets) {
425         Set<Language> languages = new HashSet<>();
426         LanguageVersionDiscoverer discoverer = configuration.getLanguageVersionDiscoverer();
427 
428         for (Rule rule : ruleSets.getAllRules()) {
429             Language language = rule.getLanguage();
430             if (languages.contains(language)) {
431                 continue;
432             }
433             LanguageVersion version = discoverer.getDefaultLanguageVersion(language);
434             if (RuleSet.applies(rule, version)) {
435                 languages.add(language);
436                 if (LOG.isLoggable(Level.FINE)) {
437                     LOG.fine("Using " + language.getShortName() + " version: " + version.getShortName());
438                 }
439             }
440         }
441         return languages;
442     }

453     /**
454      * Parses the command line arguments and executes PMD.
455      * @param args command line arguments
456      * @return the exit code, where <code>0</code> means successful execution, <code>1</code> means error,
457      * <code>4</code> means there have been violations found.
458      */
459     public static int run(String[] args) {
460         int status = 0;
461         long start = System.nanoTime();
462         final PMDParameters params = PMDCommandLineInterface.extractParameters(new PMDParameters(), args, "pmd");
463         final PMDConfiguration configuration = PMDParameters.transformParametersIntoConfiguration(params);
464 
465         final Level logLevel = params.isDebug() ? Level.FINER : Level.INFO;
466         final Handler logHandler = new ConsoleLogHandler();
467         final ScopedLogHandlersManager logHandlerManager = new ScopedLogHandlersManager(logLevel, logHandler);
468         final Level oldLogLevel = LOG.getLevel();
469         LOG.setLevel(logLevel); // Need to do this, since the static logger has
470                                 // already been initialized at this point
471         try {
472             int violations = PMD.doPMD(configuration);
473             if (violations > 0 && configuration.isFailOnViolation()) {
474                 status = PMDCommandLineInterface.VIOLATIONS_FOUND;
475             } else {
476                 status = 0;
477             }
478         } catch (Exception e) {
479             System.out.println(PMDCommandLineInterface.buildUsageText());
480             System.out.println();
481             System.err.println(e.getMessage());
482             status = PMDCommandLineInterface.ERROR_STATUS;
483         } finally {
484             logHandlerManager.close();
485             LOG.setLevel(oldLogLevel);
486             if (params.isBenchmark()) {
487                 long end = System.nanoTime();
488                 Benchmarker.mark(Benchmark.TotalPMD, end - start, 0);
489 
490                 TextReport report = new TextReport(); // TODO get specified
491                                                       // report format from
492                                                       // config
493                 report.generate(Benchmarker.values(), System.err);
494             }
495         }
496         return status;
497     }

499     /**
500      * Constant that contains always the current version of PMD.
501      */
502     public static final String VERSION;
503     /**
504      * Determines the version from maven's generated pom.properties file.
505      */
506     static {
507         String pmdVersion = null;
508         InputStream stream = PMD.class.getResourceAsStream("/META-INF/maven/net.sourceforge.pmd/pmd-core/pom.properties");
509         if (stream != null) {
510             try {
511                 Properties properties = new Properties();
512                 properties.load(stream);
513                 pmdVersion = properties.getProperty("version");
514             } catch (IOException e) {
515                 LOG.log(Level.FINE, "Couldn't determine version of PMD", e);
516             } finally {
517                 IOUtils.closeQuietly(stream);
518             }
519         }
520         if (pmdVersion == null) {
521             pmdVersion = "unknown";
522         }
523         VERSION = pmdVersion;
524     }
