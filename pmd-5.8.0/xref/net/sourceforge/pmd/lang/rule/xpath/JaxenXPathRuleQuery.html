38  /**
39   * This is a Jaxen based XPathRule query.
40   */
41  public class JaxenXPathRuleQuery extends AbstractXPathRuleQuery {
42  
43      private static final Logger LOG = Logger.getLogger(JaxenXPathRuleQuery.class.getName());
44  
45      private enum InitializationStatus {
46          NONE, PARTIAL, FULL
47      }
48  
49      // Mapping from Node name to applicable XPath queries
50      private InitializationStatus initializationStatus = InitializationStatus.NONE;
51      private Map<String, List<XPath>> nodeNameToXPaths;
52  
53      private static final String AST_ROOT = "_AST_ROOT_";
54  
55      /**
56       * {@inheritDoc}
57       */
58      @Override
59      public boolean isSupportedVersion(String version) {
60          return XPATH_1_0.equals(version);
61      }
62  
63      /**
64       * {@inheritDoc}
65       */
66      @Override
67      @SuppressWarnings("unchecked")
68      public List<Node> evaluate(Node node, RuleContext data) {
69          List<Node> results = new ArrayList<>();
70          try {
71              initializeXPathExpression(
72                      data.getLanguageVersion().getLanguageVersionHandler().getXPathHandler().getNavigator());
73              List<XPath> xpaths = nodeNameToXPaths.get(node.toString());
74              if (xpaths == null) {
75                  xpaths = nodeNameToXPaths.get(AST_ROOT);
76              }
77              for (XPath xpath : xpaths) {
78                  List<Node> nodes = xpath.selectNodes(node);
79                  results.addAll(nodes);
80              }
81          } catch (JaxenException ex) {
82              throw new RuntimeException(ex);
83          }
84          return results;
85      }
86  
87      /**
88       * {@inheritDoc}
89       */
90      @Override
91      public List<String> getRuleChainVisits() {
92          try {
93              // No Navigator available in this context
94              initializeXPathExpression(null);
95              return super.getRuleChainVisits();
96          } catch (JaxenException ex) {
97              throw new RuntimeException(ex);
98          }
99      }
100 
101     @SuppressWarnings("unchecked")
102     private void initializeXPathExpression(Navigator navigator) throws JaxenException {
103         if (initializationStatus == InitializationStatus.FULL) {
104             return;
105         } else if (initializationStatus == InitializationStatus.PARTIAL && navigator == null) {
106             if (LOG.isLoggable(Level.SEVERE)) {
107                 LOG.severe("XPathRule is not initialized because no navigator was provided. "
108                         + "Please make sure to implement getXPathHandler in the handler of the language. "
109                         + "See also AbstractLanguageVersionHandler.");
110             }
111             return;
112         }
113 
114         //
115         // Attempt to use the RuleChain with this XPath query. To do so, the
116         // queries
117         // should generally look like //TypeA or //TypeA | //TypeB. We will look
118         // at the
119         // parsed XPath AST using the Jaxen APIs to make this determination.
120         // If the query is not exactly what we are looking for, do not use the
121         // RuleChain.
122         //
123         nodeNameToXPaths = new HashMap<>();
124 
125         BaseXPath originalXPath = createXPath(xpath, navigator);
126         indexXPath(originalXPath, AST_ROOT);
127 
128         boolean useRuleChain = true;
129         Deque<Expr> pending = new ArrayDeque<>();
130         pending.push(originalXPath.getRootExpr());
131         while (!pending.isEmpty()) {
132             Expr node = pending.pop();
133 
134             // Need to prove we can handle this part of the query
135             boolean valid = false;
136 
137             // Must be a LocationPath... that is something like //Type
138             if (node instanceof LocationPath) {
139                 LocationPath locationPath = (LocationPath) node;
140                 if (locationPath.isAbsolute()) {
141                     // Should be at least two steps
142                     List<Step> steps = locationPath.getSteps();
143                     if (steps.size() >= 2) {
144                         Step step1 = steps.get(0);
145                         Step step2 = steps.get(1);
146                         // First step should be an AllNodeStep using the
147                         // descendant or self axis
148                         if (step1 instanceof AllNodeStep
149                                 && ((AllNodeStep) step1).getAxis() == Axis.DESCENDANT_OR_SELF) {
150                             // Second step should be a NameStep using the child
151                             // axis.
152                             if (step2 instanceof NameStep && ((NameStep) step2).getAxis() == Axis.CHILD) {
153                                 // Construct a new expression that is
154                                 // appropriate for RuleChain use
155                                 XPathFactory xpathFactory = new DefaultXPathFactory();
156 
157                                 // Instead of an absolute location path, we'll
158                                 // be using a relative path
159                                 LocationPath relativeLocationPath = xpathFactory.createRelativeLocationPath();
160                                 // The first step will be along the self axis
161                                 Step allNodeStep = xpathFactory.createAllNodeStep(Axis.SELF);
162                                 // Retain all predicates from the original name
163                                 // step
164                                 for (Iterator<Predicate> i = step2.getPredicates().iterator(); i.hasNext();) {
165                                     allNodeStep.addPredicate(i.next());
166                                 }
167                                 relativeLocationPath.addStep(allNodeStep);
168 
169                                 // Retain the remaining steps from the original
170                                 // location path
171                                 for (int i = 2; i < steps.size(); i++) {
172                                     relativeLocationPath.addStep(steps.get(i));
173                                 }
174 
175                                 BaseXPath xpath = createXPath(relativeLocationPath.getText(), navigator);
176                                 indexXPath(xpath, ((NameStep) step2).getLocalName());
177                                 valid = true;
178                             }
179                         }
180                     }
181                 }
182             } else if (node instanceof UnionExpr) { // Or a UnionExpr, that is
183                 // something like //TypeA |
184                 // //TypeB
185                 UnionExpr unionExpr = (UnionExpr) node;
186                 pending.push(unionExpr.getLHS());
187                 pending.push(unionExpr.getRHS());
188                 valid = true;
189             }
190             if (!valid) {
191                 useRuleChain = false;
192                 break;
193             }
194         }
195 
196         if (useRuleChain) {
197             // Use the RuleChain for all the nodes extracted from the xpath
198             // queries
199             super.ruleChainVisits.addAll(nodeNameToXPaths.keySet());
200         } else {
201             // Use original XPath if we cannot use the RuleChain
202             nodeNameToXPaths.clear();
203             indexXPath(originalXPath, AST_ROOT);
204             if (LOG.isLoggable(Level.FINE)) {
205                 LOG.log(Level.FINE, "Unable to use RuleChain for for XPath: " + xpath);
206             }
207         }
208 
209         if (navigator == null) {
210             this.initializationStatus = InitializationStatus.PARTIAL;
211             // Clear the node data, because we did not have a Navigator
212             nodeNameToXPaths = null;
213         } else {
214             this.initializationStatus = InitializationStatus.FULL;
215         }
216 
217     }
218 
219     private void indexXPath(XPath xpath, String nodeName) {
220         List<XPath> xpaths = nodeNameToXPaths.get(nodeName);
221         if (xpaths == null) {
222             xpaths = new ArrayList<>();
223             nodeNameToXPaths.put(nodeName, xpaths);
224         }
225         xpaths.add(xpath);
226     }
227 
228     private BaseXPath createXPath(String xpathQueryString, Navigator navigator) throws JaxenException {
229 
230         BaseXPath xpath = new BaseXPath(xpathQueryString, navigator);
231         if (properties.size() > 1) {
232             SimpleVariableContext vc = new SimpleVariableContext();
233             for (Entry<PropertyDescriptor<?>, Object> e : properties.entrySet()) {
234                 String propName = e.getKey().name();
235                 if (!"xpath".equals(propName)) {
236                     Object value = e.getValue();
237                     vc.setVariableValue(propName, value != null ? value.toString() : null);
238                 }
239             }
240             xpath.setVariableContext(vc);
241         }
242         return xpath;
243     }
