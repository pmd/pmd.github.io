5   package net.sourceforge.pmd.lang.rule.properties;
6   
7   import java.util.Map;
8   
9   import net.sourceforge.pmd.PropertyDescriptorFactory;
10  import net.sourceforge.pmd.lang.rule.properties.factories.BasicPropertyDescriptorFactory;
11  
12  /**
13   * Defines a datatype that supports multiple Integer property values within an
14   * upper and lower boundary.
15   *
16   * @author Brian Remedios
17   */
18  public class IntegerMultiProperty extends AbstractMultiNumericProperty<Integer[]> {
19  
20      public static final PropertyDescriptorFactory FACTORY = new BasicPropertyDescriptorFactory<IntegerMultiProperty>(
21              Integer[].class, NUMBER_FIELD_TYPES_BY_KEY) {
22  
23          @Override
24          public IntegerMultiProperty createWith(Map<String, String> valuesById) {
25              String[] minMax = minMaxFrom(valuesById);
26              char delimiter = delimiterIn(valuesById, DEFAULT_NUMERIC_DELIMITER);
27              Integer[] defaultValues = integersIn(numericDefaultValueIn(valuesById), delimiter);
28              return new IntegerMultiProperty(nameIn(valuesById), descriptionIn(valuesById), Integer.parseInt(minMax[0]),
29                      Integer.parseInt(minMax[1]), defaultValues, 0f);
30          }
31      };
32  
33      /**
34       * Constructor for IntegerProperty.
35       *
36       * @param theName
37       *            String
38       * @param theDescription
39       *            String
40       * @param min
41       *            Integer
42       * @param max
43       *            Integer
44       * @param theDefaults
45       *            Integer[]
46       * @param theUIOrder
47       *            float
48       * @throws IllegalArgumentException
49       */
50      public IntegerMultiProperty(String theName, String theDescription, Integer min, Integer max, Integer[] theDefaults,
51              float theUIOrder) {
52          super(theName, theDescription, min, max, theDefaults, theUIOrder);
53      }
54  
55      /**
56       * @return Class
57       * @see net.sourceforge.pmd.PropertyDescriptor#type()
58       */
59      @Override
60      public Class<Integer[]> type() {
61          return Integer[].class;
62      }
63  
64      /**
65       * @param value
66       *            String
67       * @return Object
68       */
69      @Override
70      protected Object createFrom(String value) {
71          return Integer.valueOf(value);
72      }
73  
74      /**
75       * @param size
76       *            int
77       * @return Object[]
78       */
79      @Override
80      protected Object[] arrayFor(int size) {
81          return new Integer[size];
82      }
83  }
