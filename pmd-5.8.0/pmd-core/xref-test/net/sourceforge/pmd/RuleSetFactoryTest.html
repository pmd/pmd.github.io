54      @Test
55      public void testExtendedReferences() throws Exception {
56          InputStream in = ResourceLoader.loadResourceAsStream("net/sourceforge/pmd/rulesets/reference-ruleset.xml",
57                  this.getClass().getClassLoader());
58          Assert.assertNotNull("Test ruleset not found - can't continue with test!", in);
59          in.close();
60  
61          RuleSetFactory rsf = new RuleSetFactory();
62          RuleSets rs = rsf.createRuleSets("net/sourceforge/pmd/rulesets/reference-ruleset.xml");
63          // added by referencing a complete ruleset (TestRuleset1.xml)
64          assertNotNull(rs.getRuleByName("MockRule1"));
65          assertNotNull(rs.getRuleByName("MockRule2"));
66          assertNotNull(rs.getRuleByName("MockRule3"));
67          assertNotNull(rs.getRuleByName("TestRuleRef"));
68  
69          // added by specific reference
70          assertNotNull(rs.getRuleByName("TestRule"));
71          // this is from TestRuleset2.xml, but not referenced
72          assertNull(rs.getRuleByName("TestRule2Ruleset2"));
73  
74          Rule mockRule3 = rs.getRuleByName("MockRule3");
75          assertEquals("Overridden message", mockRule3.getMessage());
76          assertEquals(2, mockRule3.getPriority().getPriority());
77  
78          Rule mockRule2 = rs.getRuleByName("MockRule2");
79          assertEquals("Just combine them!", mockRule2.getMessage());
80          // assert that MockRule2 is only once added to the ruleset, so that it
81          // really
82          // overwrites the configuration inherited from TestRuleset1.xml
83          assertEquals(1, countRule(rs, "MockRule2"));
84  
85          Rule mockRule1 = rs.getRuleByName("MockRule1");
86          assertNotNull(mockRule1);
87          PropertyDescriptor<?> prop = mockRule1.getPropertyDescriptor("testIntProperty");
88          Object property = mockRule1.getProperty(prop);
89          assertEquals("5", String.valueOf(property));
90  
91          // included from TestRuleset3.xml
92          assertNotNull(rs.getRuleByName("Ruleset3Rule2"));
93          // excluded from TestRuleset3.xml
94          assertNull(rs.getRuleByName("Ruleset3Rule1"));
95  
96          // overridden to 5
97          Rule ruleset4Rule1 = rs.getRuleByName("Ruleset4Rule1");
98          assertNotNull(ruleset4Rule1);
99          assertEquals(5, ruleset4Rule1.getPriority().getPriority());
100         assertEquals(1, countRule(rs, "Ruleset4Rule1"));
101         // priority overridden for whole TestRuleset4 group
102         Rule ruleset4Rule2 = rs.getRuleByName("Ruleset4Rule2");
103         assertNotNull(ruleset4Rule2);
104         assertEquals(2, ruleset4Rule2.getPriority().getPriority());
105     }

107     private int countRule(RuleSets rs, String ruleName) {
108         int count = 0;
109         for (Rule r : rs.getAllRules()) {
110             if (ruleName.equals(r.getName())) {
111                 count++;
112             }
113         }
114         return count;
115     }

868                     return null;

870             }
