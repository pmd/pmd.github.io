5   package net.sourceforge.pmd.junit;
6   
7   import java.util.logging.Logger;
8   import java.util.logging.StreamHandler;
9   
10  import org.apache.commons.io.output.ByteArrayOutputStream;
11  import org.junit.rules.ExternalResource;
12  
13  /**
14   * Junit Rule, to check for java util logging statements.
15   * 
16   * @author Andreas Dangel
17   * @see <a href="http://blog.diabol.se/?p=474">Testing the presence of log messages with java.util.logging</a>
18   */
19  public class JavaUtilLoggingRule extends ExternalResource {
20      private final Logger logger;
21      private final ByteArrayOutputStream stream;
22      private final StreamHandler customLogHandler;
23  
24  
25      /**
26       * Creates a new rule, that attaches a custom log handler
27       * to the given logger.
28       * @param loggerName the name of the logger to check
29       */
30      public JavaUtilLoggingRule(String loggerName) {
31          this.logger = Logger.getLogger(loggerName);
32          this.stream = new ByteArrayOutputStream();
33          this.customLogHandler = new StreamHandler(stream, logger.getParent().getHandlers()[0].getFormatter());
34      }
35  
36      @Override
37      protected void before() throws Throwable {
38          logger.addHandler(customLogHandler);
39      }
40  
41      @Override
42      protected void after() {
43          logger.removeHandler(customLogHandler);
44      }
45  
46      /**
47       * Gets the complete log.
48       * @return the log
49       */
50      public String getLog() {
51          customLogHandler.flush();
52          return stream.toString();
53      }
54  
55      /**
56       * Clears the log.
57       */
58      public void clear() {
59          customLogHandler.flush();
60          stream.reset();
61      }
62  }
