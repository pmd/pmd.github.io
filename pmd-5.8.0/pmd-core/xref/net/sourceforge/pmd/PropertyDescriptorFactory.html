5   package net.sourceforge.pmd;
6   
7   import java.util.Map;
8   
9   /**
10   * A factory to create {@link PropertyDescriptor}s based on a map of values.
11   *
12   * @author Brian Remedios
13   */
14  public interface PropertyDescriptorFactory {
15  
16      /**
17       * The type of the value of the {@link PropertyDescriptor} created by this
18       * factory.
19       *
20       * @return the type of the value.
21       */
22      Class<?> valueType();
23  
24      /**
25       * Denote the identifiers of the expected fields paired with booleans
26       * denoting whether they are required (non-null) or not.
27       *
28       * @return Map
29       */
30      Map<String, Boolean> expectedFields();
31  
32      /**
33       * Create a property descriptor of the appropriate type using the values
34       * provided.
35       *
36       * @param valuesById
37       *            the map of values
38       * @return a new and initialized {@link PropertyDescriptor}
39       */
40      PropertyDescriptor<?> createWith(Map<String, String> valuesById);
41  }
