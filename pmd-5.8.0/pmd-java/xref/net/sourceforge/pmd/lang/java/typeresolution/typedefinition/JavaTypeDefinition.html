5   package net.sourceforge.pmd.lang.java.typeresolution.typedefinition;
6   
7   import java.util.ArrayList;
8   import java.util.Collections;
9   import java.util.HashMap;
10  import java.util.List;
11  import java.util.Map;
12  
13  
14  public class JavaTypeDefinition implements TypeDefinition {
15      private final Class<?> clazz;
16      private List<JavaTypeDefinition> genericArgs;
17      // contains TypeDefs where only the clazz field is used
18      private static Map<Class<?>, JavaTypeDefinition> onlyClassTypeDef = new HashMap<>();
19  
20      public Class<?> getType() {
21          return clazz;
22      }
23  
24      public List<JavaTypeDefinition> getGenericArgs() {
25          if (genericArgs == null) {
26              genericArgs = Collections.unmodifiableList(new ArrayList<JavaTypeDefinition>());
27          }
28  
29          return genericArgs;
30      }
31  
32      private JavaTypeDefinition(Class<?> clazz, List<JavaTypeDefinition> genericArgs) {
33          this.clazz = clazz;
34  
35          if (genericArgs != null) {
36              this.genericArgs = Collections.unmodifiableList(genericArgs);
37          }
38      }
39  
40      //           builder part of the class
41  
42      public static JavaTypeDefinition build(Class<?> clazz) {
43          if (onlyClassTypeDef.containsKey(clazz)) {
44              return onlyClassTypeDef.get(clazz);
45          }
46  
47          JavaTypeDefinition typeDef = new JavaTypeDefinition(clazz, null);
48  
49          onlyClassTypeDef.put(clazz, typeDef);
50  
51          return typeDef;
52      }
53  
54      /**
55       * @param genericArgs This package private method expects that the genericArgs list has not been leaked,
56       *                    meaning the other references have been discarded to ensure immutability.
57       */
58      /* default */ static JavaTypeDefinition build(Class<?> clazz, List<JavaTypeDefinition> genericArgs) {
59          if (genericArgs == null) {
60              return build(clazz);
61          }
62          
63          return new JavaTypeDefinition(clazz, genericArgs);
64      }
65  
66      public static JavaTypeDefinitionBuilder builder(Class<?> clazz) {
67          return new JavaTypeDefinitionBuilder().setType(clazz);
68      }
69  
70  
71      public static JavaTypeDefinitionBuilder builder() {
72          return new JavaTypeDefinitionBuilder();
73      }
74  }
