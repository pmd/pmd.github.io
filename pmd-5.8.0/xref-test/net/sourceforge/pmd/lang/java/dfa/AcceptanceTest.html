31      @Test
32      public void testLabelledBreakLockup() {
33          getOrderedNodes(ASTMethodDeclarator.class, LABELLED_BREAK_LOCKUP);
34      }
35  
36      private static final String LABELLED_BREAK_LOCKUP = "class Foo {" + PMD.EOL + " void bar(int x) {" + PMD.EOL
37              + "  here: if (x>2) {" + PMD.EOL + "   break here;" + PMD.EOL + "  }" + PMD.EOL + " }" + PMD.EOL + "}";
38  
39      private boolean check(int[][] array, List<ASTMethodDeclarator> methodNodes) {
40          for (int i = 0; i < methodNodes.size(); i++) {
41              ASTMethodDeclarator decl = methodNodes.get(i);
42              DataFlowNode inode = decl.getDataFlowNode();
43              for (int j = 0; j < inode.getChildren().size(); j++) {
44                  DataFlowNode child = inode.getChildren().get(j);
45                  if (array[i][j] != child.getIndex() - 1) {
46                      return false;
47                  }
48              }
49          }
50          return true;
51      }
