5   package net.sourceforge.pmd.util;

7   import static org.junit.Assert.assertEquals;
8   
9   import org.junit.Test;
10  
11  public class StringUtilTest {
12  
13      @Test
14      public void testReplaceWithOneChar() {
15          assertEquals("faa", StringUtil.replaceString("foo", 'o', "a"));
16      }
17  
18      @Test
19      public void testReplaceWithMultipleChars() {
20          assertEquals("faaaa", StringUtil.replaceString("foo", 'o', "aa"));
21      }
22  
23      @Test
24      public void testReplaceStringWithString() {
25          assertEquals("foo]]&gt;bar", StringUtil.replaceString("foo]]>bar", "]]>", "]]&gt;"));
26      }
27  
28      @Test
29      public void testReplaceStringWithString2() {
30          assertEquals("replaceString didn't work with a >", "foobar",
31                  StringUtil.replaceString("foobar", "]]>", "]]&gt;"));
32      }
33  
34      @Test
35      public void testReplaceWithNull() {
36          assertEquals("replaceString didn't work with a char", "f", StringUtil.replaceString("foo", 'o', null));
37      }
38  
39      /**
40       * Usually you would set the system property
41       * "net.sourceforge.pmd.supportUTF8" to either "no" or "yes", to switch UTF8
42       * support.
43       *
44       * e.g.
45       * <code>System.setProperty("net.sourceforge.pmd.supportUTF8","yes");</code>
46       */
47      @Test
48      public void testUTF8NotSupported() {
49          StringBuilder sb = new StringBuilder();
50          String test = "é";
51          StringUtil.appendXmlEscaped(sb, test, false);
52          assertEquals("&#xe9;", sb.toString());
53      }
54  
55      @Test
56      public void testUTF8NotSupportedSurrogates() {
57          // D8 34 DD 1E -> U+1D11E
58          StringBuilder sb = new StringBuilder();
59          String test = new String(new char[] {0xd834, 0xdd1e});
60          StringUtil.appendXmlEscaped(sb, test, false);
61          assertEquals("&#x1d11e;", sb.toString());
62      }
63  
64      @Test
65      public void testUTF8Supported() {
66          StringBuilder sb = new StringBuilder();
67          String test = "é";
68          StringUtil.appendXmlEscaped(sb, test, true);
69          assertEquals("é", sb.toString());
70      }
71  
72      public static junit.framework.Test suite() {
73          return new junit.framework.JUnit4TestAdapter(StringUtilTest.class);
74      }
75  }
