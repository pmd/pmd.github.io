31      /**
32       * Creates all the links between the data flow nodes.
33       */
34      public void computePaths() throws LinkerException, SequenceException {
35          LOGGER.entering(CLASS_NAME,"computePaths");
36  	// Returns true if there are more sequences, computes the first and
37  	// the last index of the sequence.
38          LOGGER.fine("SequenceChecking continueBreakReturnStack elements");
39  	SequenceChecker sc = new SequenceChecker(braceStack);
40  	while (!sc.run()) {
41  		if (LOGGER.isLoggable(Level.FINE)) {
42            LOGGER.fine("After sc.run - starting Sequence checking loop with firstIndex=" + sc.getFirstIndex() 
43                        + ", lastIndex " + sc.getLastIndex() 
44                        +" with this StackList "+dump("braceStack",braceStack)
45                       );
46  		}
47  	    if (sc.getFirstIndex() < 0 || sc.getLastIndex() < 0) {
48  	        if (LOGGER.isLoggable(Level.SEVERE)) {
49                  LOGGER.severe("Sequence Checker problem: getFirstIndex()==" 
50                                + sc.getFirstIndex() + ", getLastIndex()==" + sc.getLastIndex() 
51                               );
52  	        }
53  		throw new SequenceException("computePaths(): return index <  0");
54  	    }
55  
56  	    StackObject firstStackObject = braceStack.get(sc.getFirstIndex());
57  
58  	    if (LOGGER.isLoggable(Level.FINE)) {
59              LOGGER.fine("Checking first braceStack element of type=="+ NodeType.stringFromType(firstStackObject.getType()) );
60  	    }
61  	    switch (firstStackObject.getType()) {
62  	    case NodeType.IF_EXPR:
63                  LOGGER.finest("IF_EXPR");
64  		int x = sc.getLastIndex() - sc.getFirstIndex();
65  		if (x == 2) {
66  		    this.computeIf(sc.getFirstIndex(), sc.getFirstIndex() + 1, sc.getLastIndex());
67  		} else if (x == 1) {
68  		    this.computeIf(sc.getFirstIndex(), sc.getLastIndex());
69  		} else {
70                      LOGGER.severe("Error - computePaths 1");
71  		}
72  		break;
73  
74  	    case NodeType.WHILE_EXPR:
75                  LOGGER.finest("WHILE_EXPR");
76  		this.computeWhile(sc.getFirstIndex(), sc.getLastIndex());
77  		break;
78  
79  	    case NodeType.SWITCH_START:
80                  LOGGER.finest("SWITCH_START");
81  		this.computeSwitch(sc.getFirstIndex(), sc.getLastIndex());
82  		break;
83  
84  	    case NodeType.FOR_INIT:
85  	    case NodeType.FOR_EXPR:
86  	    case NodeType.FOR_UPDATE:
87  	    case NodeType.FOR_BEFORE_FIRST_STATEMENT:
88                  LOGGER.finest("FOR_EXPR");
89  		this.computeFor(sc.getFirstIndex(), sc.getLastIndex());
90  		break;
91  
92  	    case NodeType.DO_BEFORE_FIRST_STATEMENT:
93                  LOGGER.finest("DO_BEFORE_FIRST_STATEMENT");
94  		this.computeDo(sc.getFirstIndex(), sc.getLastIndex());
95  		break;
96  
97  	    default:
98  	    }
99  
100 	    if (LOGGER.isLoggable(Level.FINEST)) {
101             LOGGER.finest("Removing braces from Last to first: " + sc.getLastIndex()  + " to " + sc.getFirstIndex());
102 	    }
103 	    for (int y = sc.getLastIndex(); y >= sc.getFirstIndex(); y--) {
104 	        if (LOGGER.isLoggable(Level.FINEST)) {
105                 LOGGER.finest("Removing brace : " + y );
106 	        }
107 		braceStack.remove(y);
108               }
109 	    if (LOGGER.isLoggable(Level.FINE)) {
110           LOGGER.fine("Completed Sequence checking loop" + braceStack  );
111 	    }
112 	}
113         if (LOGGER.isLoggable(Level.FINER))
114         {
115           LOGGER.log(Level.FINER, "After Sequence checking loop : remaining braces=={0}",
116               dump("braceStack", braceStack));
117         }
118 
119 
120         LOGGER.fine("Processing continueBreakReturnStack elements");
121 	while (!continueBreakReturnStack.isEmpty()) {
122           LOGGER.fine("Starting continueBreakReturnStack processing loop" );
123 	    StackObject stackObject = continueBreakReturnStack.get(0);
124 	    DataFlowNode node = stackObject.getDataFlowNode();
125 
126 	    switch (stackObject.getType()) {
127 	    case NodeType.THROW_STATEMENT:
128                 LOGGER.finest("CBR: THROW_STATEMENT");
129 		// do the same like a return
130 	    case NodeType.RETURN_STATEMENT:
131                 LOGGER.finest("CBR: RETURN_STATEMENT");
132 		// remove all children (should contain only one child)
133 		node.removePathToChild(node.getChildren().get(0));
134 		DataFlowNode lastNode = node.getFlow().get(node.getFlow().size() - 1);
135 		node.addPathToChild(lastNode);
136 		continueBreakReturnStack.remove(0);
137 		break;
138 	    case NodeType.BREAK_STATEMENT:
139                 LOGGER.finest("CBR: BREAK_STATEMENT");
140 		DataFlowNode last = getNodeToBreakStatement(node);
141 		node.removePathToChild(node.getChildren().get(0));
142 		node.addPathToChild(last);
143 		continueBreakReturnStack.remove(0);
144 		break;
145 
146 	    case NodeType.CONTINUE_STATEMENT:
147                 LOGGER.finest("CBR: CONTINUE_STATEMENT");
148 		//List cList = node.getFlow();
149 		/* traverse up the tree and find the first loop start node
150 		 */
151 		/*
152 		                               for(int i = cList.indexOf(node)-1;i>=0;i--) {
153 		                                   IDataFlowNode n = (IDataFlowNode)cList.get(i);
154 
155 		                                   if(n.isType(NodeType.FOR_UPDATE) ||
156 		                                               n.isType(NodeType.FOR_EXPR) ||
157 		                                               n.isType(NodeType.WHILE_EXPR)) {
158 		*/
159 		/*
160 		 * while(..) {
161 		 *              while(...) {
162 		 *                      ...
163 		 *              }
164 		 *              continue;
165 		 * }
166 		 *
167 		 * Without this Expression he continues the second
168 		 * WHILE loop. The continue statement and the while loop
169 		 * have to be in different scopes.
170 		 *
171 		 * TODO An error occurs if "continue" is even nested in
172 		 * scopes other than local loop scopes, like "if".
173 		 * The result is, that the data flow isn't build right
174 		 * and the pathfinder runs in invinity loop.
175 		 * */
176 		/*                                     if(n.getNode().getScope().equals(node.getNode().getScope())) {
177 		                                               System.err.println("equals");
178 		                                               continue;
179 		                                       }
180 		                                       else {
181 		                                               System.err.println("don't equals");
182 		                                       }
183 
184 		                                               //remove all children (should contain only one child)
185 		                                       node.removePathToChild((IDataFlowNode)node.getChildren().get(0));
186 
187 		                                       node.addPathToChild(n);
188 		                                       cbrStack.remove(0);
189 		                                       break;
190 
191 		                                   }else if(n.isType(NodeType.DO_BEFOR_FIRST_STATEMENT)) {
192 
193 		                                       IDataFlowNode inode = (IDataFlowNode)n.getFlow().get(n.getIndex()1);
194 
195 		                                       for(int j=0;j<inode.getParents().size();j) {
196 		                                               IDataFlowNode parent = (IDataFlowNode)inode.getParents().get(j);
197 
198 		                                               if(parent.isType(NodeType.DO_EXPR)) {
199 		                                                       node.removePathToChild((IDataFlowNode)node.getChildren().get(0));
200 		                                               node.addPathToChild(parent);
201 
202 		                                               cbrStack.remove(0);
203 		                                                       break;
204 		                                               }
205 		                                       }
206 		                                       break;
207 		                                   }
208 		                               }
209 		*/
210 		continueBreakReturnStack.remove(0); // delete this statement if you uncomment the stuff above
211 		break;
212 		default:
213                 LOGGER.finest("CBR: default");
214 		    // Do nothing
215 		    break;
216 	    }
217           LOGGER.fine("Completed continueBreakReturnStack processing loop" );
218 	}
219         LOGGER.exiting(CLASS_NAME,"computePaths");
220     }

222     private DataFlowNode getNodeToBreakStatement(DataFlowNode node) {
223         LOGGER.entering(CLASS_NAME,"getNodeToBreakStatement");
224 	// What about breaks to labels above if statements?
225 	List<DataFlowNode> bList = node.getFlow();
226 	int findEnds = 1; // ignore ends of other for's while's etc.
227 
228 	// find out index of the node where the path goes to after the break
229 	int index = bList.indexOf(node);
230 	for (; index < bList.size() - 2; index++) {
231 	    DataFlowNode n = bList.get(index);
232 	    if (n.isType(NodeType.DO_EXPR) || n.isType(NodeType.FOR_INIT) || n.isType(NodeType.WHILE_EXPR)
233 		    || n.isType(NodeType.SWITCH_START)) {
234 		findEnds++;
235 	    }
236 	    if (n.isType(NodeType.WHILE_LAST_STATEMENT) || n.isType(NodeType.SWITCH_END) || n.isType(NodeType.FOR_END)
237 		    || n.isType(NodeType.DO_EXPR)) {
238 		if (findEnds > 1) {
239 		    // thats not the right node
240 		    findEnds--;
241 		} else {
242 		    break;
243 		}
244 	    }
245 
246 	    if (n.isType(NodeType.LABEL_LAST_STATEMENT)) {
247 		Node parentNode = n.getNode().getFirstParentOfType(dataFlowHandler.getLabelStatementNodeClass());
248 		if (parentNode == null) {
249 		    break;
250 		} else {
251 		    String label = node.getNode().getImage();
252 		    if (label == null || label.equals(parentNode.getImage())) {
253 			node.removePathToChild(node.getChildren().get(0));
254 			DataFlowNode last = bList.get(index + 1);
255 			node.addPathToChild(last);
256 			break;
257 		    }
258 		}
259 	    }
260 	}
261         LOGGER.exiting(CLASS_NAME,"getNodeToBreakSttement");
262 	return node.getFlow().get(index + 1);
263     }

265     private void computeDo(int first, int last) {
266         LOGGER.entering(CLASS_NAME,"computeDo");
267 	DataFlowNode doSt = this.braceStack.get(first).getDataFlowNode();
268 	DataFlowNode doExpr = this.braceStack.get(last).getDataFlowNode();
269 	DataFlowNode doFirst = doSt.getFlow().get(doSt.getIndex() + 1);
270 	if (doFirst.getIndex() != doExpr.getIndex()) {
271 	    doExpr.addPathToChild(doFirst);
272 	}
273         LOGGER.exiting(CLASS_NAME,"computeDo");
274     }

276     private void computeFor(int firstIndex, int lastIndex) {
277         LOGGER.entering(CLASS_NAME,"computeFor");
278 	DataFlowNode fExpr = null;
279 	DataFlowNode fUpdate = null;
280 	DataFlowNode fSt = null;
281 	DataFlowNode fEnd = null;
282 	boolean isUpdate = false;
283 
284 	for (int i = firstIndex; i <= lastIndex; i++) {
285 	    StackObject so = this.braceStack.get(i);
286 	    DataFlowNode node = so.getDataFlowNode();
287 
288 	    if (so.getType() == NodeType.FOR_EXPR) {
289 		fExpr = node;
290 	    } else if (so.getType() == NodeType.FOR_UPDATE) {
291 		fUpdate = node;
292 		isUpdate = true;
293 	    } else if (so.getType() == NodeType.FOR_BEFORE_FIRST_STATEMENT) {
294 		fSt = node;
295 	    } else if (so.getType() == NodeType.FOR_END) {
296 		fEnd = node;
297 	    }
298 	}
299 	DataFlowNode end = fEnd.getFlow().get(fEnd.getIndex() + 1);
300 
301 	DataFlowNode firstSt = fSt.getChildren().get(0);
302 
303 	if (isUpdate) {
304 	    if (fSt.getIndex() != fEnd.getIndex()) {
305 		end.reverseParentPathsTo(fUpdate);
306 		fExpr.removePathToChild(fUpdate);
307 		fUpdate.addPathToChild(fExpr);
308 		fUpdate.removePathToChild(firstSt);
309 		fExpr.addPathToChild(firstSt);
310 		fExpr.addPathToChild(end);
311 	    } else {
312 		fSt.removePathToChild(end);
313 		fExpr.removePathToChild(fUpdate);
314 		fUpdate.addPathToChild(fExpr);
315 		fExpr.addPathToChild(fUpdate);
316 		fExpr.addPathToChild(end);
317 	    }
318 	} else {
319 	    if (fSt.getIndex() != fEnd.getIndex()) {
320 		end.reverseParentPathsTo(fExpr);
321 		fExpr.addPathToChild(end);
322 	    }
323 	}
324         LOGGER.exiting(CLASS_NAME,"computeFor");
325     }

327     private void computeSwitch(int firstIndex, int lastIndex) {
328         LOGGER.entering(CLASS_NAME,"computeSwitch" );
329 
330 	int diff = lastIndex - firstIndex;
331 	boolean defaultStatement = false;
332 
333 	DataFlowNode sStart = this.braceStack.get(firstIndex).getDataFlowNode();
334 	DataFlowNode sEnd = this.braceStack.get(lastIndex).getDataFlowNode();
335 	DataFlowNode end = sEnd.getChildren().get(0);
336 
337 	if (LOGGER.isLoggable(Level.FINE)) {
338         LOGGER.fine(
339                      "Stack(sStart)=>" + sStart
340                      +",Stack(sEnd)=>" + sEnd
341                      +",end=>" + end
342                    );
343 	}
344 
345 	for (int i = 0; i < diff - 2; i++) {
346 	    StackObject so = this.braceStack.get(firstIndex + 2 + i);
347 	    DataFlowNode node = so.getDataFlowNode();
348 
349 	    if (LOGGER.isLoggable(Level.FINE)) {
350             LOGGER.fine( "so(" + (firstIndex + 2 + i) + ")=>" + so 
351                          +" has  dfn=>" + node 
352                          +" with first child =>" + node.getChildren().get(0)
353                        );
354 	    }
355 	    sStart.addPathToChild(node.getChildren().get(0));
356 
357 	    if (so.getType() == NodeType.SWITCH_LAST_DEFAULT_STATEMENT) {
358 		defaultStatement = true;
359 	    }
360 	}
361 
362 	if (!defaultStatement) {
363 	    sStart.addPathToChild(end);
364 	}
365         LOGGER.exiting(CLASS_NAME,"computeSwitch");
366     }

368     private void computeWhile(int first, int last) {
369         LOGGER.entering(CLASS_NAME,"computeWhile with first and last of" 
370                         + first + "," + last
371                         );
372 	DataFlowNode wStart = this.braceStack.get(first).getDataFlowNode();
373 	DataFlowNode wEnd = this.braceStack.get(last).getDataFlowNode();
374 
375 	DataFlowNode end = wEnd.getFlow().get(wEnd.getIndex() + 1);
376 
377 	if (wStart.getIndex() != wEnd.getIndex()) {
378 	    end.reverseParentPathsTo(wStart);
379 	    wStart.addPathToChild(end);
380 	}
381         LOGGER.exiting(CLASS_NAME,"computeWhile");
382     }

384     private void computeIf(int first, int second, int last) {
385         LOGGER.entering(CLASS_NAME,"computeIf(3)",first+","+ second +","+ last);
386 	DataFlowNode ifStart = this.braceStack.get(first).getDataFlowNode();
387 	DataFlowNode ifEnd = this.braceStack.get(second).getDataFlowNode();
388 	DataFlowNode elseEnd = this.braceStack.get(last).getDataFlowNode();
389 
390 	DataFlowNode elseStart = ifEnd.getFlow().get(ifEnd.getIndex() + 1);
391 	DataFlowNode end = elseEnd.getFlow().get(elseEnd.getIndex() + 1);
392 
393         LOGGER.log(Level.FINEST, "If ifstart={0}, ifEnd={1}, elseEnd={2}, elseStart={3}, end={4}"
394                    , new Object[]{ifStart, ifEnd, elseEnd, elseStart, end}
395                   );
396 
397 	// if if-statement and else-statement contains statements or expressions
398 	if (ifStart.getIndex() != ifEnd.getIndex() && ifEnd.getIndex() != elseEnd.getIndex()) {
399 	    elseStart.reverseParentPathsTo(end);
400 	    ifStart.addPathToChild(elseStart);
401 	}
402 	// if only if-statement contains no expressions
403 	else if (ifStart.getIndex() == ifEnd.getIndex() && ifEnd.getIndex() != elseEnd.getIndex()) {
404 	    ifStart.addPathToChild(end);
405 	}
406 	// if only else-statement contains no expressions
407 	else if (ifEnd.getIndex() == elseEnd.getIndex() && ifStart.getIndex() != ifEnd.getIndex()) {
408 	    ifStart.addPathToChild(end);
409 	}
410         LOGGER.exiting(CLASS_NAME,"computeIf(3)");
411     }

413     private void computeIf(int first, int last) {
414         LOGGER.entering(CLASS_NAME,"computeIf(2)");
415 	DataFlowNode ifStart = this.braceStack.get(first).getDataFlowNode();
416 	DataFlowNode ifEnd = this.braceStack.get(last).getDataFlowNode();
417 
418 	// only if the if-statement contains another Statement or Expression
419 	if (ifStart.getIndex() != ifEnd.getIndex()) {
420 	    DataFlowNode end = ifEnd.getFlow().get(ifEnd.getIndex() + 1);
421 	    ifStart.addPathToChild(end);
422 	}
423         LOGGER.exiting(CLASS_NAME,"computeIf(2)");
424     }
425     /**
426      * 
427      * @return formatted dump of the StackList
428      */
429     private static String dump(String description, List<StackObject>  stackList) {
430       StringBuilder stringDump = new StringBuilder("Stack List (");
431       stringDump.append(description).append(") ListDump:\n");
432       for (StackObject stackObject  : stackList )
433       {
434 	stringDump.append ('\n').append(stackObject.toString());
435       }
436       return stringDump.toString();
437     }
