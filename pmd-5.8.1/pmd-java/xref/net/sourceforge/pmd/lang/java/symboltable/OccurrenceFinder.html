5   package net.sourceforge.pmd.lang.java.symboltable;
6   
7   import java.util.HashSet;
8   import java.util.List;
9   import java.util.Set;
10  
11  import net.sourceforge.pmd.lang.java.ast.ASTPrimaryExpression;
12  import net.sourceforge.pmd.lang.java.ast.JavaParserVisitorAdapter;
13  import net.sourceforge.pmd.lang.symboltable.NameDeclaration;
14  import net.sourceforge.pmd.lang.symboltable.Scope;
15  
16  public class OccurrenceFinder extends JavaParserVisitorAdapter {
17  
18      // Maybe do some sort of State pattern thingy for when NameDeclaration
19      // is empty/not empty?
20      private final Set<NameDeclaration> declarations = new HashSet<>();
21  
22      private final Set<NameDeclaration> additionalDeclarations = new HashSet<>();
23  
24      public Object visit(ASTPrimaryExpression node, Object data) {
25          NameFinder nameFinder = new NameFinder(node);
26  
27          declarations.clear();
28          additionalDeclarations.clear();
29  
30          List<JavaNameOccurrence> names = nameFinder.getNames();
31          for (JavaNameOccurrence occ : names) {
32              Search search = new Search(occ);
33              if (declarations.isEmpty()) {
34                  // doing the first name lookup
35                  search.execute();
36                  declarations.addAll(search.getResult());
37                  if (declarations.isEmpty()) {
38                      // we can't find it, so just give up
39                      // when we decide to do full symbol resolution
40                      // force this to either find a symbol or throw a
41                      // SymbolNotFoundException
42                      break;
43                  }
44              } else {
45                  for (NameDeclaration decl : declarations) {
46                      // now we've got a scope we're starting with, so work from
47                      // there
48                      Scope startingScope = decl.getScope();
49                      // in case the previous found declaration is a class
50                      // reference
51                      // for a class inside the same source file
52                      // we need to search this class
53                      // e.g. the list of name occurrence could come from
54                      // outerClassRef.member. See also bug #1302
55                      if (decl instanceof VariableNameDeclaration) {
56                          String typeImage = ((VariableNameDeclaration) decl).getTypeImage();
57                          ClassNameDeclaration clazzDeclaration = startingScope.getEnclosingScope(SourceFileScope.class)
58                                  .findClassNameDeclaration(typeImage);
59                          if (clazzDeclaration != null) {
60                              startingScope = clazzDeclaration.getScope();
61                          }
62                      }
63                      search.execute(startingScope);
64                      Set<NameDeclaration> result = search.getResult();
65                      additionalDeclarations.addAll(result);
66                      if (result.isEmpty()) {
67                          // nothing found
68                          // This seems to be a lack of type resolution here.
69                          // Theoretically we have the previous declaration node
70                          // and
71                          // know from there the Type of
72                          // the variable. The current occurrence (occ) should
73                          // then be
74                          // found in the declaration of
75                          // this type. The type however may or may not be known
76                          // to
77                          // PMD (see aux classpath).
78  
79                          // we can't find it, so just give up
80                          // when we decide to do full symbol resolution
81                          // force this to either find a symbol or throw a
82                          // SymbolNotFoundException
83                          break;
84                      }
85                  }
86                  declarations.addAll(additionalDeclarations);
87              }
88          }
89          return super.visit(node, data);
90      }
91  
92  }
