5   package net.sourceforge.pmd.lang.java.ast;

7   import net.sourceforge.pmd.lang.java.typeresolution.typedefinition.JavaTypeDefinition;
8   
9   public abstract class AbstractJavaAccessTypeNode extends AbstractJavaAccessNode implements TypeNode {
10      private JavaTypeDefinition typeDefinition;
11  
12      public AbstractJavaAccessTypeNode(int i) {
13          super(i);
14      }
15  
16      public AbstractJavaAccessTypeNode(JavaParser parser, int i) {
17          super(parser, i);
18      }
19  
20      @Override
21      public Class<?> getType() {
22          if (typeDefinition != null) {
23              return typeDefinition.getType();
24          }
25  
26          return null;
27      }
28  
29      @Override
30      public void setType(Class<?> type) {
31          typeDefinition = JavaTypeDefinition.forClass(type);
32      }
33  
34      @Override
35      public JavaTypeDefinition getTypeDefinition() {
36          return typeDefinition;
37      }
38  
39      @Override
40      public void setTypeDefinition(JavaTypeDefinition typeDefinition) {
41          this.typeDefinition = typeDefinition;
42      }
43  }
