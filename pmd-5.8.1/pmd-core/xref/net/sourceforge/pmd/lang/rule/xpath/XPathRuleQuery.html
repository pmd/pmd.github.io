5   package net.sourceforge.pmd.lang.rule.xpath;
6   
7   import java.util.List;
8   import java.util.Map;
9   
10  import net.sourceforge.pmd.PropertyDescriptor;
11  import net.sourceforge.pmd.RuleContext;
12  import net.sourceforge.pmd.lang.ast.Node;
13  
14  /**
15   * This interface captures the logic needed by XPathRule to implement an XPath
16   * based query on an AST Node.
17   * <p>
18   * Implementations of this class do not need to be thread-safe, but they will be
19   * reused to query against different AST Nodes. Therefore, internal state should
20   * be maintained in a fashion consistent with reuse. Further, implementations
21   * are recommended to manage internal state that is invariant over AST Nodes in
22   * a fashion which facilities high performance (e.g. caching).
23   */
24  public interface XPathRuleQuery {
25  
26      /**
27       * XPath 1.0 version.
28       */
29      String XPATH_1_0 = "1.0";
30  
31      /**
32       * XPath 1.0 compatibility version.
33       */
34      String XPATH_1_0_COMPATIBILITY = "1.0 compatibility";
35  
36      /**
37       * XPath 2.0 version.
38       */
39      String XPATH_2_0 = "2.0";
40  
41      /**
42       * Set the XPath query string to be used.
43       *
44       * @param xpath
45       *            The XPath query string.
46       */
47      void setXPath(String xpath);
48  
49      /**
50       * Set the XPath version to be used.
51       *
52       * @param version
53       *            The XPath version.
54       * @throws UnsupportedOperationException
55       *             if the version cannot be handled.
56       */
57      void setVersion(String version) throws UnsupportedOperationException;
58  
59      /**
60       * Set the properties to use during the XPath query.
61       */
62      void setProperties(Map<PropertyDescriptor<?>, Object> properties);
63  
64      /**
65       * Indicates which AST Nodes (if any) should be used with the RuleChain. Use
66       * of the RuleChain will allow the query execute on a targed sub-tree of the
67       * AST, instead of the entire AST from the root. This can result in great
68       * performance benefits.
69       */
70      List<String> getRuleChainVisits();
71  
72      /**
73       * Evaluate the XPath query against the given Node.
74       *
75       * @param node
76       *            The Node.
77       * @param data
78       *            The RuleContext.
79       * @return The matching Nodes.
80       */
81      List<Node> evaluate(Node node, RuleContext data);
82  }
