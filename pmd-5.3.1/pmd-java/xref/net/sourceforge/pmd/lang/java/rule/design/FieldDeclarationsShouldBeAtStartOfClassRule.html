6   import net.sourceforge.pmd.lang.ast.Node;
7   import net.sourceforge.pmd.lang.java.ast.ASTAnnotationTypeDeclaration;
8   import net.sourceforge.pmd.lang.java.ast.ASTClassOrInterfaceBodyDeclaration;
9   import net.sourceforge.pmd.lang.java.ast.ASTClassOrInterfaceDeclaration;
10  import net.sourceforge.pmd.lang.java.ast.ASTConstructorDeclaration;
11  import net.sourceforge.pmd.lang.java.ast.ASTEnumDeclaration;
12  import net.sourceforge.pmd.lang.java.ast.ASTFieldDeclaration;
13  import net.sourceforge.pmd.lang.java.ast.ASTMethodDeclaration;
14  import net.sourceforge.pmd.lang.java.rule.AbstractJavaRule;
15  import net.sourceforge.pmd.lang.rule.properties.BooleanProperty;

17  /**
18   * Detects fields that are declared after methods, constructors, etc.
19   * It was a XPath rule, but the Java version is much faster.
20   * The XPath rule for reference:
21   * <pre>
22  //ClassOrInterfaceBody/ClassOrInterfaceBodyDeclaration/FieldDeclaration
23  [not(.//ClassOrInterfaceBodyDeclaration) or $ignoreAnonymousClassDeclarations = 'false']
24  [../preceding-sibling::ClassOrInterfaceBodyDeclaration
25      [  count(ClassOrInterfaceDeclaration) > 0
26      or count(ConstructorDeclaration) > 0
27      or count(MethodDeclaration) > 0
28      or count(AnnotationTypeDeclaration) > 0
29      or ($ignoreEnumDeclarations = 'false' and count(EnumDeclaration) > 0)
30      ]
31  ]
32   * </pre>
33   */
34  public class FieldDeclarationsShouldBeAtStartOfClassRule extends AbstractJavaRule {
35  
36      private BooleanProperty ignoreEnumDeclarations = new BooleanProperty("ignoreEnumDeclarations", "Ignore Enum Declarations that precede fields.", true, 1.0f);
37      private BooleanProperty ignoreAnonymousClassDeclarations = new BooleanProperty("ignoreAnonymousClassDeclarations", "Ignore Field Declarations, that are initialized with anonymous class declarations", true, 2.0f);
38  
39      /**
40       * Initializes the rule {@link FieldDeclarationsShouldBeAtStartOfClassRule}.
41       */
42      public FieldDeclarationsShouldBeAtStartOfClassRule() {
43          definePropertyDescriptor(ignoreEnumDeclarations);
44          definePropertyDescriptor(ignoreAnonymousClassDeclarations);
45      }
46  
47      @Override
48      public Object visit(ASTFieldDeclaration node, Object data) {
49          Node parent = node.jjtGetParent().jjtGetParent();
50          for (int i = 0; i < parent.jjtGetNumChildren(); i++) {
51              Node child = parent.jjtGetChild(i);
52              if (child.jjtGetNumChildren() > 0) {
53                  child = child.jjtGetChild(0);
54              }
55              if (child.equals(node)) {
56                  break;
57              }
58              if (child instanceof ASTFieldDeclaration) {
59                  continue;
60              }
61              if (node.hasDescendantOfType(ASTClassOrInterfaceBodyDeclaration.class) && getProperty(ignoreAnonymousClassDeclarations).booleanValue()) {
62                  continue;
63              }
64              if (child instanceof ASTClassOrInterfaceDeclaration
65                      || child instanceof ASTMethodDeclaration
66                      || child instanceof ASTConstructorDeclaration
67                      || child instanceof ASTAnnotationTypeDeclaration) {
68                  addViolation(data, node);
69                  break;
70              }
71              if (child instanceof ASTEnumDeclaration && !getProperty(ignoreEnumDeclarations).booleanValue()) {
72                  addViolation(data, node);
73                  break;
74              }
75          }
76          return data;
77      }
78  }
