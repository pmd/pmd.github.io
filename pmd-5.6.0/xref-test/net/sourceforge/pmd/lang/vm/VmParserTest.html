56      private Node parse(final String code) {
57          final LanguageVersionHandler vmLang = LanguageRegistry.getLanguage(VmLanguageModule.NAME).getDefaultVersion()
58                  .getLanguageVersionHandler();
59          final Parser parser = vmLang.getParser(vmLang.getDefaultParserOptions());
60          final Node node = parser.parse(null, new StringReader(code));
61          return node;
62      }
