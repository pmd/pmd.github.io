42      @Test
43      public void visibilityTest() {
44          final String TEST = "class Bzaz{ "
45              + "public int bar;"
46              + "String k;"
47              + "protected double d;"
48              + "private int i;"
49              + "protected int x;"
50              + "public Bzaz(){} "
51              + "void bar(){} "
52              + "protected void foo(int x){}"
53              + "private Bzaz(int y){}"
54              + "}";
55  
56  
57          List<ASTMethodOrConstructorDeclaration> operationDeclarations = getOrderedNodes(ASTMethodOrConstructorDeclaration.class, TEST);
58          List<ASTFieldDeclaration> fieldDeclarations = getOrderedNodes(ASTFieldDeclaration.class, TEST);
59          List<Signature> sigs = new ArrayList<>();
60  
61          for (ASTMethodOrConstructorDeclaration node : operationDeclarations) {
62              sigs.add(OperationSignature.buildFor(node));
63          }
64  
65          // operations
66          assertEquals(Visibility.PUBLIC, sigs.get(0).visibility);
67          assertEquals(Visibility.PACKAGE, sigs.get(1).visibility);
68          assertEquals(Visibility.PROTECTED, sigs.get(2).visibility);
69          assertEquals(Visibility.PRIVATE, sigs.get(3).visibility);
70  
71          sigs.clear();
72          for (ASTFieldDeclaration node : fieldDeclarations) {
73              sigs.add(FieldSignature.buildFor(node));
74          }
75  
76          // fields
77          assertEquals(Visibility.PUBLIC, sigs.get(0).visibility);
78          assertEquals(Visibility.PACKAGE, sigs.get(1).visibility);
79          assertEquals(Visibility.PROTECTED, sigs.get(2).visibility);
80          assertEquals(Visibility.PRIVATE, sigs.get(3).visibility);
81      }

84      public void operationRoleTest() {
85          final String TEST = "class Bzaz{ int x; "
86              + "public static void foo(){} "
87              + "Bzaz(){} "
88              + "int getX(){return x;}"
89              + " void setX(int a){x=a;}"
90              + " public void doSomething(){}}";
91  
92  
93          List<ASTMethodOrConstructorDeclaration> nodes = getOrderedNodes(ASTMethodOrConstructorDeclaration.class, TEST);
94          List<OperationSignature> sigs = new ArrayList<>();
95  
96          for (ASTMethodOrConstructorDeclaration node : nodes) {
97              sigs.add(OperationSignature.buildFor(node));
98          }
99  
100         assertEquals(Role.STATIC, sigs.get(0).role);
101         assertEquals(Role.CONSTRUCTOR, sigs.get(1).role);
102         assertEquals(Role.GETTER_OR_SETTER, sigs.get(2).role);
103         assertEquals(Role.GETTER_OR_SETTER, sigs.get(3).role);
104         assertEquals(Role.METHOD, sigs.get(4).role);

209     @Test
210     public void operationPoolTest() {
211         final String TEST = "class Bzaz{ "
212             + "public static void foo(){} "
213             + "public static void az(){} "
214             + "public static int getX(){return x;}}";
215 
216         final String TEST2 = "class Bzaz{ "
217             + "void foo(){} "
218             + "void az(){} "
219             + "int rand(){return x;}}";
220 
221 
222         List<ASTMethodOrConstructorDeclaration> nodes = getOrderedNodes(ASTMethodOrConstructorDeclaration.class, TEST);
223         List<ASTMethodOrConstructorDeclaration> nodes2 = getOrderedNodes(ASTMethodOrConstructorDeclaration.class, TEST2);
224 
225         List<OperationSignature> sigs = new ArrayList<>();
226         List<OperationSignature> sigs2 = new ArrayList<>();
227 
228         for (int i = 0; i < sigs.size(); i++) {
229             sigs.add(OperationSignature.buildFor(nodes.get(i)));
230             sigs2.add(OperationSignature.buildFor(nodes2.get(i)));
231 
232         }
233 
234         for (int i = 0; i < sigs.size() - 1; i++) {
235             assertTrue(sigs.get(i) == sigs.get(i + 1));
236             assertTrue(sigs2.get(i) == sigs2.get(i + 1));
237         }
238     }

274     private ASTCompilationUnit parseClass(Class<?> clazz) {
275         String sourceFile = clazz.getName().replace('.', '/') + ".java";
276         InputStream is = ClassTypeResolverTest.class.getClassLoader().getResourceAsStream(sourceFile);
277         if (is == null) {
278             throw new IllegalArgumentException(
279                 "Unable to find source file " + sourceFile + " for " + clazz);
280         }
281         String source;
282         try {
283             source = IOUtils.toString(is);
284         } catch (IOException e) {
285             throw new RuntimeException(e);
286         }
287         return parseJava17(source);
288     }
