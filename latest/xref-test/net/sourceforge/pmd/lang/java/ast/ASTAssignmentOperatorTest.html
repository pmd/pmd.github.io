14  import net.sourceforge.pmd.PMD;

17  public class ASTAssignmentOperatorTest extends ParserTst {
18  
19      @Test
20      public void testSimpleAssignmentRecognized() {
21          Set<ASTAssignmentOperator> ops = super.getNodes(ASTAssignmentOperator.class, TEST1);
22          assertFalse((ops.iterator().next()).isCompound());
23      }
24  
25      @Test
26      public void testCompoundAssignmentPlusRecognized() {
27          Set<ASTAssignmentOperator> ops = super.getNodes(ASTAssignmentOperator.class, TEST2);
28          assertTrue((ops.iterator().next()).isCompound());
29      }
30  
31      @Test
32      public void testCompoundAssignmentMultRecognized() {
33          Set<ASTAssignmentOperator> ops = super.getNodes(ASTAssignmentOperator.class, TEST3);
34          assertTrue((ops.iterator().next()).isCompound());
35      }
36  
37      private static final String TEST1 = "public class Foo {" + PMD.EOL + " void bar() {" + PMD.EOL + "  int x;"
38              + PMD.EOL + "  x=2;" + PMD.EOL + " }" + PMD.EOL + "}";
39  
40      private static final String TEST2 = "public class Foo {" + PMD.EOL + " void bar() {" + PMD.EOL + "  int x;"
41              + PMD.EOL + "  x += 2;" + PMD.EOL + " }" + PMD.EOL + "}";
42  
43      private static final String TEST3 = "public class Foo {" + PMD.EOL + " void bar() {" + PMD.EOL + "  int x;"
44              + PMD.EOL + "  x *= 2;" + PMD.EOL + " }" + PMD.EOL + "}";
45  }
