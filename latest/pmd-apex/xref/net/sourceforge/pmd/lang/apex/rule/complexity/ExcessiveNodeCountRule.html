5   package net.sourceforge.pmd.lang.apex.rule.complexity;
6   
7   import net.sourceforge.pmd.lang.apex.ast.ApexNode;
8   import net.sourceforge.pmd.lang.apex.rule.AbstractStatisticalApexRule;
9   import net.sourceforge.pmd.stat.DataPoint;
10  
11  /**
12   * This is a common super class for things which shouldn't have excessive nodes
13   * underneath.
14   *
15   * <p>
16   * It expects all "visit" calls to return an Integer. It will sum all the values
17   * it gets, and use that as its score.
18   * </p>
19   *
20   * <p>
21   * To use it, override the "visit" for the nodes that need to be counted. On
22   * those return "new Integer(1)"
23   * </p>
24   *
25   * <p>
26   * All others will return 0 (or the sum of counted nodes underneath.)
27   * </p>
28   */
29  
30  public class ExcessiveNodeCountRule extends AbstractStatisticalApexRule {
31      private Class<?> nodeClass;
32  
33      public ExcessiveNodeCountRule(Class<?> nodeClass) {
34          this.nodeClass = nodeClass;
35      }
36  
37      @Override
38      public Object visit(ApexNode<?> node, Object data) {
39          int numNodes = 0;
40  
41          for (int i = 0; i < node.jjtGetNumChildren(); i++) {
42              Integer treeSize = (Integer) ((ApexNode<?>) node.jjtGetChild(i)).jjtAccept(this, data);
43              numNodes += treeSize;
44          }
45  
46          if (nodeClass.isInstance(node)) {
47              DataPoint point = new DataPoint();
48              point.setNode(node);
49              point.setScore(1.0 * numNodes);
50              point.setMessage(getMessage());
51              addDataPoint(point);
52          }
53  
54          return Integer.valueOf(numNodes);
55      }
56  }
