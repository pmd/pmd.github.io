80      public ApexCRUDViolationRule() {
81          setProperty(CODECLIMATE_CATEGORIES, new String[] { "Security" });
82          setProperty(CODECLIMATE_REMEDIATION_MULTIPLIER, 100);
83          setProperty(CODECLIMATE_BLOCK_HIGHLIGHTING, false);

86      @Override
87      public Object visit(ASTUserClass node, Object data) {
88          for (ASTMethod n : node.findDescendantsOfType(ASTMethod.class)) {
89              StringBuilder sb = new StringBuilder().append(n.getNode().getDefiningType().getApexName()).append(":")
90                      .append(n.getNode().getMethodInfo().getCanonicalName()).append(":")
91                      .append(n.getNode().getMethodInfo().getParameterTypes().size());
92              classMethods.put(sb.toString(), n);
93          }
94  
95          node.childrenAccept(this, data);
96          return data;
97      }

181     private void addVariableToMapping(final String variableName, final String type) {
182         varToTypeMapping.put(variableName, getSimpleType(type));
183     }

185     private String getSimpleType(final String type) {
186         String typeToUse = type;
187 
188         Pattern pattern = Pattern.compile("^[list<]?list<(\\S+?)>[>]?$", Pattern.CASE_INSENSITIVE);
189         Matcher matcher = pattern.matcher(typeToUse);
190 
191         if (matcher.find()) {
192             typeToUse = matcher.group(1);
193         }
194         return typeToUse;
195     }

209     private void collectCRUDMethodLevelChecks(final ASTMethodCallExpression node) {
210         final String method = node.getNode().getMethodName();
211         final ASTReferenceExpression ref = node.getFirstChildOfType(ASTReferenceExpression.class);
212         if (ref == null) {
213             return;
214         }
215 
216         List<Identifier> a = ref.getNode().getJadtIdentifiers();
217         if (!a.isEmpty()) {
218             extractObjectAndFields(a, method, node.getNode().getDefiningType().getApexName());
219         } else {
220             // see if ESAPI
221             if (Helper.isMethodCallChain(node, ESAPI_ISAUTHORIZED_TO_VIEW)) {
222                 extractObjectTypeFromESAPI(node, IS_ACCESSIBLE);
223             }
224 
225             if (Helper.isMethodCallChain(node, ESAPI_ISAUTHORIZED_TO_CREATE)) {
226                 extractObjectTypeFromESAPI(node, IS_CREATEABLE);
227             }
228 
229             if (Helper.isMethodCallChain(node, ESAPI_ISAUTHORIZED_TO_UPDATE)) {
230                 extractObjectTypeFromESAPI(node, IS_UPDATEABLE);
231             }
232 
233             if (Helper.isMethodCallChain(node, ESAPI_ISAUTHORIZED_TO_DELETE)) {
234                 extractObjectTypeFromESAPI(node, IS_DELETABLE);
235             }
236 
237             // see if getDescribe()
238             final ASTDottedExpression dottedExpr = ref.getFirstChildOfType(ASTDottedExpression.class);
239             if (dottedExpr != null) {
240                 final ASTMethodCallExpression nestedMethodCall = dottedExpr
241                         .getFirstChildOfType(ASTMethodCallExpression.class);
242                 if (nestedMethodCall != null) {
243                     if (isLastMethodName(nestedMethodCall, S_OBJECT_TYPE, GET_DESCRIBE)) {
244                         String resolvedType = getType(nestedMethodCall);
245                         typeToDMLOperationMapping.put(resolvedType, method);
246                     }
247                 }
248             }
249 
250         }
251     }

253     private boolean isLastMethodName(final ASTMethodCallExpression methodNode, final String className,
254             final String methodName) {
255         final ASTReferenceExpression reference = methodNode.getFirstChildOfType(ASTReferenceExpression.class);
256         if (reference.getNode().getJadtIdentifiers().size() > 0) {
257             if (reference.getNode().getJadtIdentifiers().get(reference.getNode().getJadtIdentifiers().size() - 1).value
258                     .equalsIgnoreCase(className) && Helper.isMethodName(methodNode, methodName)) {
259                 return true;
260             }
261         }
262 
263         return false;
264     }

266     private String getType(final ASTMethodCallExpression methodNode) {
267         final ASTReferenceExpression reference = methodNode.getFirstChildOfType(ASTReferenceExpression.class);
268         if (reference.getNode().getJadtIdentifiers().size() > 0) {
269             return new StringBuilder().append(reference.getNode().getDefiningType().getApexName()).append(":")
270                     .append(reference.getNode().getJadtIdentifiers().get(0).value).toString();
271         }
272         return "";
273     }

275     private void extractObjectAndFields(final List<Identifier> listIdentifiers, final String method,
276             final String definingType) {
277         final List<String> strings = listIdentifiers.stream().map(id -> id.value).collect(Collectors.toList());
278 
279         int flsIndex = Collections.lastIndexOfSubList(strings, Arrays.asList(RESERVED_KEYS_FLS));
280         if (flsIndex != -1) {
281             String objectTypeName = strings.get(flsIndex + RESERVED_KEYS_FLS.length);
282             typeToDMLOperationMapping.put(definingType + ":" + objectTypeName, method);
283         }
284     }

286     private void checkForCRUD(final AbstractApexNode<?> node, final Object data, final String crudMethod) {
287         final HashSet<ASTMethodCallExpression> prevCalls = getPreviousMethodCalls(node);
288         for (ASTMethodCallExpression prevCall : prevCalls) {
289             collectCRUDMethodLevelChecks(prevCall);
290         }
291 
292         final ASTMethod wrappingMethod = node.getFirstParentOfType(ASTMethod.class);
293         final ASTUserClass wrappingClass = node.getFirstParentOfType(ASTUserClass.class);
294 
295         if ((wrappingClass != null && Helper.isTestMethodOrClass(wrappingClass))
296                 || (wrappingMethod != null && Helper.isTestMethodOrClass(wrappingMethod))) {
297             return;
298         }
299 
300         final ASTNewNameValueObjectExpression newObj = node.getFirstChildOfType(ASTNewNameValueObjectExpression.class);
301         if (newObj != null) {
302             final String type = Helper.getFQVariableName(newObj);
303             validateCRUDCheckPresent(node, data, crudMethod, type);
304         }
305 
306         final ASTVariableExpression variable = node.getFirstChildOfType(ASTVariableExpression.class);
307         if (variable != null) {
308             final String type = varToTypeMapping.get(Helper.getFQVariableName(variable));
309             if (type != null) {
310                 StringBuilder typeCheck = new StringBuilder().append(node.getNode().getDefiningType().getApexName())
311                         .append(":").append(type);
312 
313                 validateCRUDCheckPresent(node, data, crudMethod, typeCheck.toString());
314             }
315         }
316     }

318     private HashSet<ASTMethodCallExpression> getPreviousMethodCalls(final AbstractApexNode<?> self) {
319         final HashSet<ASTMethodCallExpression> innerMethodCalls = new HashSet<>();
320         final ASTMethod outerMethod = self.getFirstParentOfType(ASTMethod.class);
321         if (outerMethod != null) {
322             final ASTBlockStatement blockStatement = outerMethod.getFirstChildOfType(ASTBlockStatement.class);
323             recursivelyEvaluateCRUDMethodCalls(self, innerMethodCalls, blockStatement);
324 
325             final List<ASTMethod> constructorMethods = findConstructorlMethods(self);
326             for (ASTMethod method : constructorMethods) {
327                 innerMethodCalls.addAll(method.findDescendantsOfType(ASTMethodCallExpression.class));
328             }
329 
330             // some methods might be within this class
331             mapCallToMethodDecl(self, innerMethodCalls, new ArrayList<ASTMethodCallExpression>(innerMethodCalls));
332         }
333 
334         return innerMethodCalls;
335     }
336 
337     private void recursivelyEvaluateCRUDMethodCalls(final AbstractApexNode<?> self,
338             final HashSet<ASTMethodCallExpression> innerMethodCalls, final ASTBlockStatement blockStatement) {
339         if (blockStatement != null) {
340             int numberOfStatements = blockStatement.jjtGetNumChildren();
341             for (int i = 0; i < numberOfStatements; i++) {
342                 Node n = blockStatement.jjtGetChild(i);
343 
344                 if (n instanceof ASTIfElseBlockStatement) {
345                     List<ASTBlockStatement> innerBlocks = n.findDescendantsOfType(ASTBlockStatement.class);
346                     for (ASTBlockStatement innerBlock : innerBlocks) {
347                         recursivelyEvaluateCRUDMethodCalls(self, innerMethodCalls, innerBlock);
348                     }
349                 }
350 
351                 AbstractApexNode<?> match = n.getFirstDescendantOfType(self.getClass());
352                 if (match == self) {
353                     break;
354                 }
355                 ASTMethodCallExpression methodCall = n.getFirstDescendantOfType(ASTMethodCallExpression.class);
356                 if (methodCall != null) {
357                     mapCallToMethodDecl(self, innerMethodCalls, Arrays.asList(methodCall));
358                 }
359             }
360 
361         }
362     }
363 
364     private void mapCallToMethodDecl(final AbstractApexNode<?> self,
365             final HashSet<ASTMethodCallExpression> innerMethodCalls, final List<ASTMethodCallExpression> nodes) {
366         for (ASTMethodCallExpression node : nodes) {
367             if (node == self) {
368                 break;
369             }
370 
371             final ASTMethod methodBody = resolveMethodCalls(node);
372             if (methodBody != null) {
373                 innerMethodCalls.addAll(methodBody.findDescendantsOfType(ASTMethodCallExpression.class));
374             }
375 
376         }
377     }

379     private List<ASTMethod> findConstructorlMethods(final AbstractApexNode<?> node) {
380         final ArrayList<ASTMethod> ret = new ArrayList<>();
381         final Set<String> constructors = classMethods.keySet().stream()
382                 .filter(p -> (p.contains("<init>") || p.contains("<clinit>"))).collect(Collectors.toSet());
383 
384         for (String c : constructors) {
385             ret.add(classMethods.get(c));
386         }
387 
388         return ret;
389     }

391     private ASTMethod resolveMethodCalls(final ASTMethodCallExpression node) {
392         StringBuilder sb = new StringBuilder().append(node.getNode().getDefiningType().getApexName()).append(":")
393                 .append(node.getNode().getMethodName()).append(":").append(node.getNode().getInputParameters().size());
394         return classMethods.get(sb.toString());
395     }

397     private boolean isProperESAPICheckForDML(final String typeToCheck, final String dmlOperation) {
398         final boolean hasMapping = checkedTypeToDMLOperationViaESAPI.containsKey(typeToCheck.toString());
399         if (hasMapping) {
400             if (dmlOperation.equals(ANY)) {
401                 return true;
402             }
403 
404             String dmlChecked = checkedTypeToDMLOperationViaESAPI.get(typeToCheck);
405             return dmlChecked.equals(dmlOperation);
406         }
407 
408         return false;
409     }

411     private void extractObjectTypeFromESAPI(final ASTMethodCallExpression node, final String dmlOperation) {
412         final ASTVariableExpression var = node.getFirstChildOfType(ASTVariableExpression.class);
413         if (var != null) {
414             final ASTReferenceExpression reference = var.getFirstChildOfType(ASTReferenceExpression.class);
415             if (reference != null) {
416                 List<Identifier> identifiers = reference.getNode().getJadtIdentifiers();
417                 if (identifiers.size() == 1) {
418                     StringBuilder sb = new StringBuilder().append(node.getNode().getDefiningType().getApexName())
419                             .append(":").append(identifiers.get(0).value);
420                     checkedTypeToDMLOperationViaESAPI.put(sb.toString(), dmlOperation);
421                 }
422 
423             }
424         }
425 
426     }

428     private void validateCRUDCheckPresent(final AbstractApexNode<?> node, final Object data, final String CRUDMethod,
429             final String typeCheck) {
430         if (!typeToDMLOperationMapping.containsKey(typeCheck)) {
431             if (!isProperESAPICheckForDML(typeCheck, CRUDMethod)) {
432                 addViolation(data, node);
433             }
434         } else {
435             boolean properChecksHappened = false;
436 
437             List<String> dmlOperationsChecked = typeToDMLOperationMapping.get(typeCheck);
438             for (String dmlOp : dmlOperationsChecked) {
439                 if (dmlOp.equalsIgnoreCase(CRUDMethod)) {
440                     properChecksHappened = true;
441                     break;
442                 }
443                 if (CRUDMethod.equals(ANY)) {
444                     properChecksHappened = true;
445                     break;
446                 }
447             }
448 
449             if (!properChecksHappened) {
450                 addViolation(data, node);
451             }
452         }
453     }

455     private void checkForAccessibility(final ASTSoqlExpression node, Object data) {
456         final HashSet<ASTMethodCallExpression> prevCalls = getPreviousMethodCalls(node);
457         for (ASTMethodCallExpression prevCall : prevCalls) {
458             collectCRUDMethodLevelChecks(prevCall);
459         }
460 
461         boolean isGetter = false;
462         String returnType = null;
463 
464         final ASTMethod wrappingMethod = node.getFirstParentOfType(ASTMethod.class);
465         final ASTUserClass wrappingClass = node.getFirstParentOfType(ASTUserClass.class);
466 
467         if ((wrappingClass != null && Helper.isTestMethodOrClass(wrappingClass))
468                 || (wrappingMethod != null && Helper.isTestMethodOrClass(wrappingMethod))) {
469             return;
470         }
471 
472         if (wrappingMethod != null) {
473             isGetter = isMethodAGetter(wrappingMethod);
474             returnType = getReturnType(wrappingMethod);
475         }
476 
477         final ASTVariableDeclaration variableDecl = node.getFirstParentOfType(ASTVariableDeclaration.class);
478         if (variableDecl != null) {
479             String type = variableDecl.getNode().getLocalInfo().getType().getApexName();
480             type = getSimpleType(type);
481             StringBuilder typeCheck = new StringBuilder().append(variableDecl.getNode().getDefiningType().getApexName())
482                     .append(":").append(type);
483 
484             if (!isGetter) {
485                 validateCRUDCheckPresent(node, data, ANY, typeCheck.toString());
486             }
487 
488         }
489 
490         final ASTAssignmentExpression assignment = node.getFirstParentOfType(ASTAssignmentExpression.class);
491         if (assignment != null) {
492             final ASTVariableExpression variable = assignment.getFirstChildOfType(ASTVariableExpression.class);
493             if (variable != null) {
494                 String variableWithClass = Helper.getFQVariableName(variable);
495                 if (varToTypeMapping.containsKey(variableWithClass)) {
496                     String type = varToTypeMapping.get(variableWithClass);
497                     if (!isGetter) {
498                         validateCRUDCheckPresent(node, data, ANY, type);
499                     }
500                 }
501             }
502 
503         }
504 
505         final ASTReturnStatement returnStatement = node.getFirstParentOfType(ASTReturnStatement.class);
506         if (returnStatement != null) {
507             if (!isGetter) {
508                 validateCRUDCheckPresent(node, data, ANY, returnType == null ? "" : returnType);
509             }
510         }
511     }

513     private String getReturnType(final ASTMethod method) {
514         return new StringBuilder().append(method.getNode().getDefiningType().getApexName()).append(":")
515                 .append(method.getNode().getMethodInfo().getEmitSignature().getReturnType().getApexName()).toString();
516     }

518     private boolean isMethodAGetter(final ASTMethod method) {
519         final boolean startsWithGet = method.getNode().getMethodInfo().getCanonicalName().startsWith("get");
520         final boolean voidOrString = p
521                 .matcher(method.getNode().getMethodInfo().getEmitSignature().getReturnType().getApexName()).matches();
522 
523         return (startsWithGet && !voidOrString);
524     }
