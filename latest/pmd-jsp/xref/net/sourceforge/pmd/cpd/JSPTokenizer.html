5   package net.sourceforge.pmd.cpd;
6   
7   import java.io.Reader;
8   import java.io.StringReader;
9   
10  import org.apache.commons.io.IOUtils;
11  
12  import net.sourceforge.pmd.lang.LanguageRegistry;
13  import net.sourceforge.pmd.lang.LanguageVersionHandler;
14  import net.sourceforge.pmd.lang.TokenManager;
15  import net.sourceforge.pmd.lang.jsp.JspLanguageModule;
16  import net.sourceforge.pmd.lang.jsp.ast.Token;
17  import net.sourceforge.pmd.util.IOUtil;
18  
19  public class JSPTokenizer implements Tokenizer {
20  
21      public void tokenize(SourceCode sourceCode, Tokens tokenEntries) {
22          StringBuilder buffer = sourceCode.getCodeBuffer();
23          LanguageVersionHandler languageVersionHandler = LanguageRegistry.getLanguage(JspLanguageModule.NAME)
24                  .getDefaultVersion().getLanguageVersionHandler();
25          Reader reader = null;
26  
27          try {
28              reader = new StringReader(buffer.toString());
29              reader = IOUtil.skipBOM(reader);
30              TokenManager tokenMgr = languageVersionHandler.getParser(languageVersionHandler.getDefaultParserOptions())
31                      .getTokenManager(sourceCode.getFileName(), reader);
32              Token currentToken = (Token) tokenMgr.getNextToken();
33  
34              while (currentToken.image.length() > 0) {
35                  tokenEntries.add(new TokenEntry(String.valueOf(currentToken.kind), sourceCode.getFileName(),
36                          currentToken.beginLine));
37                  currentToken = (Token) tokenMgr.getNextToken();
38              }
39          } finally {
40              IOUtils.closeQuietly(reader);
41          }
42          tokenEntries.add(TokenEntry.getEOF());
43      }
44  }
