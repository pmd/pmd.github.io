23  public class AbstractWsdlRuleTest {
24  
25      @Test
26      public void testVisit() throws Exception {
27          String source = "<?xml version=\"1.0\"?><foo abc=\"abc\"><bar/></foo>";
28          XmlParserOptions parserOptions = new XmlParserOptions();
29          Parser parser =
30                  LanguageRegistry.getLanguage(WsdlLanguageModule.NAME).getDefaultVersion().getLanguageVersionHandler()
31                          .getParser(parserOptions);
32          XmlNode xmlNode = (XmlNode) parser.parse(null, new StringReader(source));
33          List<XmlNode> nodes = new ArrayList<>();
34          nodes.add(xmlNode);
35  
36          MyRule rule = new MyRule();
37          rule.apply(nodes, null);
38  
39          assertEquals(3, rule.visitedNodes.size());
40          assertEquals("document", rule.visitedNodes.get(0).toString());
41          assertEquals("foo", rule.visitedNodes.get(1).toString());
42          assertEquals("bar", rule.visitedNodes.get(2).toString());
43      }
44  
45      private static class MyRule extends AbstractWsdlRule {
46          final List<XmlNode> visitedNodes = new ArrayList<>();
47  
48          public MyRule() {
49          }
50  
51          @Override
52          public void apply(List<? extends Node> nodes, RuleContext ctx) {
53              visitedNodes.clear();
54              super.apply(nodes, ctx);
55          }
56  
57          @Override
58          protected void visit(XmlNode node, RuleContext ctx) {
59              visitedNodes.add(node);
60              super.visit(node, ctx);
61          }
62      }
63  
64      public static junit.framework.Test suite() {
65          return new junit.framework.JUnit4TestAdapter(AbstractWsdlRuleTest.class);
66      }
67  }
