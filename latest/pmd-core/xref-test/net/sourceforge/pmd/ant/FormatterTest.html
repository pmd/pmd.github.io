7   import static org.junit.Assert.assertFalse;

11  import java.io.File;

13  import org.junit.Test;

15  import net.sourceforge.pmd.renderers.CSVRenderer;
16  import net.sourceforge.pmd.renderers.HTMLRenderer;
17  import net.sourceforge.pmd.renderers.TextRenderer;
18  import net.sourceforge.pmd.renderers.XMLRenderer;
19  
20  public class FormatterTest {
21  
22      @Test
23      public void testType() {
24          Formatter f = new Formatter();
25          f.setType("xml");
26          assertTrue(f.createRenderer() instanceof XMLRenderer);
27          f.setType("text");
28          assertTrue(f.createRenderer() instanceof TextRenderer);
29          f.setType("csv");
30          assertTrue(f.createRenderer() instanceof CSVRenderer);
31          f.setType("html");
32          assertTrue(f.createRenderer() instanceof HTMLRenderer);
33          try {
34              f.setType("FAIL");
35              f.createRenderer();
36              fail("Expected IllegalArgumentException");
37          } catch (IllegalArgumentException be) {
38              assertTrue(be.getMessage().startsWith("Can't find the custom format FAIL"));
39          }
40      }
41  
42      @Test
43      public void testNull() {
44          Formatter f = new Formatter();
45          assertTrue("Formatter toFile should start off null!", f.isNoOutputSupplied());
46          f.setToFile(new File("foo"));
47          assertFalse("Formatter toFile should not be null!", f.isNoOutputSupplied());
48      }
49  }
