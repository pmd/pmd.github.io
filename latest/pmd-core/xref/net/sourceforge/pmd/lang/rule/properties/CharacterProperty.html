5   package net.sourceforge.pmd.lang.rule.properties;
6   
7   import java.util.Map;
8   
9   import net.sourceforge.pmd.PropertyDescriptorFactory;
10  import net.sourceforge.pmd.lang.rule.properties.factories.BasicPropertyDescriptorFactory;
11  
12  /**
13   * Defines a property type that supports single Character values.
14   *
15   * @author Brian Remedios
16   */
17  public class CharacterProperty extends AbstractProperty<Character> {
18  
19      public static final PropertyDescriptorFactory FACTORY = new BasicPropertyDescriptorFactory<CharacterProperty>(
20              Character.class) {
21  
22          @Override
23          public CharacterProperty createWith(Map<String, String> valuesById) {
24              return new CharacterProperty(nameIn(valuesById), descriptionIn(valuesById),
25                      defaultValueIn(valuesById) != null ? new Character(defaultValueIn(valuesById).charAt(0)) : null,
26                      0f);
27          }
28      };
29  
30      /**
31       * Constructor for CharacterProperty.
32       *
33       * @param theName
34       *            String
35       * @param theDescription
36       *            String
37       * @param theDefault
38       *            Character
39       * @param theUIOrder
40       *            float
41       */
42      public CharacterProperty(String theName, String theDescription, Character theDefault, float theUIOrder) {
43          super(theName, theDescription, theDefault, theUIOrder);
44      }
45  
46      /**
47       * Constructor for CharacterProperty.
48       *
49       * @param theName
50       *            String
51       * @param theDescription
52       *            String
53       * @param defaultStr
54       *            String
55       * @param theUIOrder
56       *            float
57       * @throws IllegalArgumentException
58       */
59      public CharacterProperty(String theName, String theDescription, String defaultStr, float theUIOrder) {
60          this(theName, theDescription, charFrom(defaultStr), theUIOrder);
61      }
62  
63      /**
64       * @param charStr
65       *            String
66       * @return Character
67       */
68      public static Character charFrom(String charStr) {
69  
70          if (charStr == null || charStr.length() != 1) {
71              throw new IllegalArgumentException("missing/invalid character value");
72          }
73          return charStr.charAt(0);
74      }
75  
76      /**
77       * @return Class
78       * @see net.sourceforge.pmd.PropertyDescriptor#type()
79       */
80      @Override
81      public Class<Character> type() {
82          return Character.class;
83      }
84  
85      /**
86       * @param valueString
87       *            String
88       * @return Object
89       * @throws IllegalArgumentException
90       * @see net.sourceforge.pmd.PropertyDescriptor#valueFrom(String)
91       */
92      @Override
93      public Character valueFrom(String valueString) throws IllegalArgumentException {
94          return charFrom(valueString);
95      }
96  
97      /**
98       * Method defaultAsString.
99       *
100      * @return String
101      */
102     @Override
103     protected String defaultAsString() {
104         return Character.toString(defaultValue());
105     }
106 }
