19  /**
20   * Provide a single parameter to specify database objects to process.
21   *
22   * <p>
23   * Wrap JDBC settings for use by PMD: optional parameters specify the source
24   * code to be passed to PMD, or are inherited from the associated
25   * {@link DBType}.
26   * </p>
27   *
28   * <p>
29   * A DBURI is a <i>faux</i>-URI: it does not have a formal specification and
30   * comprises a JDBC(-ish) URL and an optional query, e.g.
31   * <code>jdbc : subprotocol  [ : subname ] : connection details [ query ] </code>.
32   * </p>
33   *
34   * <p>
35   * The subprotocol and optional subname parts should be a valid DBType
36   * JDBC(-ish) URL jdbc:oracle:thin:username/password@//192.168.100.21:1521/ORCL
37   * JDBC(-ish) URL jdbc:thin:username/password@//192.168.100.21:1521/ORCL
38   * </p>
39   *
40   * <p>
41   * The query includes one or more of these:
42   * </p>
43   * <dl>
44   * <dt>characterset</dt>
45   * <dd>utf8</dd>
46   * <dt>languages</dt>
47   * <dd>comma-separated list of desired PMD languages</dd>
48   * <dt>schemas</dt>
49   * <dd>comma-separated list of database schemas</dd>
50   * <dt>sourcecodetypes</dt>
51   * <dd>comma-separated list of database source code types</dd>
52   * <dt>sourcecodenames</dt>
53   * <dd>comma-separated list of database source code names</dd>
54   * </dl>
55   *
56   * @see URI
57   * @author sturton
58   */

126     /**
127      * Create DBURI from a string, combining a JDBC URL and query parameters.
128      *
129      * <p>
130      * From the JDBC URL component, infer:
131      * </p>
132      * <ul>
133      * <li>JDBC driver class</li>
134      * <li>supported languages</li>
135      * <li>default source code types</li>
136      * <li>default schemas</li>
137      * </ul>
138      *
139      * <p>
140      * From the query component, define these values, overriding any defaults:
141      * </p>
142      * <ul>
143      * <li>parsing language</li>
144      * <li>source code types</li>
145      * <li>schemas</li>
146      * <li>source code</li>
147      * </ul>
148      *
149      * @param string
150      *            URL string
151      * @throws URISyntaxException
152      */
153     public DBURI(String string) throws URISyntaxException {
154         /*
155          * A JDBC URL is an opaque URL and does not have a query.
156          *
157          * We pretend that it does, strip off the query, use the real JDBC URL
158          * component to infer languages JDBC driver class supported languages
159          * default source code types default schemas generate a faux HTTP URI
160          * with the query, extract the query parameters
161          */
162 
163         uri = new URI(string);
164 
165         try {
166             // Split the string between JDBC URL and the query
167             String[] splitURI = string.split("\\?");
168 
169             if (splitURI.length > 1) {
170                 url = splitURI[0];
171             } else {
172                 url = string;
173             }
174 
175             LOGGER.log(Level.FINE, "Extracted URL={0}", url);
176 
177             // Explode URL into its separate components
178             setFields();
179 
180             // If the original URI string contained a query component, split it
181             // into parameters
182             if (splitURI.length > 1) {
183                 // Generate a fake HTTP URI to allow easy extraction of the
184                 // query parameters
185                 String chimeraString = "http://local?" + string.substring(url.length() + 1);
186                 LOGGER.log(Level.FINEST, "chimeraString={0}", chimeraString);
187                 URI chimeraURI = new URI(chimeraString);
188                 dump("chimeraURI", chimeraURI);
189 
190                 parameters = getParameterMap(chimeraURI);
191 
192                 LOGGER.log(Level.FINEST, "parameterMap=={0}", parameters);
193 
194                 characterSet = parameters.get("characterset");
195                 sourceCodeTypes = parameters.get("sourcecodetypes");
196                 sourceCodeNames = parameters.get("sourcecodenames");
197                 languages = parameters.get("languages");
198 
199                 // Populate the lists
200                 if (null != sourceCodeNames) {
201                     sourceCodeNamesList = Arrays.asList(sourceCodeNames.split(","));
202                 }
203 
204                 if (null != languages) {
205                     languagesList = Arrays.asList(languages.split(","));
206                 }
207 
208                 if (null != parameters.get("schemas")) {
209                     schemasList = Arrays.asList(parameters.get("schemas").split(","));
210                 }
211 
212                 if (null != sourceCodeTypes) {
213                     sourceCodeTypesList = Arrays.asList(sourceCodeTypes.split(","));
214                 }
215 
216             }
217 
218         } catch (URISyntaxException ex) {
219             URISyntaxException uriException = new URISyntaxException(string, "Problem generating DBURI.");
220             uriException.initCause(ex);
221             throw uriException;
222         } catch (IOException e) {
223             throw new RuntimeException(e);
224         }
225     }

265     /**
266      * Return extracted parameters from dburi.
267      *
268      * @param dburi
269      * @return extracted parameters
270      * @throws UnsupportedEncodingException
271      */
272     private Map<String, String> getParameterMap(URI dburi) throws UnsupportedEncodingException {
273 
274         Map<String, String> map = new HashMap<>();
275         String query = dburi.getRawQuery();
276         LOGGER.log(Level.FINEST, "dburi,getQuery()={0}", query);
277         if (null != query && !"".equals(query)) {
278             String[] params = query.split("&");
279             for (String param : params) {
280                 String[] splits = param.split("=");
281                 String name = splits[0];
282                 String value = null;
283                 if (splits.length > 1) {
284                     value = splits[1];
285                 }
286                 map.put(name, (null == value) ? value : URLDecoder.decode(value, "UTF-8"));
287             }
288         }
289         return map;
290     }

298     static void dump(String description, URI dburi) {
299 
300         String dumpString = String.format(
301                 "dump (%s)\n: isOpaque=%s, isAbsolute=%s Scheme=%s,\n SchemeSpecificPart=%s,\n Host=%s,\n Port=%s,\n Path=%s,\n Fragment=%s,\n Query=%s\n",
302                 description, dburi.isOpaque(), dburi.isAbsolute(), dburi.getScheme(), dburi.getSchemeSpecificPart(),
303                 dburi.getHost(), dburi.getPort(), dburi.getPath(), dburi.getFragment(), dburi.getQuery());
304 
305         LOGGER.fine(dumpString);
306 
307         String query = dburi.getQuery();
308         if (null != query && !"".equals(query)) {
309             String[] params = query.split("&");
310             Map<String, String> map = new HashMap<>();
311             for (String param : params) {
312                 String[] splits = param.split("=");
313                 String name = splits[0];
314                 String value = null;
315                 if (splits.length > 1) {
316                     value = splits[1];
317                 }
318                 map.put(name, value);
319                 LOGGER.fine(String.format("name=%s,value=%s\n", name, value));
320             }
321         }
322         // return map;
323     }

436     /**
437      * Populate the URI and query collections from the original string
438      *
439      * @throws URISyntaxException
440      * @throws IOException
441      */
442     private void setFields() throws URISyntaxException, IOException {
443         if (url.startsWith("jdbc:")) {
444             // java.net.URI is intended for "normal" URLs
445             URI jdbcURI = new URI(getURL().substring(5));
446 
447             LOGGER.log(Level.FINE, "setFields - substr(jdbcURL,5):{0}", getURL().substring(5));
448             dump("substr(jdbcURL,5)", jdbcURI);
449 
450             // jdbc:subprotocol:subname
451             String[] uriParts = url.split(":");
452             for (String part : uriParts) {
453                 LOGGER.log(Level.FINEST, "JDBCpart={0}", part);
454             }
455 
456             /*
457              * Expect jdbc : subprotocol [ : subname ] : connection details
458              * uriParts.length < 3 Error uriParts.length = 3 Driver information
459              * may be inferred from part[1] - the subprotocol uriParts.length >=
460              * 4 Driver information may be inferred from part[2]- the first part
461              * of the subname
462              */
463             if (3 == uriParts.length) {
464                 subprotocol = uriParts[1];
465             } else if (4 <= uriParts.length) {
466                 subprotocol = uriParts[1];
467                 subnamePrefix = uriParts[2];
468             } else {
469                 throw new URISyntaxException(getURL(), "Could not understand JDBC URL", 1);
470             }
471 
472             LOGGER.log(Level.FINE, "subprotocol={0}'' subnamePrefix={1}", new Object[] { subprotocol, subnamePrefix });
473 
474             // Set values from DBType defaults
475             this.dbType = new DBType(subprotocol, subnamePrefix);
476 
477             LOGGER.log(Level.FINER, "DBType properties found at {0} with {1} properties.",
478                     new Object[] { dbType.getPropertiesSource(), dbType.getProperties().size() });
479 
480             LOGGER.log(Level.FINEST, "DBType properties are:- {0}", dbType.getProperties());
481 
482             if (null != dbType.getDriverClass()) {
483                 this.driverClass = dbType.getDriverClass();
484             }
485 
486             if (null != dbType.getCharacterSet()) {
487                 this.characterSet = dbType.getCharacterSet();
488             }
489 
490             if (null != dbType.getLanguages()) {
491                 this.languages = dbType.getLanguages();
492             }
493 
494             if (null != dbType.getSourceCodeTypes()) {
495                 sourceCodeTypes = dbType.getSourceCodeTypes();
496             }
497 
498             LOGGER.finer("DBType other properties follow  ...");
499 
500             if (null != dbType.getProperties().getProperty("schemas")) {
501                 schemasList = Arrays.asList(dbType.getProperties().getProperty("schemas").split(","));
502             }
503 
504             sourceCodeNames = dbType.getProperties().getProperty("sourcecodenames");
505 
506             String returnType = dbType.getProperties().getProperty("returnType");
507             if (null != returnType) {
508                 sourceCodeType = Integer.parseInt(returnType);
509             }
510 
511             LOGGER.finer("DBType populating lists ");
512             // Populate the lists
513             if (null != sourceCodeNames) {
514                 sourceCodeNamesList = Arrays.asList(sourceCodeNames.split(","));
515             }
516 
517             if (null != languages) {
518                 languagesList = Arrays.asList(languages.split(","));
519             }
520 
521             if (null != sourceCodeTypes) {
522                 sourceCodeTypesList = Arrays.asList(sourceCodeTypes.split(","));
523             }
524 
525             LOGGER.finer("DBType lists generated");
526         }
527
