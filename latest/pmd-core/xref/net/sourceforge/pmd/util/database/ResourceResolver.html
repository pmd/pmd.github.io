5   package net.sourceforge.pmd.util.database;
6   
7   import javax.xml.transform.Source;
8   import javax.xml.transform.TransformerException;
9   import javax.xml.transform.URIResolver;
10  import javax.xml.transform.stream.StreamSource;
11  
12  public class ResourceResolver implements URIResolver {
13      @Override
14      public Source resolve(String href, String base) throws TransformerException {
15          if (null == href || href.length() == 0) {
16              return null; // will make Oracle XSLT processor explode,
17              // even though it's correct
18          }
19          try {
20              String resource = href;
21              ResourceLoader loader = new ResourceLoader();
22              return new StreamSource(loader.getResourceStream(resource), resource);
23          } catch (Exception ex) {
24              throw new TransformerException(ex);
25          }
26      }
27  }
