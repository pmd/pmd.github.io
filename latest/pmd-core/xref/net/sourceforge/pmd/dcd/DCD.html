19  /**
20   * The Dead Code Detector is used to find dead code. What is dead code? Dead
21   * code is code which is not used by other code? It exists, but it not used.
22   * Unused code is clutter, which can generally be a candidate for removal.
23   *
24   * <p>When performing dead code detection, there are various sets of files/classes
25   * which must be identified. An analogy of the dead code analysis as a <em>foot
26   * race</em> is used to help clarify each of these sets:</p>
27   *
28   * <ol>
29   * <li>The <em>direct users</em> is the set of Classes which will always be
30   * parsed to determine what code they use. This set is the starting point of the
31   * race.</li>
32   * <li>The <em>indirect users</em> is the set of Classes which will only be
33   * parsed if they are accessed by code in the <em>direct users</em> set, or in
34   * the <em>indirect users</em> set. This set is the course of the race.</li>
35   * <li>The <em>dead code candidates</em> are the set of Classes which are the
36   * focus of the dead code detection. This set is the finish line of the
37   * race.</li>
38   * </ol>
39   *
40   * <p>Typically there is intersection between the set of <em>direct users</em>,
41   * <em>indirect users</em> and <em>dead code candidates</em>, although it is not
42   * required. If the sets are defined too tightly, there the potential for a lot
43   * of code to be considered as dead code. You may need to expand the <em>direct
44   * users</em> or <em>indirect users</em> sets, or explore using different
45   * options.</p>
46   *
47   * @author Ryan Gustafson &lt;ryan.gustafson@gmail.com&gt;
48   */

86      public static void dump(UsageGraph usageGraph, boolean verbose) {
87          usageGraph.accept(new DumpNodeVisitor(), Boolean.valueOf(verbose));
88      }

91          usageGraph.accept(new UsageNodeVisitor(), Boolean.valueOf(verbose));

94      public static void main(String[] args) throws Exception {
95          // 1) Directories
96          List<File> directories = new ArrayList<>();
97          directories.add(new File("C:/pmd/workspace/pmd-trunk/src"));
98  
99          // Basic filter
100         FilenameFilter javaFilter = new FilenameFilter() {
101             @Override
102             public boolean accept(File dir, String name) {
103                 // Recurse on directories
104                 if (new File(dir, name).isDirectory()) {
105                     return true;
106                 } else {
107                     return name.endsWith(".java");
108                 }
109             }
110         };
111 
112         // 2) Filename filters
113         List<FilenameFilter> filters = new ArrayList<>();
114         filters.add(javaFilter);
115 
116         assert directories.size() == filters.size();
117 
118         // Find all files, convert to class names
119         List<String> classes = new ArrayList<>();
120         for (int i = 0; i < directories.size(); i++) {
121             File directory = directories.get(i);
122             FilenameFilter filter = filters.get(i);
123             List<File> files = new FileFinder().findFilesFrom(directory, filter, true);
124             for (File file : files) {
125                 String name = file.getPath();
126 
127                 // Chop off directory
128                 name = name.substring(directory.getPath().length() + 1);
129 
130                 // Drop extension
131                 name = name.replaceAll("\\.java$", "");
132 
133                 // Trim path separators
134                 name = name.replace('\\', '.');
135                 name = name.replace('/', '.');
136 
137                 classes.add(name);
138             }
139         }
140 
141         long start = System.currentTimeMillis();
142 
143         // Define filter for "indirect users" and "dead code candidates".
144         // TODO Need to support these are different concepts.
145         List<String> includeRegexes = Arrays.asList(new String[] { "net\\.sourceforge\\.pmd\\.dcd.*", "us\\..*" });
146         List<String> excludeRegexes = Arrays.asList(new String[] { "java\\..*", "javax\\..*", ".*\\.twa\\..*" });
147         Filter<String> classFilter = Filters.buildRegexFilterExcludeOverInclude(includeRegexes, excludeRegexes);
148         System.out.println("Class filter: " + classFilter);
149 
150         // Index each of the "direct users"
151         UsageGraphBuilder builder = new UsageGraphBuilder(classFilter);
152         int total = 0;
153         for (String clazz : classes) {
154             System.out.println("indexing class: " + clazz);
155             builder.index(clazz);
156             total++;
157             if (total % 20 == 0) {
158                 System.out.println(total + " : " + total / ((System.currentTimeMillis() - start) / 1000.0));
159             }
160         }
161 
162         // Reporting
163         boolean dump = true;
164         boolean deadCode = true;
165         UsageGraph usageGraph = builder.getUsageGraph();
166         if (dump) {
167             System.out.println("--- Dump ---");
168             dump(usageGraph, true);
169         }
170         if (deadCode) {
171             System.out.println("--- Dead Code ---");
172             report(usageGraph, true);
173         }
174         long end = System.currentTimeMillis();
175         System.out.println("Time: " + (end - start) / 1000.0);
176     }
