4   package net.sourceforge.pmd.processor;

6   import java.io.IOException;
7   import java.util.List;
8   
9   import net.sourceforge.pmd.PMDConfiguration;
10  import net.sourceforge.pmd.Report;
11  import net.sourceforge.pmd.RuleContext;
12  import net.sourceforge.pmd.RuleSetFactory;
13  import net.sourceforge.pmd.RuleSets;
14  import net.sourceforge.pmd.RulesetsFactoryUtils;
15  import net.sourceforge.pmd.SourceCodeProcessor;
16  import net.sourceforge.pmd.benchmark.Benchmark;
17  import net.sourceforge.pmd.benchmark.Benchmarker;
18  import net.sourceforge.pmd.renderers.Renderer;
19  import net.sourceforge.pmd.util.datasource.DataSource;
20  
21  /**
22   * @author Romain Pelisse <belaran@gmail.com>
23   *
24   */
25  public abstract class AbstractPMDProcessor {
26  
27      protected final PMDConfiguration configuration;
28  
29      public AbstractPMDProcessor(PMDConfiguration configuration) {
30          this.configuration = configuration;
31      }
32  
33      public void renderReports(final List<Renderer> renderers, final Report report) {
34          long start = System.nanoTime();
35  
36  		try {	
37  			for (Renderer r : renderers) {
38  				r.renderFileReport(report);
39  			}
40  			long end = System.nanoTime();
41  			Benchmarker.mark(Benchmark.Reporting, end - start, 1);
42  		} catch (IOException ioe) {
43  
44  		}
45  	}
46  
47  	protected String filenameFrom(DataSource dataSource) {
48  		return dataSource.getNiceFileName(
49  			 configuration.isReportShortNames(), 
50  			 configuration.getInputPaths()
51  			 );
52  	}
53  
54  	/**
55  	 * Create instances for each rule defined in the ruleset(s) in
56  	 * the configuration.
57  	 * Please note, that the returned instances <strong>must not</strong>
58  	 * be used by different threads. Each thread must create its own
59  	 * copy of the rules (see {@link PmdRunnable.PmdThread#getRuleSets(String)}).
60  	 * @param factory
61  	 * @return the rules within a rulesets
62  	 */
63  	protected RuleSets createRuleSets(RuleSetFactory factory) {
64  		return RulesetsFactoryUtils.getRuleSets(configuration.getRuleSets(), factory);
65  	}
66  
67      public void processFiles(RuleSetFactory ruleSetFactory, List<DataSource> files, RuleContext ctx,
68              List<Renderer> renderers) {
69          SourceCodeProcessor processor = new SourceCodeProcessor(configuration);
70  
71          for (DataSource dataSource : files) {
72              String niceFileName = filenameFrom(dataSource);
73  
74              runAnalysis(new PmdRunnable(configuration, dataSource, niceFileName, renderers,
75                      ctx, ruleSetFactory, processor));
76          }
77  
78          collectReports(renderers);
79      }
80  
81      protected abstract void runAnalysis(PmdRunnable runnable);
82  
83      protected abstract void collectReports(List<Renderer> renderers);
84  }
