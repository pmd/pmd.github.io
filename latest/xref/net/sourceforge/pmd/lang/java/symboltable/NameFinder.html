5   package net.sourceforge.pmd.lang.java.symboltable;
6   
7   import java.util.ArrayList;
8   import java.util.List;
9   import java.util.StringTokenizer;
10  
11  import net.sourceforge.pmd.PMD;
12  import net.sourceforge.pmd.lang.java.ast.ASTArguments;
13  import net.sourceforge.pmd.lang.java.ast.ASTMemberSelector;
14  import net.sourceforge.pmd.lang.java.ast.ASTMethodReference;
15  import net.sourceforge.pmd.lang.java.ast.ASTName;
16  import net.sourceforge.pmd.lang.java.ast.ASTPrimaryExpression;
17  import net.sourceforge.pmd.lang.java.ast.ASTPrimaryPrefix;
18  import net.sourceforge.pmd.lang.java.ast.ASTPrimarySuffix;
19  import net.sourceforge.pmd.lang.java.ast.JavaNode;
20  import net.sourceforge.pmd.lang.symboltable.NameOccurrence;
21  
22  public class NameFinder {
23  
24      private List<JavaNameOccurrence> names = new ArrayList<>();
25  
26      public NameFinder(ASTPrimaryExpression node) {
27          ASTPrimaryPrefix prefix = (ASTPrimaryPrefix) node.jjtGetChild(0);
28          if (prefix.usesSuperModifier()) {
29              add(new JavaNameOccurrence(prefix, "super"));
30          } else if (prefix.usesThisModifier()) {
31              add(new JavaNameOccurrence(prefix, "this"));
32          }
33          for (int i = 0; i < node.jjtGetNumChildren(); i++) {
34              checkForNameChild((JavaNode) node.jjtGetChild(i));
35          }
36      }
37  
38      public List<JavaNameOccurrence> getNames() {
39          return names;
40      }
41  
42      private void checkForNameChild(JavaNode node) {
43          if (node.getImage() != null) {
44              add(new JavaNameOccurrence(node, node.getImage()));
45          }
46          if (node.jjtGetNumChildren() > 0 && node.jjtGetChild(0) instanceof ASTName) {
47              ASTName grandchild = (ASTName) node.jjtGetChild(0);
48              for (StringTokenizer st = new StringTokenizer(grandchild.getImage(), "."); st.hasMoreTokens();) {
49                  add(new JavaNameOccurrence(grandchild, st.nextToken()));
50              }
51          }
52          if (node.jjtGetNumChildren() > 1 && node.jjtGetChild(1) instanceof ASTMethodReference) {
53              ASTMethodReference methodRef = (ASTMethodReference) node.jjtGetChild(1);
54              add(new JavaNameOccurrence(methodRef, methodRef.getImage()));
55          }
56          if (node instanceof ASTPrimarySuffix) {
57              ASTPrimarySuffix suffix = (ASTPrimarySuffix) node;
58              if (suffix.isArguments()) {
59                  JavaNameOccurrence occurrence = names.get(names.size() - 1);
60                  occurrence.setIsMethodOrConstructorInvocation();
61                  ASTArguments args = (ASTArguments) ((ASTPrimarySuffix) node).jjtGetChild(0);
62                  occurrence.setArgumentCount(args.getArgumentCount());
63              } else if (suffix.jjtGetNumChildren() == 1 && suffix.jjtGetChild(0) instanceof ASTMemberSelector) {
64                  ASTMemberSelector member = (ASTMemberSelector) suffix.jjtGetChild(0);
65                  if (member.jjtGetNumChildren() == 1 && member.jjtGetChild(0) instanceof ASTMethodReference) {
66                      ASTMethodReference methodRef = (ASTMethodReference) member.jjtGetChild(0);
67                      add(new JavaNameOccurrence(methodRef, methodRef.getImage()));
68                  } else {
69                      add(new JavaNameOccurrence(member, member.getImage()));
70                  }
71              }
72          }
73      }
74  
75      private void add(JavaNameOccurrence name) {
76          names.add(name);
77          if (names.size() > 1) {
78              JavaNameOccurrence qualifiedName = names.get(names.size() - 2);
79              qualifiedName.setNameWhichThisQualifies(name);
80          }
81      }
82  
83      @Override
84      public String toString() {
85          StringBuilder result = new StringBuilder();
86          for (NameOccurrence occ : names) {
87              result.append(occ);
88              result.append(PMD.EOL);
89          }
90          return result.toString();
91      }
92  }
