11  //FUTURE This is not referenced by any RuleSet?
12  public class StringConcatenationRule extends AbstractJavaRule {
13  
14      public Object visit(ASTForStatement node, Object data) {
15          Node forLoopStmt = null;
16          for (int i = 0; i < 4; i++) {
17              forLoopStmt = node.jjtGetChild(i);
18              if (forLoopStmt instanceof ASTBlockStatement) {
19                  break;
20              }
21          }
22          if (forLoopStmt == null) {
23              return data;
24          }
25  
26          return data;
27      }
28  }
