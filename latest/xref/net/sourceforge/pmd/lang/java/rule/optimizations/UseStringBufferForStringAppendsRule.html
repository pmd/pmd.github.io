5   package net.sourceforge.pmd.lang.java.rule.optimizations;
6   
7   import net.sourceforge.pmd.lang.ast.Node;
8   import net.sourceforge.pmd.lang.java.ast.ASTArgumentList;
9   import net.sourceforge.pmd.lang.java.ast.ASTAssignmentOperator;
10  import net.sourceforge.pmd.lang.java.ast.ASTConditionalExpression;
11  import net.sourceforge.pmd.lang.java.ast.ASTEqualityExpression;
12  import net.sourceforge.pmd.lang.java.ast.ASTLocalVariableDeclaration;
13  import net.sourceforge.pmd.lang.java.ast.ASTName;
14  import net.sourceforge.pmd.lang.java.ast.ASTPrimaryExpression;
15  import net.sourceforge.pmd.lang.java.ast.ASTStatementExpression;
16  import net.sourceforge.pmd.lang.java.ast.ASTVariableDeclaratorId;
17  import net.sourceforge.pmd.lang.java.rule.AbstractJavaRule;
18  import net.sourceforge.pmd.lang.java.typeresolution.TypeHelper;
19  import net.sourceforge.pmd.lang.symboltable.NameOccurrence;
20  
21  public class UseStringBufferForStringAppendsRule extends AbstractJavaRule {
22      
23      @Override
24      public Object visit(ASTVariableDeclaratorId node, Object data) {
25          if (!TypeHelper.isA(node, String.class) || node.isArray()) {
26              return data;
27          }
28          Node parent = node.jjtGetParent().jjtGetParent();
29          if (!(parent instanceof ASTLocalVariableDeclaration)) {
30              return data;
31          }
32          for (NameOccurrence no : node.getUsages()) {
33              Node name = no.getLocation();
34              ASTStatementExpression statement = name.getFirstParentOfType(ASTStatementExpression.class);
35              if (statement == null) {
36                  continue;
37              }
38              ASTArgumentList argList = name.getFirstParentOfType(ASTArgumentList.class);
39              if (argList != null && argList.getFirstParentOfType(ASTStatementExpression.class) == statement) {
40                  // used in method call
41                  continue;
42              }
43              ASTEqualityExpression equality = name.getFirstParentOfType(ASTEqualityExpression.class);
44              if (equality != null && equality.getFirstParentOfType(ASTStatementExpression.class) == statement) {
45                  // used in condition
46                  continue;
47              }
48              ASTConditionalExpression conditional = name.getFirstParentOfType(ASTConditionalExpression.class);
49  
50              if (conditional != null) {
51                  Node thirdParent = name.jjtGetParent().jjtGetParent().jjtGetParent();
52                  if ((thirdParent == conditional || thirdParent.jjtGetParent() == conditional)
53                          && conditional.getFirstParentOfType(ASTStatementExpression.class) == statement) {
54                      // is used in ternary as only option (not appended to other
55                      // string)
56                      continue;
57                  }
58              }
59              if (statement.jjtGetNumChildren() > 0 && statement.jjtGetChild(0) instanceof ASTPrimaryExpression) {
60                  ASTName astName = statement.jjtGetChild(0).getFirstDescendantOfType(ASTName.class);
61                  if (astName != null) {
62                      if (astName.equals(name)) {
63                          ASTAssignmentOperator assignmentOperator = statement
64                                  .getFirstDescendantOfType(ASTAssignmentOperator.class);
65                          if (assignmentOperator != null && assignmentOperator.isCompound()) {
66                              addViolation(data, assignmentOperator);
67                          }
68                      } else if (astName.getImage().equals(name.getImage())) {
69                          ASTAssignmentOperator assignmentOperator = statement
70                                  .getFirstDescendantOfType(ASTAssignmentOperator.class);
71                          if (assignmentOperator != null && !assignmentOperator.isCompound()) {
72                              addViolation(data, astName);
73                          }
74                      }
75                  }
76              }
77          }
78          return data;
79      }
80  }
