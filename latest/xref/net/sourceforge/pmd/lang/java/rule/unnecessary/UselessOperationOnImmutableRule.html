5   package net.sourceforge.pmd.lang.java.rule.unnecessary;
6   
7   import java.util.HashMap;
8   import java.util.Map;
9   import java.util.Set;
10  
11  import net.sourceforge.pmd.lang.ast.Node;
12  import net.sourceforge.pmd.lang.java.ast.ASTLocalVariableDeclaration;
13  import net.sourceforge.pmd.lang.java.ast.ASTStatementExpression;
14  import net.sourceforge.pmd.lang.java.ast.ASTType;
15  import net.sourceforge.pmd.lang.java.ast.ASTVariableDeclaratorId;
16  import net.sourceforge.pmd.lang.java.rule.AbstractJavaRule;
17  import net.sourceforge.pmd.lang.symboltable.NameOccurrence;
18  import net.sourceforge.pmd.util.CollectionUtil;

20  /**
21   * An operation on an Immutable object (String, BigDecimal or BigInteger) won't
22   * change the object itself. The result of the operation is a new object.
23   * Therefore, ignoring the operation result is an error.
24   */
25  public class UselessOperationOnImmutableRule extends AbstractJavaRule {
26  
27      /**
28       * These are the BigDecimal methods which are immutable
29       */
30      private static final Set<String> BIG_DECIMAL_METHODS = CollectionUtil
31              .asSet(new String[] { ".abs", ".add", ".divide", ".divideToIntegralValue", ".max", ".min", ".movePointLeft",
32                  ".movePointRight", ".multiply", ".negate", ".plus", ".pow", ".remainder", ".round",
33                  ".scaleByPowerOfTen", ".setScale", ".stripTrailingZeros", ".subtract", ".ulp", });
34  
35      /**
36       * These are the BigInteger methods which are immutable
37       */
38      private static final Set<String> BIG_INTEGER_METHODS = CollectionUtil
39              .asSet(new String[] { ".abs", ".add", ".and", ".andNot", ".clearBit", ".divide", ".flipBit", ".gcd", ".max",
40                  ".min", ".mod", ".modInverse", ".modPow", ".multiply", ".negate", ".nextProbablePrine", ".not", ".or",
41                  ".pow", ".remainder", ".setBit", ".shiftLeft", ".shiftRight", ".subtract", ".xor", });
42  
43      /**
44       * These are the String methods which are immutable
45       */
46      private static final Set<String> STRING_METHODS = CollectionUtil
47              .asSet(new String[] { ".concat", ".intern", ".replace", ".replaceAll", ".replaceFirst", ".substring",
48                  ".toLowerCase", ".toString", ".toUpperCase", ".trim", });
49  
50      /**
51       * These are the classes that the rule can apply to
52       */
53      private static final Map<String, Set<String>> MAP_CLASSES = new HashMap<>();
54  
55      static {
56          MAP_CLASSES.put("java.math.BigDecimal", BIG_DECIMAL_METHODS);
57          MAP_CLASSES.put("BigDecimal", BIG_DECIMAL_METHODS);
58          MAP_CLASSES.put("java.math.BigInteger", BIG_INTEGER_METHODS);
59          MAP_CLASSES.put("BigInteger", BIG_INTEGER_METHODS);
60          MAP_CLASSES.put("java.lang.String", STRING_METHODS);
61          MAP_CLASSES.put("String", STRING_METHODS);
62      }
63  
64      @Override
65      public Object visit(ASTLocalVariableDeclaration node, Object data) {
66  
67          ASTVariableDeclaratorId var = getDeclaration(node);
68          if (var == null) {
69              return super.visit(node, data);
70          }
71          String variableName = var.getImage();
72          for (NameOccurrence no : var.getUsages()) {
73              // FIXME - getUsages will return everything with the same name as
74              // the variable,
75              // see JUnit test, case 6. Changing to Node below, revisit when
76              // getUsages is fixed
77              Node sn = no.getLocation();
78              Node primaryExpression = sn.jjtGetParent().jjtGetParent();
79              Class<? extends Node> parentClass = primaryExpression.jjtGetParent().getClass();
80              if (parentClass.equals(ASTStatementExpression.class)) {
81                  String methodCall = sn.getImage().substring(variableName.length());
82                  ASTType nodeType = node.getTypeNode();
83                  if (nodeType != null) {
84                      if (MAP_CLASSES.get(nodeType.getTypeImage()).contains(methodCall)) {
85                          addViolation(data, sn);
86                      }
87                  }
88              }
89          }
90          return super.visit(node, data);
91      }
92  
93      /**
94       * This method checks the variable declaration if it is on a class we care
95       * about. If it is, it returns the DeclaratorId
96       *
97       * @param node
98       *            The ASTLocalVariableDeclaration which is a problem
99       * @return ASTVariableDeclaratorId
100      */
101     private ASTVariableDeclaratorId getDeclaration(ASTLocalVariableDeclaration node) {
102         ASTType type = node.getTypeNode();
103         if (MAP_CLASSES.keySet().contains(type.getTypeImage())) {
104             return node.getFirstDescendantOfType(ASTVariableDeclaratorId.class);
105         }
106         return null;
107     }
108 }
