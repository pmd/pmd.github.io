5   package net.sourceforge.pmd.lang.java.rule.controversial;
6   
7   import net.sourceforge.pmd.lang.java.ast.ASTImportDeclaration;
8   import net.sourceforge.pmd.lang.java.rule.AbstractJavaRule;
9   
10  public class DontImportSunRule extends AbstractJavaRule {
11  
12      public Object visit(ASTImportDeclaration node, Object data) {
13          String img = node.jjtGetChild(0).getImage();
14          if (img.startsWith("sun.") && !img.startsWith("sun.misc.Signal")) {
15              addViolation(data, node);
16          }
17          return data;
18      }
19  
20  }
