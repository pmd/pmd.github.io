5   package net.sourceforge.pmd.lang.java.oom.signature;
6   
7   import net.sourceforge.pmd.lang.java.ast.AccessNode;
8   
9   /**
10   * Generic signature. This class is extended by classes specific to operations and fields.
11   *
12   * @author Cl√©ment Fournier
13   */
14  public abstract class Signature {
15  
16      public final Visibility visibility;
17  
18      public Signature(Visibility visibility) {
19          this.visibility = visibility;
20      }
21  
22      @Override
23      public boolean equals(Object o) {
24          return o instanceof Signature && visibility == ((Signature) o).visibility;
25      }
26  
27      @Override
28      public int hashCode() {
29          return visibility.hashCode() * 2;
30      }
31  
32      /**
33       * The visibility of a node.
34       */
35      public enum Visibility {
36          PUBLIC, PACKAGE, PROTECTED, PRIVATE;
37  
38          /**
39           * Returns the Visibility enum key for a node.
40           *
41           * @param node A node
42           *
43           * @return The visibility enum key for a node.
44           */
45          public static Visibility get(AccessNode node) {
46              if (node.isPublic()) {
47                  return PUBLIC;
48              } else if (node.isPackagePrivate()) {
49                  return PACKAGE;
50              } else if (node.isProtected()) {
51                  return PROTECTED;
52              } else {
53                  return PRIVATE;
54              }
55          }
56      }
57  }
