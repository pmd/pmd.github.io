5   package net.sourceforge.pmd.lang.ast;
6   
7   import java.util.Arrays;
8   
9   /**
10   * Calculates from an absolute offset in the source file the line/column
11   * coordinate. This is needed as Rhino only offers absolute positions for each
12   * node. Some other languages like XML and Apex use this, too.
13   *
14   * Idea from:
15   * http://code.google.com/p/closure-compiler/source/browse/trunk/src/com/google/javascript/jscomp/SourceFile.java
16   */
17  public class SourceCodePositioner {
18  
19      private int[] lineOffsets;
20      private int sourceCodeLength;
21  
22      public SourceCodePositioner(String sourceCode) {
23          analyzeLineOffsets(sourceCode);
24      }
25  
26      private void analyzeLineOffsets(String sourceCode) {
27          String[] lines = sourceCode.split("\n");
28          sourceCodeLength = sourceCode.length();
29  
30          int startOffset = 0;
31          int lineNumber = 0;
32  
33          lineOffsets = new int[lines.length];
34  
35          for (String line : lines) {
36              lineOffsets[lineNumber] = startOffset;
37              lineNumber++;
38              startOffset += line.length() + 1; // +1 for the "\n" character
39          }
40      }
41  
42      public int lineNumberFromOffset(int offset) {
43          int search = Arrays.binarySearch(lineOffsets, offset);
44          int lineNumber;
45          if (search >= 0) {
46              lineNumber = search;
47          } else {
48              int insertionPoint = search;
49              insertionPoint += 1;
50              insertionPoint *= -1;
51              lineNumber = insertionPoint - 1; // take the insertion point one
52              // before
53          }
54          return lineNumber + 1; // 1-based line numbers
55      }
56  
57      public int columnFromOffset(int lineNumber, int offset) {
58          int lineIndex = lineNumber - 1;
59          if (lineIndex < 0 || lineIndex >= lineOffsets.length) {
60              // no line number found...
61              return 0;
62          }
63          int columnOffset = offset - lineOffsets[lineNumber - 1];
64          return columnOffset + 1; // 1-based column offsets
65      }
66  
67      public int getLastLine() {
68          return lineOffsets.length;
69      }
70  
71      public int getLastLineColumn() {
72          return columnFromOffset(getLastLine(), sourceCodeLength - 1);
73      }
74  }
