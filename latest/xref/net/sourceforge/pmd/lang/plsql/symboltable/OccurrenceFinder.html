5   package net.sourceforge.pmd.lang.plsql.symboltable;
6   
7   import java.util.HashSet;
8   import java.util.List;
9   import java.util.Set;
10  import java.util.logging.Level;
11  import java.util.logging.Logger;
12  
13  import net.sourceforge.pmd.lang.plsql.ast.ASTPrimaryExpression;
14  import net.sourceforge.pmd.lang.plsql.ast.PLSQLParserVisitorAdapter;
15  import net.sourceforge.pmd.lang.symboltable.NameDeclaration;
16  import net.sourceforge.pmd.lang.symboltable.Scope;
17  
18  public class OccurrenceFinder extends PLSQLParserVisitorAdapter {
19      private static final Logger LOGGER = Logger.getLogger(OccurrenceFinder.class.getName());
20  
21      @Override
22      public Object visit(ASTPrimaryExpression node, Object data) {
23          NameFinder nameFinder = new NameFinder(node);
24  
25          // Maybe do some sort of State pattern thingy for when NameDeclaration
26          // is empty/not empty
27          Set<NameDeclaration> declarations = new HashSet<>();
28  
29          List<PLSQLNameOccurrence> names = nameFinder.getNames();
30          for (PLSQLNameOccurrence occ : names) {
31              Search search = new Search(occ);
32              if (declarations.isEmpty()) {
33                  // doing the first name lookup
34                  search.execute();
35                  declarations.addAll(search.getResult());
36                  if (declarations.isEmpty()) {
37                      // we can't find it, so just give up
38                      // when we decide to do full symbol resolution
39                      // force this to either find a symbol or throw a
40                      // SymbolNotFoundException
41                      break;
42                  }
43              } else {
44                  Set<NameDeclaration> additionalDeclarations = new HashSet<>();
45                  for (NameDeclaration decl : declarations) {
46                      // now we've got a scope we're starting with, so work from
47                      // there
48                      Scope scope = decl.getScope();
49                      if (null == scope) {
50                          if (LOGGER.isLoggable(Level.FINEST)) {
51                              LOGGER.finest("NameOccurrence has no Scope:" + decl.getClass().getCanonicalName() + "=>"
52                                      + decl.getImage());
53                          }
54                          break;
55                      }
56                      search.execute(scope);
57                      Set<NameDeclaration> found = search.getResult();
58                      additionalDeclarations.addAll(found);
59                      if (found.isEmpty()) {
60                          // nothing found
61                          // This seems to be a lack of type resolution here.
62                          // Theoretically we have the previous declaration node
63                          // and know from there the Type of
64                          // the variable. The current occurrence (occ) should
65                          // then be found in the declaration of
66                          // this type. The type however may or may not be known
67                          // to PMD (see aux classpath).
68  
69                          // we can't find it, so just give up
70                          // when we decide to do full symbol resolution
71                          // force this to either find a symbol or throw a
72                          // SymbolNotFoundException
73                          break;
74                      }
75                  }
76                  declarations.addAll(additionalDeclarations);
77              }
78          }
79          return super.visit(node, data);
80      }
81  
82  }
