111     private void expandBuff(boolean wrapAround) {
112         char[] newbuffer = new char[bufsize + nextBufExpand];
113         int[] newbufline = new int[bufsize + nextBufExpand];
114         int[] newbufcolumn = new int[bufsize + nextBufExpand];
115 
116         try {
117             if (wrapAround) {
118                 System.arraycopy(buffer, tokenBegin, newbuffer, 0, bufsize - tokenBegin);
119                 System.arraycopy(buffer, 0, newbuffer, bufsize - tokenBegin, bufpos);
120                 buffer = newbuffer;
121 
122                 System.arraycopy(bufline, tokenBegin, newbufline, 0, bufsize - tokenBegin);
123                 System.arraycopy(bufline, 0, newbufline, bufsize - tokenBegin, bufpos);
124                 bufline = newbufline;
125 
126                 System.arraycopy(bufcolumn, tokenBegin, newbufcolumn, 0, bufsize - tokenBegin);
127                 System.arraycopy(bufcolumn, 0, newbufcolumn, bufsize - tokenBegin, bufpos);
128                 bufcolumn = newbufcolumn;
129 
130                 bufpos += bufsize - tokenBegin;
131                 maxNextCharInd = bufpos;
132             } else {
133                 System.arraycopy(buffer, tokenBegin, newbuffer, 0, bufsize - tokenBegin);
134                 buffer = newbuffer;
135 
136                 System.arraycopy(bufline, tokenBegin, newbufline, 0, bufsize - tokenBegin);
137                 bufline = newbufline;
138 
139                 System.arraycopy(bufcolumn, tokenBegin, newbufcolumn, 0, bufsize - tokenBegin);
140                 bufcolumn = newbufcolumn;
141 
142                 bufpos -= tokenBegin;
143                 maxNextCharInd = bufpos;
144             }
145         } catch (Throwable t) {
146             throw new Error(t.getMessage());
147         }
148 
149         bufsize += nextBufExpand;
150         nextBufExpand = bufsize;
151         available = bufsize;
152         tokenBegin = 0;
153     }
154 
155     private void fillBuff() throws java.io.IOException {
156         if (maxNextCharInd == available) {
157             if (available == bufsize) {
158                 if (tokenBegin > nextBufExpand) {
159                     bufpos = 0;
160                     maxNextCharInd = 0;
161                     available = tokenBegin;
162                 } else if (tokenBegin < 0) {
163                     bufpos = 0;
164                     maxNextCharInd = 0;
165                 } else {
166                     expandBuff(false);
167                 }
168             } else if (available > tokenBegin) {
169                 available = bufsize;
170             } else if ((tokenBegin - available) < nextBufExpand) {
171                 expandBuff(true);
172             } else {
173                 available = tokenBegin;
174             }
175         }
176 
177         int i;
178         try {
179             i = inputStream.read(buffer, maxNextCharInd, available - maxNextCharInd);
180             if (i == -1) {
181                 inputStream.close();
182                 throw new java.io.IOException();
183             } else {
184                 maxNextCharInd += i;
185             }
186             return;
187         } catch (java.io.IOException e) {
188             --bufpos;
189             backup(0);
190             if (tokenBegin == -1) {
191                 tokenBegin = bufpos;
192             }
193             throw e;
194         }
195     }

206     private void updateLineColumn(char c) {
207         column++;
208 
209         if (prevCharIsLF) {
210             prevCharIsLF = false;
211             column = 1;
212             line += 1;
213         } else if (prevCharIsCR) {
214             prevCharIsCR = false;
215             if (c == '\n') {
216                 prevCharIsLF = true;
217             } else {
218                 column = 1;
219                 line += 1;
220             }
221         }
222 
223         switch (c) {
224         case '\r':
225             prevCharIsCR = true;
226             break;
227         case '\n':
228             prevCharIsLF = true;
229             break;
230         case '\t':
231             column--;
232             column += 8 - (column & 07);
233             break;
234         default:
235             break;
236         }
237 
238         bufline[bufpos] = line;
239         bufcolumn[bufpos] = column;
240     }

242     @Override
243     public char readChar() throws java.io.IOException {
244         if (inBuf > 0) {
245             --inBuf;
246 
247             /*
248              * was : return (char)((char)0xff & buffer[(bufpos == bufsize - 1) ?
249              * (bufpos = 0) : ++bufpos]);
250              */
251             if (bufpos == bufsize - 1) {
252                 bufpos = 0;
253             } else {
254                 bufpos++;
255             }
256             return buffer[bufpos];
257         }
258 
259         bufpos++;
260         if (bufpos >= maxNextCharInd) {
261             fillBuff();
262         }
263 
264         /*
265          * was : char c = (char)((char)0xff & buffer[bufpos]);
266          */
267         char c = buffer[bufpos];
268 
269         updateLineColumn(c);
270         return c;
271     }

311     @Override
312     public void backup(int amount) {
313 
314         inBuf += amount;
315         bufpos -= amount;
316         if (bufpos < 0) {
317             bufpos += bufsize;
318         }
319     }

327     public void reInit(java.io.Reader dstream, int startline, int startcolumn, int buffersize) {
328         inputStream = dstream;
329         line = startline;
330         column = startcolumn - 1;
331 
332         if (buffer == null || buffersize != buffer.length) {
333             available = buffersize;
334             bufsize = buffersize;
335             nextBufExpand = buffersize;
336             buffer = new char[buffersize];
337             bufline = new int[buffersize];
338             bufcolumn = new int[buffersize];
339         }
340         prevCharIsLF = false;
341         prevCharIsCR = false;
342         tokenBegin = 0;
343         inBuf = 0;
344         maxNextCharInd = 0;
345         bufpos = -1;
346     }

376     @Override
377     public String GetImage() {
378         if (bufpos >= tokenBegin) {
379             return new String(buffer, tokenBegin, bufpos - tokenBegin + 1);
380         } else {
381             return new String(buffer, tokenBegin, bufsize - tokenBegin) + new String(buffer, 0, bufpos + 1);
382         }
383     }

385     @Override
386     public char[] GetSuffix(int len) {
387         char[] ret = new char[len];
388 
389         if ((bufpos + 1) >= len) {
390             System.arraycopy(buffer, bufpos - len + 1, ret, 0, len);
391         } else {
392             System.arraycopy(buffer, bufsize - (len - bufpos - 1), ret, 0, len - bufpos - 1);
393             System.arraycopy(buffer, 0, ret, len - bufpos - 1, bufpos + 1);
394         }
395 
396         return ret;
397     }

406     /**
407      * Method to adjust line and column numbers for the start of a token.<BR>
408      * 
409      * @param newLine
410      * @param newCol
411      */
412     public void adjustBeginLineColumn(int newLine, int newCol) {
413         int start = tokenBegin;
414         int len;
415 
416         if (bufpos >= tokenBegin) {
417             len = bufpos - tokenBegin + inBuf + 1;
418         } else {
419             len = bufsize - tokenBegin + bufpos + 1 + inBuf;
420         }
421 
422         int i = 0;
423         int j = start % bufsize;
424         int k = (start + 1) % bufsize;
425         int nextColDiff = 0;
426         int columnDiff = 0;
427 
428         while (i < len && bufline[j] == bufline[k]) {
429             bufline[j] = newLine;
430             nextColDiff = columnDiff + bufcolumn[k] - bufcolumn[j];
431             bufcolumn[j] = newCol + columnDiff;
432             columnDiff = nextColDiff;
433             i++;
434 
435             start++;
436             j = start % bufsize;
437             k = (start + 1) % bufsize;
438         }
439 
440         if (i < len) {
441             bufline[j] = newLine++;
442             bufcolumn[j] = newCol + columnDiff;
443 
444             while (i++ < len) {
445                 j = start % bufsize;
446                 start++;
447                 if (bufline[j] != bufline[start % bufsize]) {
448                     bufline[j] = newLine++;
449                 } else {
450                     bufline[j] = newLine;
451                 }
452             }
453         }
454 
455         line = bufline[j];
456         column = bufcolumn[j];
457     }
