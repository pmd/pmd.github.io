5   package net.sourceforge.pmd.lang.vm.rule.basic;
6   
7   import net.sourceforge.pmd.lang.vm.ast.ASTprocess;
8   import net.sourceforge.pmd.lang.vm.rule.AbstractStatisticalVmRule;
9   import net.sourceforge.pmd.stat.DataPoint;
10  
11  public class ExcessiveTemplateLengthRule extends AbstractStatisticalVmRule {
12  
13      @Override
14      public Object visit(final ASTprocess node, final Object data) {
15          final DataPoint point = new DataPoint();
16          point.setNode(node);
17          point.setScore(1.0 * (node.getEndLine() - node.getBeginLine()));
18          point.setMessage(getMessage());
19          addDataPoint(point);
20          return node.childrenAccept(this, data);
21      }
22  }
