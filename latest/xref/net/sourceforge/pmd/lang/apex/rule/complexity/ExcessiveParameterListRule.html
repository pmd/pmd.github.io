4   package net.sourceforge.pmd.lang.apex.rule.complexity;
5   
6   import net.sourceforge.pmd.lang.apex.ast.ASTMethod;
7   import net.sourceforge.pmd.lang.apex.ast.ASTParameter;
8   import net.sourceforge.pmd.util.NumericConstants;

10  /**
11   * This rule detects an abnormally long parameter list. Note: This counts Nodes,
12   * and not necessarily parameters, so the numbers may not match up. (But
13   * topcount and sigma should work.)
14   */
15  public class ExcessiveParameterListRule extends ExcessiveNodeCountRule {
16  	public ExcessiveParameterListRule() {
17  		super(ASTMethod.class);
18  		setProperty(MINIMUM_DESCRIPTOR, 4d);
19  		setProperty(CODECLIMATE_CATEGORIES, new String[]{ "Complexity" });
20  		setProperty(CODECLIMATE_REMEDIATION_MULTIPLIER, 50);
21  		setProperty(CODECLIMATE_BLOCK_HIGHLIGHTING, false);
22  	}
23  
24  	public Object visit(ASTParameter node, Object data) {
25  		return NumericConstants.ONE;
26  	}
27  }
