5   package net.sourceforge.pmd.stat;
6   
7   import java.util.Random;
8   
9   import net.sourceforge.pmd.lang.ast.Node;
10  
11  /**
12   * Datapoint used for rules that deal with metrics.
13   *
14   * @author David Dixon-Peugh
15   * @since Aug 8, 2002
16   */
17  public class DataPoint implements Comparable<DataPoint> {
18  
19      private Node node;
20      private int random;
21      private double score;
22      private String message;
23  
24      /**
25       * Constructor for DataPoint.
26       */
27      public DataPoint() {
28          super();
29          // Random number is so that the TreeSet doesn't
30          // whack things with the same score.
31          Random rand = new Random();
32          random = rand.nextInt(11061973);
33      }
34  
35      /**
36       * Compares this data point with the given datapoint.
37       *
38       * @param rhs
39       *            the other data point
40       * @return 0 if equal; a value less than 0 if this point's score is smaller
41       *         than the other data point; a value greater than 0 if this point's
42       *         score is greater than the other data point.
43       */
44      @Override
45      public int compareTo(DataPoint rhs) {
46          if (score != rhs.getScore()) {
47              return Double.compare(score, rhs.getScore());
48          }
49          return random - rhs.random;
50      }
51  
52      public Node getNode() {
53          return node;
54      }
55  
56      public void setNode(Node node) {
57          this.node = node;
58      }
59  
60      public String getMessage() {
61          return message;
62      }
63  
64      public void setMessage(String message) {
65          this.message = message;
66      }
67  
68      public double getScore() {
69          return score;
70      }
71  
72      public void setScore(double score) {
73          this.score = score;
74      }
75  }
