5   package net.sourceforge.pmd.lang.java.rule.optimizations;
6   
7   import java.util.Set;
8   
9   import net.sourceforge.pmd.RuleContext;
10  import net.sourceforge.pmd.lang.LanguageRegistry;
11  import net.sourceforge.pmd.lang.ast.Node;
12  import net.sourceforge.pmd.lang.java.JavaLanguageModule;
13  import net.sourceforge.pmd.lang.java.ast.ASTName;
14  import net.sourceforge.pmd.lang.java.ast.ASTPrimaryExpression;
15  import net.sourceforge.pmd.lang.java.ast.ASTPrimaryPrefix;
16  import net.sourceforge.pmd.lang.java.ast.ASTPrimarySuffix;
17  import net.sourceforge.pmd.lang.java.rule.AbstractJavaRule;
18  import net.sourceforge.pmd.util.CollectionUtil;
19  
20  public class UnnecessaryWrapperObjectCreationRule extends AbstractJavaRule {
21  
22      private static final Set<String> PREFIX_SET = CollectionUtil.asSet(new String[] { "Byte.valueOf", "Short.valueOf",
23          "Integer.valueOf", "Long.valueOf", "Float.valueOf", "Double.valueOf", "Character.valueOf", });
24  
25      private static final Set<String> SUFFIX_SET = CollectionUtil.asSet(new String[] { "toString", "byteValue",
26          "shortValue", "intValue", "longValue", "floatValue", "doubleValue", "charValue", });
27  
28      public Object visit(ASTPrimaryPrefix node, Object data) {
29          if (node.jjtGetNumChildren() == 0 || !(node.jjtGetChild(0) instanceof ASTName)) {
30              return super.visit(node, data);
31          }
32  
33          String image = ((ASTName) node.jjtGetChild(0)).getImage();
34          if (image.startsWith("java.lang.")) {
35              image = image.substring(10);
36          }
37  
38          boolean checkBoolean = ((RuleContext) data).getLanguageVersion()
39                  .compareTo(LanguageRegistry.getLanguage(JavaLanguageModule.NAME).getVersion("1.5")) >= 0;
40  
41          if (PREFIX_SET.contains(image) || checkBoolean && "Boolean.valueOf".equals(image)) {
42              ASTPrimaryExpression parent = (ASTPrimaryExpression) node.jjtGetParent();
43              if (parent.jjtGetNumChildren() >= 3) {
44                  Node n = parent.jjtGetChild(2);
45                  if (n instanceof ASTPrimarySuffix) {
46                      ASTPrimarySuffix suffix = (ASTPrimarySuffix) n;
47                      image = suffix.getImage();
48  
49                      if (SUFFIX_SET.contains(image) || checkBoolean && "booleanValue".equals(image)) {
50                          super.addViolation(data, node);
51                          return data;
52                      }
53                  }
54              }
55          }
56          return super.visit(node, data);
57      }
58  
59  }
