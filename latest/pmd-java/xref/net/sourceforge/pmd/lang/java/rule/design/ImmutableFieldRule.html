32  /**
33   * @author Olander
34   */
35  public class ImmutableFieldRule extends AbstractJavaRule {
36  
37      private enum FieldImmutabilityType {
38          /** Variable is changed in methods and/or in lambdas */
39          MUTABLE,
40          /** Variable is not changed outside the constructor. */
41          IMMUTABLE,
42          /** Variable is only written during declaration, if at all. */
43          CHECKDECL;
44      }
45  
46      @Override
47      public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
48          Map<VariableNameDeclaration, List<NameOccurrence>> vars = node.getScope()
49                  .getDeclarations(VariableNameDeclaration.class);
50          List<ASTConstructorDeclaration> constructors = findAllConstructors(node);
51          for (Map.Entry<VariableNameDeclaration, List<NameOccurrence>> entry : vars.entrySet()) {
52              VariableNameDeclaration field = entry.getKey();
53              AccessNode accessNodeParent = field.getAccessNodeParent();
54              if (accessNodeParent.isStatic() || !accessNodeParent.isPrivate() || accessNodeParent.isFinal()
55                      || accessNodeParent.isVolatile()) {
56                  continue;
57              }
58  
59              FieldImmutabilityType result = initializedInConstructor(entry.getValue(), new HashSet<>(constructors));
60              if (result == FieldImmutabilityType.MUTABLE) {
61                  continue;
62              }
63              if (result == FieldImmutabilityType.IMMUTABLE || result == FieldImmutabilityType.CHECKDECL && initializedWhenDeclared(field)) {
64                  addViolation(data, field.getNode(), field.getImage());
65              }
66          }
67          return super.visit(node, data);
68      }
69  
70      private boolean initializedWhenDeclared(VariableNameDeclaration field) {
71          return ((Node) field.getAccessNodeParent()).hasDescendantOfType(ASTVariableInitializer.class);
72      }
73  
74      private FieldImmutabilityType initializedInConstructor(List<NameOccurrence> usages, Set<ASTConstructorDeclaration> allConstructors) {
75          FieldImmutabilityType result = FieldImmutabilityType.MUTABLE;
76          int methodInitCount = 0;
77          int lambdaUsage = 0;
78          Set<Node> consSet = new HashSet<>();
79          for (NameOccurrence occ : usages) {
80              JavaNameOccurrence jocc = (JavaNameOccurrence) occ;
81              if (jocc.isOnLeftHandSide() || jocc.isSelfAssignment()) {
82                  Node node = jocc.getLocation();
83                  ASTConstructorDeclaration constructor = node.getFirstParentOfType(ASTConstructorDeclaration.class);
84                  if (constructor != null) {
85                      if (inLoopOrTry(node)) {
86                          continue;
87                      }
88                      // Check for assigns in if-statements, which can depend on
89                      // constructor
90                      // args or other runtime knowledge and can be a valid reason
91                      // to instantiate
92                      // in one constructor only
93                      if (node.getFirstParentOfType(ASTIfStatement.class) != null) {
94                          methodInitCount++;
95                      }
96                      if (inAnonymousInnerClass(node)) {
97                          methodInitCount++;
98                      } else {
99                          consSet.add(constructor);
100                     }
101                 } else {
102                     if (node.getFirstParentOfType(ASTMethodDeclaration.class) != null) {
103                         methodInitCount++;
104                     } else if (node.getFirstParentOfType(ASTLambdaExpression.class) != null) {
105                         lambdaUsage++;
106                     }
107                 }
108             }
109         }
110         if (usages.isEmpty() || methodInitCount == 0 && lambdaUsage == 0 && consSet.isEmpty()) {
111             result = FieldImmutabilityType.CHECKDECL;
112         } else {
113             allConstructors.removeAll(consSet);
114             if (allConstructors.isEmpty() && methodInitCount == 0 && lambdaUsage == 0) {
115                 result = FieldImmutabilityType.IMMUTABLE;
116             }
117         }
118         return result;
119     }
120 
121     private boolean inLoopOrTry(Node node) {
122         return node.getFirstParentOfType(ASTTryStatement.class) != null
123                 || node.getFirstParentOfType(ASTForStatement.class) != null
124                 || node.getFirstParentOfType(ASTWhileStatement.class) != null
125                 || node.getFirstParentOfType(ASTDoStatement.class) != null;
126     }
127 
128     private boolean inAnonymousInnerClass(Node node) {
129         ASTClassOrInterfaceBodyDeclaration parent = node.getFirstParentOfType(ASTClassOrInterfaceBodyDeclaration.class);
130         return parent != null && parent.isAnonymousInnerClass();
131     }
132 
133     private List<ASTConstructorDeclaration> findAllConstructors(ASTClassOrInterfaceDeclaration node) {
134         List<ASTConstructorDeclaration> cons = new ArrayList<>();
135         node.getFirstChildOfType(ASTClassOrInterfaceBody.class).findDescendantsOfType(ASTConstructorDeclaration.class,
136                 cons, false);
137         return cons;
138     }
