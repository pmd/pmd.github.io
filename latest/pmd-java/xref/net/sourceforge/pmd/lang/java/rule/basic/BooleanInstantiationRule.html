21  /**
22   * Avoid instantiating Boolean objects; you can reference Boolean.TRUE,
23   * Boolean.FALSE, or call Boolean.valueOf() instead.
24   *
25   * <pre>
26   *  public class Foo {
27   *       Boolean bar = new Boolean("true");    // just do a Boolean
28   *       bar = Boolean.TRUE;                   //ok
29   *       Boolean buz = Boolean.valueOf(false); // just do a Boolean buz = Boolean.FALSE;
30   *  }
31   * </pre>
32   */
33  public class BooleanInstantiationRule extends AbstractJavaRule {
34  
35  	/*
36  	 *  see bug 1744065 : If somebody create it owns Boolean, the rule should not be triggered
37  	 *   Therefore, we use this boolean to flag if the source code contains such an import
38  	 *
39  	 */
40  	private boolean customBoolean;
41  
42      @Override
43      public Object visit(ASTCompilationUnit decl,Object data) {
44          // customBoolean needs to be reset for each new file
45          customBoolean = false;
46  
47          return super.visit(decl, data);
48      }
49  
50  	@Override
51  	public Object visit(ASTImportDeclaration decl,Object data) {
52  		// If the import actually import a Boolean class that overrides java.lang.Boolean
53  		if ( decl.getImportedName().endsWith("Boolean") && ! decl.getImportedName().equals("java.lang"))
54  		{
55  			customBoolean = true;
56  		}
57  		return super.visit(decl, data);
58  	}
59  
60      @Override
61      public Object visit(ASTAllocationExpression node, Object data) {
62  
63      	if ( ! customBoolean ) {
64  	        if (node.hasDescendantOfType(ASTArrayDimsAndInits.class)) {
65  	            return super.visit(node, data);
66  	        }
67  	        
68  	        Node n1 = node.getFirstChildOfType(ASTClassOrInterfaceType.class);
69  	    	if (TypeHelper.isA((ASTClassOrInterfaceType) n1, Boolean.class)) {
70                  super.addViolation(data, node);
71                  return data;
72              }
73      	}
74          return super.visit(node, data);
75      }
76  
77      @Override
78      public Object visit(ASTPrimaryPrefix node, Object data) {
79  
80      	if ( ! customBoolean )
81      	{
82  	        if (node.jjtGetNumChildren() == 0 || !(node.jjtGetChild(0) instanceof ASTName)) {
83  	            return super.visit(node, data);
84  	        }
85  
86  	        if ("Boolean.valueOf".equals(((ASTName) node.jjtGetChild(0)).getImage())
87  	                || "java.lang.Boolean.valueOf".equals(((ASTName) node.jjtGetChild(0)).getImage())) {
88  	            ASTPrimaryExpression parent = (ASTPrimaryExpression) node.jjtGetParent();
89  	            ASTPrimarySuffix suffix = parent.getFirstDescendantOfType(ASTPrimarySuffix.class);
90  	            if (suffix == null) {
91  	                return super.visit(node, data);
92  	            }
93  	            ASTPrimaryPrefix prefix = suffix.getFirstDescendantOfType(ASTPrimaryPrefix.class);
94  	            if (prefix == null) {
95  	                return super.visit(node, data);
96  	            }
97  
98  	            if (prefix.hasDescendantOfType(ASTBooleanLiteral.class)) {
99  	                super.addViolation(data, node);
100 	                return data;
101 	            }
102 	            ASTLiteral literal = prefix.getFirstDescendantOfType(ASTLiteral.class);
103 	            if (literal != null && ("\"true\"".equals(literal.getImage()) || "\"false\"".equals(literal.getImage()))) {
104 	                super.addViolation(data, node);
105 	                return data;
106 	            }
107 	        }
108     	}
109         return super.visit(node, data);
110     }
111 }
