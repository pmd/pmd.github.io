27  public class BrokenNullCheckRule extends AbstractJavaRule {
28  
29      @Override
30      public Object visit(ASTIfStatement node, Object data) {
31          ASTExpression expression = (ASTExpression) node.jjtGetChild(0);
32  
33          ASTConditionalAndExpression conditionalAndExpression = expression
34                  .getFirstDescendantOfType(ASTConditionalAndExpression.class);
35          if (conditionalAndExpression != null) {
36              checkForViolations(node, data, conditionalAndExpression);
37          }
38  
39          ASTConditionalOrExpression conditionalOrExpression = expression
40                  .getFirstDescendantOfType(ASTConditionalOrExpression.class);
41          if (conditionalOrExpression != null) {
42              checkForViolations(node, data, conditionalOrExpression);
43          }
44  
45          return super.visit(node, data);
46      }
47  
48      private void checkForViolations(ASTIfStatement node, Object data, Node conditionalExpression) {
49          ASTEqualityExpression equalityExpression = conditionalExpression
50                  .getFirstChildOfType(ASTEqualityExpression.class);
51          if (equalityExpression == null) {
52              return;
53          }
54          if (conditionalExpression instanceof ASTConditionalAndExpression
55                  && !"==".equals(equalityExpression.getImage())) {
56              return;
57          }
58          if (conditionalExpression instanceof ASTConditionalOrExpression
59                  && !"!=".equals(equalityExpression.getImage())) {
60              return;
61          }
62          ASTNullLiteral nullLiteral = equalityExpression.getFirstDescendantOfType(ASTNullLiteral.class);
63          if (nullLiteral == null) {
64              return; // No null check
65          }
66          // If there is an assignment in the equalityExpression we give up,
67          // because things get too complex
68          if (conditionalExpression.hasDescendantOfType(ASTAssignmentOperator.class)) {
69              return;
70          }
71  
72          // Find the expression used in the null compare
73          ASTPrimaryExpression nullCompareExpression = findNullCompareExpression(equalityExpression);
74          if (nullCompareExpression == null) {
75              return; // No good null check
76          }
77  
78          // Now we find the expression to compare to and do the comparison
79          for (int i = 0; i < conditionalExpression.jjtGetNumChildren(); i++) {
80              Node conditionalSubnode = conditionalExpression.jjtGetChild(i);
81  
82              // We skip the null compare branch
83              ASTEqualityExpression nullEqualityExpression = nullLiteral
84                      .getFirstParentOfType(ASTEqualityExpression.class);
85              if (conditionalSubnode.equals(nullEqualityExpression)) {
86                  continue;
87              }
88              ASTPrimaryExpression conditionalPrimaryExpression;
89              if (conditionalSubnode instanceof ASTPrimaryExpression) {
90                  conditionalPrimaryExpression = (ASTPrimaryExpression) conditionalSubnode;
91              } else {
92                  // The ASTPrimaryExpression is hidden (in a negation, braces or
93                  // EqualityExpression)
94                  conditionalPrimaryExpression = conditionalSubnode.getFirstDescendantOfType(ASTPrimaryExpression.class);
95              }
96  
97              if (primaryExpressionsAreEqual(nullCompareExpression, conditionalPrimaryExpression)) {
98                  addViolation(data, node); // We have a match
99              }
100 
101         }
102     }
103 
104     private boolean primaryExpressionsAreEqual(ASTPrimaryExpression nullCompareVariable,
105             ASTPrimaryExpression expressionUsage) {
106         List<String> nullCompareNames = new ArrayList<>();
107         findExpressionNames(nullCompareVariable, nullCompareNames);
108 
109         List<String> expressionUsageNames = new ArrayList<>();
110         findExpressionNames(expressionUsage, expressionUsageNames);
111 
112         for (int i = 0; i < nullCompareNames.size(); i++) {
113             if (expressionUsageNames.size() == i) {
114                 // The used expression is shorter than the null
115                 // compare expression (and we don't want to crash
116                 // below)
117                 return false;
118             }
119 
120             String nullCompareExpressionName = nullCompareNames.get(i);
121             String expressionUsageName = expressionUsageNames.get(i);
122 
123             // Variablenames should match or the expressionUsage should have the
124             // variable with a method call (ie. var.equals())
125             if (!nullCompareExpressionName.equals(expressionUsageName)
126                     && !expressionUsageName.startsWith(nullCompareExpressionName + ".")) {
127                 // Some other expression is being used after the
128                 // null compare
129                 return false; 
130             }
131         }
132 
133         return true;
134     }

136     /**
137      * Find the names of variables, methods and array arguments in a
138      * PrimaryExpression.
139      */
140     private void findExpressionNames(Node nullCompareVariable, List<String> results) {
141         for (int i = 0; i < nullCompareVariable.jjtGetNumChildren(); i++) {
142             Node child = nullCompareVariable.jjtGetChild(i);
143 
144             if (child instanceof ASTName) {
145                 // Variable names and some method calls
146                 results.add(((ASTName) child).getImage());
147             } else if (child instanceof ASTLiteral) { // Array arguments
148                 String literalImage = ((ASTLiteral) child).getImage();
149                 // Skip other null checks
150                 if (literalImage != null) {
151                     results.add(literalImage);
152                 }
153             } else if (child instanceof ASTPrimarySuffix) { // More method calls
154                 String name = ((ASTPrimarySuffix) child).getImage();
155                 if (StringUtil.isNotEmpty(name)) {
156                     results.add(name);
157                 }
158             } else if (child instanceof ASTClassOrInterfaceType) {
159                 // A class can be an argument too
160                 String name = ((ASTClassOrInterfaceType) child).getImage();
161                 results.add(name);
162             }
163 
164             if (child.jjtGetNumChildren() > 0) {
165                 findExpressionNames(child, results);
166             }
167         }
168     }
169 
170     private ASTPrimaryExpression findNullCompareExpression(ASTEqualityExpression equalityExpression) {
171         List<ASTPrimaryExpression> primaryExpressions = equalityExpression
172                 .findDescendantsOfType(ASTPrimaryExpression.class);
173         for (ASTPrimaryExpression primaryExpression : primaryExpressions) {
174             List<ASTPrimaryPrefix> primaryPrefixes = primaryExpression.findDescendantsOfType(ASTPrimaryPrefix.class);
175             for (ASTPrimaryPrefix primaryPrefix : primaryPrefixes) {
176                 if (primaryPrefix.hasDescendantOfType(ASTName.class)) {
177                     // We found the variable that is compared to null
178                     return primaryExpression;
179                 }
180             }
181         }
182         return null; // Nothing found
183     }
