29      /**
30       * {@inheritDoc}
31       */
32      @Override
33      public void start() throws IOException {
34      	getWriter().write(header());
35      }

37      /**
38       * {@inheritDoc}
39       */
40      @Override
41      public void renderFileViolations(Iterator<RuleViolation> violations) throws IOException {
42  		if (!violations.hasNext()) {
43  		    return;
44  		}
45  	
46  		Writer writer = getWriter();
47  		StringBuilder sb = new StringBuilder(500);
48  		String filename = null;
49  		String lineSep = PMD.EOL;
50  	
51  		boolean colorize = false;
52  		while (violations.hasNext()) {
53  		    sb.setLength(0);
54  		    RuleViolation rv = violations.next();
55  			if (!rv.getFilename().equals(filename)) { // New File
56  				if (filename != null) {
57  				    sb.append("</table></br>");
58  				    colorize = false;
59  				}
60  				filename = rv.getFilename();
61  				sb.append("<table border=\"0\" width=\"80%\">");
62  				sb.append("<tr id=TableHeader><td colspan=\"2\"><font class=title>&nbsp;").append(filename).append(
63  					"</font></tr>");
64  				sb.append(lineSep);
65  			    }
66  	
67  		    if (colorize) {
68  				sb.append("<tr id=RowColor1>");
69  			    } else {
70  				sb.append("<tr id=RowColor2>");
71  			    }
72  	
73  		    colorize = !colorize;
74  		    sb.append("<td width=\"50\" align=\"right\"><font class=body>" + rv.getBeginLine()
75  			    + "&nbsp;&nbsp;&nbsp;</font></td>");
76  		    sb.append("<td><font class=body>" + rv.getDescription() + "</font></td>");
77  		    sb.append("</tr>");
78  		    sb.append(lineSep);
79  		    writer.write(sb.toString());
80  			}
81  		if (filename != null) {
82  		    writer.write("</table>");
83  		}
84      }
85  
86      /**
87       * {@inheritDoc}
88       */
89      @Override
90      public void end() throws IOException {
91  		Writer writer = getWriter();
92  		StringBuilder sb = new StringBuilder();
93  	
94  		writer.write("<br>");
95  	
96  		// output the problems
97  		if (!errors.isEmpty()) {
98  		    sb.setLength(0);
99  		    sb.append("<table border=\"0\" width=\"80%\">");
100 		    sb.append("<tr id=TableHeader><td><font class=title>&nbsp;Problems found</font></td></tr>");
101 		    boolean colorize = false;
102 		    for (Report.ProcessingError error : errors) {
103 			if (colorize) {
104 			    sb.append("<tr id=RowColor1>");
105 			} else {
106 			    sb.append("<tr id=RowColor2>");
107 			}
108 			colorize = !colorize;
109 			sb.append("<td><font class=body>").append(error).append("\"</font></td></tr>");
110 		    }
111 		    sb.append("</table>");
112 		    writer.write(sb.toString());
113 		}
114 	
115 		writer.write(footer());
116     }

118     private String header() {
119     	StringBuilder sb = new StringBuilder(600);
120 		sb.append("<html><head><title>PMD</title></head>");
121 		sb.append("<style type=\"text/css\">");
122 		sb.append("<!--" + PMD.EOL);

124 				+ PMD.EOL);
125 		sb.append(".title { font-family: verdana, arial, helvetica,geneva; font-size: 12px; font-weight:bold; color: white; }"
126 				+ PMD.EOL);
127 		sb.append(".body { font-family: verdana, arial, helvetica, geneva; font-size: 12px; font-weight:plain; color: black; }"
128 				+ PMD.EOL);
129 		sb.append("#TableHeader { background-color: #003366; }" + PMD.EOL);
130 		sb.append("#RowColor1 { background-color: #eeeeee; }" + PMD.EOL);
131 		sb.append("#RowColor2 { background-color: white; }" + PMD.EOL);
132 		sb.append("-->");
133 		sb.append("</style>");
134 		sb.append("<body><center>");
135 		return sb.toString();
136     }
