474     private static <T> List<T> convertList(List<Node> nodes, Class<T> target) {
475         List<T> converted = new ArrayList<>();
476         for (Node n : nodes) {
477             converted.add(target.cast(n));
478         }
479         return converted;
480     }

524     /**
525      * The type should be filled also on the ASTVariableDeclaratorId node,
526      * not only on the variable name declaration.
527      */
528     @Test
529     public void testFullyQualifiedType() {
530         String source = "public class Foo {\n" +
531                 "    public void bar() {\n" +
532                 "        java.util.StringTokenizer st = new StringTokenizer(\"a.b.c.d\", \".\");\n" +
533                 "        while (st.hasMoreTokens()) {\n" + 
534                 "            System.out.println(st.nextToken());\n" + 
535                 "        }\n" +
536                 "    }\n" +
537                 "}";
538         ASTCompilationUnit acu = parseAndTypeResolveForString(source, "1.5");
539         List<ASTName> names = acu.findDescendantsOfType(ASTName.class);
540         ASTName theStringTokenizer = null;
541         for (ASTName name : names) {
542             if (name.hasImageEqualTo("st.hasMoreTokens")) {
543                 theStringTokenizer = name;
544                 break;
545             }
546         }
547         Assert.assertNotNull(theStringTokenizer);
548         VariableNameDeclaration declaration = (VariableNameDeclaration)theStringTokenizer.getNameDeclaration();
549         Assert.assertNotNull(declaration);
550         Assert.assertEquals("java.util.StringTokenizer", declaration.getTypeImage());
551         Assert.assertNotNull(declaration.getType());
552         Assert.assertSame(StringTokenizer.class, declaration.getType());
553         ASTVariableDeclaratorId id = (ASTVariableDeclaratorId)declaration.getNode();
554         Assert.assertNotNull(id.getType());
555         Assert.assertSame(StringTokenizer.class, id.getType());
556     }

562     // Note: If you're using Eclipse or some other IDE to run this test, you _must_ have the src/test/java folder in
563     // the classpath.  Normally the IDE doesn't put source directories themselves directly in the classpath, only
564     // the output directories are in the classpath.
565     private ASTCompilationUnit parseAndTypeResolveForClass(Class<?> clazz, String version) {
566         String sourceFile = clazz.getName().replace('.', '/') + ".java";
567         InputStream is = ClassTypeResolverTest.class.getClassLoader().getResourceAsStream(sourceFile);
568         if (is == null) {
569             throw new IllegalArgumentException("Unable to find source file " + sourceFile + " for " + clazz);
570         }
571         String source;
572         try {
573             source = IOUtils.toString(is);
574         } catch (IOException e) {
575             throw new RuntimeException(e);
576         }
577         return parseAndTypeResolveForString(source, version);
578     }

580     private ASTCompilationUnit parseAndTypeResolveForString(String source, String version) {
581         LanguageVersionHandler languageVersionHandler = LanguageRegistry.getLanguage(JavaLanguageModule.NAME).getVersion(version).getLanguageVersionHandler();
582         ASTCompilationUnit acu = (ASTCompilationUnit) languageVersionHandler.getParser(languageVersionHandler.getDefaultParserOptions()).parse(null, new StringReader(source));
583         languageVersionHandler.getSymbolFacade().start(acu);
584         languageVersionHandler.getTypeResolutionFacade(ClassTypeResolverTest.class.getClassLoader()).start(acu);
585         return acu;
586     }
