159 	private void doTask() {
160 		setupClassLoader();
161 
162 		// Setup RuleSetFactory and validate RuleSets
163 		RuleSetFactory ruleSetFactory = new RuleSetFactory();
164 		ruleSetFactory.setClassLoader(configuration.getClassLoader());
165 		try {
166 			// This is just used to validate and display rules. Each thread will create its own ruleset
167 			ruleSetFactory.setMinimumPriority(configuration.getMinimumPriority());
168 			ruleSetFactory.setWarnDeprecated(true);
169 			String ruleSets = configuration.getRuleSets();
170 			if (StringUtil.isNotEmpty(ruleSets)) {
171 				// Substitute env variables/properties
172 				configuration.setRuleSets(getProject().replaceProperties(ruleSets));
173 			}
174 			RuleSets rules = ruleSetFactory.createRuleSets(configuration.getRuleSets());
175 			ruleSetFactory.setWarnDeprecated(false);
176 			logRulesUsed(rules);
177 		} catch (RuleSetNotFoundException e) {
178 			throw new BuildException(e.getMessage(), e);
179 		}
180 
181 		if (configuration.getSuppressMarker() != null) {
182 			log("Setting suppress marker to be " + configuration.getSuppressMarker(), Project.MSG_VERBOSE);
183 		}
184 
185 		// Start the Formatters
186 		for (Formatter formatter : formatters) {
187 			log("Sending a report to " + formatter, Project.MSG_VERBOSE);
188 			formatter.start(getProject().getBaseDir().toString());
189 		}
190 
191 		//log("Setting Language Version " + languageVersion.getShortName(), Project.MSG_VERBOSE);
192 
193 		// TODO Do we really need all this in a loop over each FileSet?  Seems like a lot of redundancy
194 		RuleContext ctx = new RuleContext();
195 		Report errorReport = new Report();
196 		final AtomicInteger reportSize = new AtomicInteger();
197 		final String separator = System.getProperty("file.separator");
198 
199 		for (FileSet fs : filesets) {
200 			List<DataSource> files = new LinkedList<DataSource>();
201 			DirectoryScanner ds = fs.getDirectoryScanner(getProject());
202 			String[] srcFiles = ds.getIncludedFiles();
203 			for (String srcFile : srcFiles) {
204 				File file = new File(ds.getBasedir() + separator + srcFile);
205 				files.add(new FileDataSource(file));
206 			}
207 
208 			final String inputPaths = ds.getBasedir().getPath();
209 			configuration.setInputPaths(inputPaths);
210 
211 			Renderer logRenderer = new AbstractRenderer("log", "Logging renderer") {
212 				public void start() {
213 					// Nothing to do
214 				}
215 
216 				public void startFileAnalysis(DataSource dataSource) {
217 					log("Processing file " + dataSource.getNiceFileName(false, inputPaths), Project.MSG_VERBOSE);
218 				}
219 
220 				public void renderFileReport(Report r) {
221 					int size = r.size();
222 					if (size > 0) {
223 						reportSize.addAndGet(size);
224 					}
225 				}
226 
227 				public void end() {
228 					// Nothing to do
229 				}
230 
231 				public String defaultFileExtension() { return null;	}	// not relevant
232 			};
233 			List<Renderer> renderers = new LinkedList<Renderer>();
234 			renderers.add(logRenderer);
235 			for (Formatter formatter : formatters) {
236 				renderers.add(formatter.getRenderer());
237 			}
238 			try {
239 				PMD.processFiles(configuration, ruleSetFactory, files, ctx, renderers);
240 			} catch (RuntimeException pmde) {
241 				handleError(ctx, errorReport, pmde);
242 			}
243 		}
244 
245 		int problemCount = reportSize.get();
246 		log(problemCount + " problems found", Project.MSG_VERBOSE);
247 
248 		for (Formatter formatter : formatters) {
249 			formatter.end(errorReport);
250 		}
251 
252 		if (failuresPropertyName != null && problemCount > 0) {
253 			getProject().setProperty(failuresPropertyName, String.valueOf(problemCount));
254 			log("Setting property " + failuresPropertyName + " to " + problemCount, Project.MSG_VERBOSE);
255 		}
256 
257 		if (failOnRuleViolation && problemCount > maxRuleViolations) {
258 			throw new BuildException("Stopping build since PMD found " + problemCount + " rule violations in the code");
259 		}
260 	}
261 
262 	private void handleError(RuleContext ctx, Report errorReport, RuntimeException pmde) {
263 		
264 		pmde.printStackTrace();
265 		log(pmde.toString(), Project.MSG_VERBOSE);
266 		
267 		Throwable cause = pmde.getCause();
268 		
269 		if (cause != null) {
270 			StringWriter strWriter = new StringWriter();
271 			PrintWriter printWriter = new PrintWriter(strWriter);
272 			cause.printStackTrace(printWriter);
273 			log(strWriter.toString(), Project.MSG_VERBOSE);
274 			IOUtils.closeQuietly(printWriter);
275 			
276 			if (StringUtil.isNotEmpty(cause.getMessage())) {
277 				log(cause.getMessage(), Project.MSG_VERBOSE);
278 			}
279 		}
280 		
281 		if (failOnError) {
282 			throw new BuildException(pmde);
283 		}
284 		errorReport.addError(new Report.ProcessingError(pmde.getMessage(), ctx.getSourceCodeFilename()));
285 	}

287 	private void setupClassLoader() {
288 
289 		if (classpath == null) {
290 			log("Using the normal ClassLoader", Project.MSG_VERBOSE);
291 		} else {
292 			log("Using the AntClassLoader", Project.MSG_VERBOSE);
293 			// must be true, otherwise you'll get ClassCastExceptions as classes are loaded twice
294 			// and exist in multiple class loaders
295 			boolean parentFirst = true;
296 			configuration.setClassLoader(
297 			        new AntClassLoader(Thread.currentThread().getContextClassLoader(), getProject(),
298 			                classpath, parentFirst));
299 		}
300 		try {
301 			/*
302 			 * 'basedir' is added to the path to make sure that relative paths
303 			 * such as "<ruleset>resources/custom_ruleset.xml</ruleset>" still
304 			 * work when ant is invoked from a different directory using "-f"
305 			 */
306 			configuration.prependClasspath(getProject().getBaseDir().toString());
307 			if (auxClasspath != null) {
308 				log("Using auxclasspath: " + auxClasspath, Project.MSG_VERBOSE);
309 				configuration.prependClasspath(auxClasspath.toString());
310 			}
311 		} catch (IOException ioe) {
312 			throw new BuildException(ioe.getMessage(), ioe);
313 		}
314 	}

316 	@Override
317 	public void execute() throws BuildException {
318 		validate();
319 		final Handler antLogHandler = new AntLogHandler(this);
320 		final ScopedLogHandlersManager logManager = new ScopedLogHandlersManager(Level.FINEST, antLogHandler);
321 		try {
322 			doTask();
323 		} finally {
324 			logManager.close();
325 		}
326 	}

328 	private void logRulesUsed(RuleSets rules) {
329 		log("Using these rulesets: " + configuration.getRuleSets(), Project.MSG_VERBOSE);
330 
331 		RuleSet[] ruleSets = rules.getAllRuleSets();
332 		for (RuleSet ruleSet : ruleSets) {
333 			for (Rule rule : ruleSet.getRules()) {
334 				log("Using rule " + rule.getName(), Project.MSG_VERBOSE);
335 			}
336 		}
337 	}

339 	private void validate() throws BuildException {
340 		if (formatters.isEmpty()) {
341 			Formatter defaultFormatter = new Formatter();
342 			defaultFormatter.setType("text");
343 			defaultFormatter.setToConsole(true);
344 			formatters.add(defaultFormatter);
345 		} else {
346 			for (Formatter f : formatters) {
347 				if (f.isNoOutputSupplied()) {
348 					throw new BuildException("toFile or toConsole needs to be specified in Formatter");
349 				}
350 			}
351 		}
352 
353 		if (configuration.getRuleSets() == null) {
354 			if (nestedRules.isEmpty()) {
355 				throw new BuildException("No rulesets specified");
356 			}
357 			configuration.setRuleSets(getNestedRuleSetFiles());
358 		}
359 	}
360 
361 	private String getNestedRuleSetFiles() {
362 		final StringBuilder sb = new StringBuilder();
363 		for (Iterator<RuleSetWrapper> it = nestedRules.iterator(); it.hasNext();) {
364 			RuleSetWrapper rs = it.next();
365 			sb.append(rs.getFile());
366 			if (it.hasNext()) {
367 				sb.append(',');
368 			}
369 		}
370 		return sb.toString();
371 	}
