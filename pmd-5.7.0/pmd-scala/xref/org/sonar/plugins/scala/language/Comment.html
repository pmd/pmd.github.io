21  package org.sonar.plugins.scala.language;
22  
23  import java.io.IOException;
24  import java.util.List;
25  
26  import org.apache.commons.lang3.builder.EqualsBuilder;
27  import org.apache.commons.lang3.builder.HashCodeBuilder;
28  import org.apache.commons.lang3.builder.ToStringBuilder;
29  import org.sonar.plugins.scala.util.StringUtils;
30  
31  /**
32   * This class implements a Scala comment and the computation of several base
33   * metrics for a comment.
34   *
35   * @author Felix MÃ¼ller
36   * @since 0.1
37   */
38  public class Comment {
39  
40      private final CommentType type;
41      private final List<String> lines;
42  
43      public Comment(String content, CommentType type) throws IOException {
44          lines = StringUtils.convertStringToListOfLines(content);
45          this.type = type;
46      }
47  
48      public int getNumberOfLines() {
49          return lines.size() - getNumberOfBlankLines() - getNumberOfCommentedOutLinesOfCode();
50      }
51  
52      public int getNumberOfBlankLines() {
53          int numberOfBlankLines = 0;
54          for (String comment : lines) {
55              boolean isBlank = true;
56  
57              for (int i = 0; isBlank && i < comment.length(); i++) {
58                  char character = comment.charAt(i);
59                  if (!Character.isWhitespace(character) && character != '*' && character != '/') {
60                      isBlank = false;
61                  }
62              }
63  
64              if (isBlank) {
65                  numberOfBlankLines++;
66              }
67          }
68          return numberOfBlankLines;
69      }
70  
71      public int getNumberOfCommentedOutLinesOfCode() {
72          if (isDocComment()) {
73              return 0;
74          }
75  
76          int numberOfCommentedOutLinesOfCode = 0;
77          for (String line : lines) {
78              String strippedLine = org.apache.commons.lang3.StringUtils.strip(line, " /*");
79              if (CodeDetector.hasDetectedCode(strippedLine)) {
80                  numberOfCommentedOutLinesOfCode++;
81              }
82          }
83          return numberOfCommentedOutLinesOfCode;
84      }
85  
86      public boolean isDocComment() {
87          return type == CommentType.DOC;
88      }
89  
90      public boolean isHeaderComment() {
91          return type == CommentType.HEADER;
92      }
93  
94      @Override
95      public int hashCode() {
96          return new HashCodeBuilder().append(type).append(lines).toHashCode();
97      }
98  
99      @Override
100     public boolean equals(Object obj) {
101         if (!(obj instanceof Comment)) {
102             return false;
103         }
104 
105         Comment other = (Comment) obj;
106         return new EqualsBuilder().append(type, other.type).append(lines, other.lines).isEquals();
107     }
108 
109     @Override
110     public String toString() {
111         final String firstLine = lines.isEmpty() ? "" : lines.get(0);
112         final String lastLine = lines.isEmpty() ? "" : lines.get(lines.size() - 1);
113         return new ToStringBuilder(this).append("type", type).append("firstLine", firstLine)
114                 .append("lastLine", lastLine).append("numberOfLines", getNumberOfLines())
115                 .append("numberOfCommentedOutLinesOfCode", getNumberOfCommentedOutLinesOfCode()).toString();
116     }
117 }
