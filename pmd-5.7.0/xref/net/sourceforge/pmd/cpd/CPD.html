28  public class CPD {
29      private static final Logger LOGGER = Logger.getLogger(CPD.class.getName());
30  
31      private CPDConfiguration configuration;
32  
33      private Map<String, SourceCode> source = new TreeMap<>();
34      private CPDListener listener = new CPDNullListener();
35      private Tokens tokens = new Tokens();
36      private MatchAlgorithm matchAlgorithm;
37      private Set<String> current = new HashSet<>();
38  
39      public CPD(CPDConfiguration theConfiguration) {
40          configuration = theConfiguration;
41          // before we start any tokenizing (add(File...)), we need to reset the
42          // static TokenEntry status
43          TokenEntry.clearImages();
44      }
45  
46      public void setCpdListener(CPDListener cpdListener) {
47          this.listener = cpdListener;
48      }
49  
50      public void go() {
51          matchAlgorithm = new MatchAlgorithm(source, tokens, configuration.getMinimumTileSize(), listener);
52          matchAlgorithm.findMatches();
53      }
54  
55      public Iterator<Match> getMatches() {
56          return matchAlgorithm.matches();
57      }
58  
59      public void addAllInDirectory(File dir) throws IOException {
60          addDirectory(dir, false);
61      }
62  
63      public void addRecursively(File dir) throws IOException {
64          addDirectory(dir, true);
65      }
66  
67      public void add(List<File> files) throws IOException {
68          for (File f : files) {
69              add(f);
70          }
71      }
72  
73      private void addDirectory(File dir, boolean recurse) throws IOException {
74          if (!dir.exists()) {
75              throw new FileNotFoundException("Couldn't find directory " + dir);
76          }
77          FileFinder finder = new FileFinder();
78          // TODO - could use SourceFileSelector here
79          add(finder.findFilesFrom(dir, configuration.filenameFilter(), recurse));
80      }
81  
82      public void add(File file) throws IOException {
83  
84          if (configuration.isSkipDuplicates()) {
85              // TODO refactor this thing into a separate class
86              String signature = file.getName() + '_' + file.length();
87              if (current.contains(signature)) {
88                  System.err.println("Skipping " + file.getAbsolutePath()
89                          + " since it appears to be a duplicate file and --skip-duplicate-files is set");
90                  return;
91              }
92              current.add(signature);
93          }
94  
95          if (!FilenameUtils.equalsNormalizedOnSystem(file.getAbsoluteFile().getCanonicalPath(),
96                  file.getAbsolutePath())) {
97              System.err.println("Skipping " + file + " since it appears to be a symlink");
98              return;
99          }
100 
101         if (!file.exists()) {
102             System.err.println("Skipping " + file + " since it doesn't exist (broken symlink?)");
103             return;
104         }
105 
106         SourceCode sourceCode = configuration.sourceCodeFor(file);
107         add(sourceCode);
108     }

110     public void add(DBURI dburi) throws IOException {
111 
112         try {
113             DBMSMetadata dbmsmetadata = new DBMSMetadata(dburi);
114 
115             List<SourceObject> sourceObjectList = dbmsmetadata.getSourceObjectList();
116             LOGGER.log(Level.FINER, "Located {0} database source objects", sourceObjectList.size());
117 
118             for (SourceObject sourceObject : sourceObjectList) {
119                 // Add DBURI as a faux-file
120                 String falseFilePath = sourceObject.getPseudoFileName();
121                 LOGGER.log(Level.FINEST, "Adding database source object {0}", falseFilePath);
122 
123                 SourceCode sourceCode = configuration.sourceCodeFor(dbmsmetadata.getSourceCode(sourceObject),
124                         falseFilePath);
125                 add(sourceCode);
126             }
127         } catch (Exception sqlException) {
128             LOGGER.log(Level.SEVERE, "Problem with Input URI", sqlException);
129             throw new RuntimeException("Problem with DBURI: " + dburi, sqlException);
130         }
131     }
132 
133     private void add(SourceCode sourceCode) throws IOException {
134         if (configuration.isSkipLexicalErrors()) {
135             addAndSkipLexicalErrors(sourceCode);
136         } else {
137             addAndThrowLexicalError(sourceCode);
138         }
139     }
140 
141     private void addAndThrowLexicalError(SourceCode sourceCode) throws IOException {
142         configuration.tokenizer().tokenize(sourceCode, tokens);
143         listener.addedFile(1, new File(sourceCode.getFileName()));
144         source.put(sourceCode.getFileName(), sourceCode);
145     }
146 
147     private void addAndSkipLexicalErrors(SourceCode sourceCode) throws IOException {
148         TokenEntry.State savedTokenEntry = new TokenEntry.State(tokens.getTokens());
149         try {
150             addAndThrowLexicalError(sourceCode);
151         } catch (TokenMgrError e) {
152             System.err.println("Skipping " + sourceCode.getFileName() + ". Reason: " + e.getMessage());
153             tokens.getTokens().clear();
154             tokens.getTokens().addAll(savedTokenEntry.restore());
155         }
156     }
