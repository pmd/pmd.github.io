38       * The {@link TypeSet} provides type resolution for the symbol facade.
39       * 
40       * @param classLoader

42       *            auxiliary class path)

49          hasAuxclasspath = cl instanceof ClasspathClassLoader;
50          pmdClassLoader = PMDASMClassLoader.getInstance(cl);

63      /**
64       * A resolver that can resolve a class by name. The name can be a simple
65       * name or a fully qualified name.
66       */
67      // TODO should Resolver provide a canResolve() and a resolve()? Requiring 2
68      // calls seems clunky... but so does this throwing an exception for flow
69      // control...
70      public interface Resolver {
71          /**
72           * Resolve the class by the given name
73           *
74           * @param name
75           *            the name of the class, might be fully classified or not.
76           * @return the class
77           * @throws ClassNotFoundException
78           *             if the class couldn't be found
79           */
80          Class<?> resolve(String name) throws ClassNotFoundException;
81  
82          /**
83           * Checks if the given class could be resolved by this resolver. Notice,
84           * that a resolver's ability to resolve a class does not imply that the
85           * class will actually be found and resolved.
86           * 
87           * @param name
88           *            the name of the class, might be fully classified or not.
89           * @return whether the class can be resolved
90           */
91          boolean couldResolve(String name);
92      }

94      /**
95       * Base Resolver class that support a {@link PMDASMClassLoader} class
96       * loader.
97       */
98      public abstract static class AbstractResolver implements Resolver {
99          /** the class loader. */
100         protected final PMDASMClassLoader pmdClassLoader;
101         private final Map<String, String> classNames;
102 
103         /**
104          * Creates a new AbstractResolver that uses the given class loader.
105          * 
106          * @param pmdClassLoader
107          *            the class loader to use
108          */
109         public AbstractResolver(final PMDASMClassLoader pmdClassLoader) {
110             this.pmdClassLoader = pmdClassLoader;
111             classNames = new HashMap<>();
112         }
113 
114         /**
115          * Resolves the given class name with the given FQCN, considering it may
116          * be an inner class.
117          *
118          * @param name
119          *            The name of the class to load.
120          * @param fqName
121          *            The proposed FQCN for the class.
122          * @return The matched class or null if not found.
123          */
124         protected Class<?> resolveMaybeInner(final String name, final String fqName) {
125             // Do we know the actual class name?
126             final String className = classNames.get(name);
127             if (className != null) {
128                 try {
129                     return pmdClassLoader.loadClass(className);
130                 } catch (final ClassNotFoundException e) {
131                     // Ignored, can never actually happen
132                 }
133             }
134 
135             if (fqName != null) {
136                 final StringBuilder sb = new StringBuilder(fqName);
137                 String actualClassName = fqName;
138                 // We have a FQCN, but it may be an inner class, so we have to
139                 // brute force our way...
140                 do {
141                     if (pmdClassLoader.couldResolve(actualClassName)) {
142                         try {
143                             final Class<?> c = pmdClassLoader.loadClass(actualClassName);
144                             // Update the mapping
145                             classNames.put(name, actualClassName);
146                             return c;
147                         } catch (final ClassNotFoundException e) {
148                             // Ignored
149                         }
150                     }
151 
152                     // Check if the last segment is an inner class
153                     final int lastDot = actualClassName.lastIndexOf('.');
154                     if (lastDot == -1) {
155                         break;
156                     }
157 
158                     sb.setCharAt(lastDot, '$');
159                     actualClassName = sb.toString();
160                 } while (true);
161             }
162 
163             return null;
164         }
165 
166         public boolean couldResolve(final String name) {
167             /*
168              * Resolvers based on this one, will attempt to load the class from
169              * the class loader, so ask him
170              */
171             return classNames.containsKey(name) || pmdClassLoader.couldResolve(name);
172         }
173     }

175     /**
176      * Resolver that tries to resolve the given simple class name with the
177      * explicit import statements.
178      */
179     public static class ExplicitImportResolver extends AbstractResolver {
180         private Map<String, String> importStmts;
181 
182         /**
183          * Creates a new {@link ExplicitImportResolver}.
184          * 
185          * @param pmdClassLoader
186          *            the class loader to use.
187          * @param importStmts
188          *            the import statements
189          */
190         public ExplicitImportResolver(PMDASMClassLoader pmdClassLoader, Set<String> importStmts) {
191             super(pmdClassLoader);
192 
193             // unfold imports, to store both FQ and unqualified names mapped to
194             // the FQ name
195             this.importStmts = new HashMap<>();
196             for (final String stmt : importStmts) {
197                 if (stmt.endsWith("*")) {
198                     continue;
199                 }
200 
201                 this.importStmts.put(stmt, stmt);
202                 final int lastDotIdx = stmt.lastIndexOf('.');
203                 if (lastDotIdx != -1) {
204                     this.importStmts.put(stmt.substring(lastDotIdx + 1), stmt);
205                 }
206             }
207         }
208 
209         @Override
210         public Class<?> resolve(final String name) throws ClassNotFoundException {
211             final Class<?> c = resolveMaybeInner(name, importStmts.get(name));
212 
213             if (c == null) {
214                 throw new ClassNotFoundException("Type " + name + " not found");
215             }
216 
217             return c;
218         }
219 
220         @Override
221         public boolean couldResolve(final String name) {
222             return importStmts.containsKey(name);
223         }
224     }

226     /**
227      * Resolver that uses the current package to resolve a simple class name.
228      */
229     public static class CurrentPackageResolver extends AbstractResolver {
230         private final String pkg;
231 
232         /**
233          * Creates a new {@link CurrentPackageResolver}
234          * 
235          * @param pmdClassLoader
236          *            the class loader to use
237          * @param pkg
238          *            the package name
239          */
240         public CurrentPackageResolver(PMDASMClassLoader pmdClassLoader, String pkg) {
241             super(pmdClassLoader);
242             if (pkg == null || pkg.length() == 0) {
243                 this.pkg = null;
244             } else {
245                 this.pkg = pkg + ".";
246             }
247         }
248 
249         @Override
250         public Class<?> resolve(String name) throws ClassNotFoundException {
251             if (name == null) {
252                 throw new ClassNotFoundException();
253             }
254 
255             return pmdClassLoader.loadClass(qualifyName(name));
256         }
257 
258         @Override
259         public boolean couldResolve(String name) {
260             return pmdClassLoader.couldResolve(qualifyName(name));
261         }
262 
263         private String qualifyName(final String name) {
264             final String qualifiedName = name.replace('.', '$');
265             if (pkg == null) {
266                 return qualifiedName;
267             }
268 
269             /*
270              * String.concat is bad in general, but for simple 2 string concatenation, it's the fastest
271              * See http://www.rationaljava.com/2015/02/the-optimum-method-to-concatenate.html
272              */
273             return pkg.concat(qualifiedName);
274         }
275     }

277     /**
278      * Resolver that resolves simple class names from the implicit import of
279      * <code>java.lang.*</code>.
280      */
281     // TODO cite the JLS section on implicit imports
282     public static class ImplicitImportResolver extends AbstractResolver {
283         /*
284          * They aren't so many to bother about memory, but are used all the
285          * time, so we worry about performance. On average, you can expect this
286          * cache to have ~90% hit ratio unless abusing star imports (import on
287          * demand)
288          */
289         private static final ConcurrentHashMap<String, Class<?>> CLASS_CACHE = new ConcurrentHashMap<>();
290 
291         /**
292          * Creates a {@link ImplicitImportResolver}
293          * 
294          * @param pmdClassLoader
295          *            the class loader
296          */
297         public ImplicitImportResolver(PMDASMClassLoader pmdClassLoader) {
298             super(pmdClassLoader);
299         }
300 
301         @Override
302         public Class<?> resolve(String name) throws ClassNotFoundException {
303             if (name == null) {
304                 throw new ClassNotFoundException();
305             }
306 
307             Class<?> clazz = CLASS_CACHE.get(name);
308             if (clazz != null) {
309                 return clazz;
310             }
311 
312             /*
313              * String.concat is bad in general, but for simple 2 string concatenation, it's the fastest
314              * See http://www.rationaljava.com/2015/02/the-optimum-method-to-concatenate.html
315              */
316             clazz = pmdClassLoader.loadClass("java.lang.".concat(name.replace('.', '$')));
317             CLASS_CACHE.putIfAbsent(name, clazz);
318 
319             return clazz;
320         }
321 
322         @Override
323         public boolean couldResolve(String name) {
324             /*
325              * String.concat is bad in general, but for simple 2 string concatenation, it's the fastest
326              * See http://www.rationaljava.com/2015/02/the-optimum-method-to-concatenate.html
327              */
328             return pmdClassLoader.couldResolve("java.lang.".concat(name.replace('.', '$')));
329         }
330     }

332     /**
333      * Resolver that uses the "on demand" import statements.
334      */
335     public static class ImportOnDemandResolver extends AbstractResolver {
336         private Set<String> importStmts;
337 
338         /**
339          * Creates a {@link ImportOnDemandResolver}
340          * 
341          * @param pmdClassLoader
342          *            the class loader to use
343          * @param importStmts
344          *            the import statements
345          */
346         public ImportOnDemandResolver(PMDASMClassLoader pmdClassLoader, Set<String> importStmts) {
347             super(pmdClassLoader);
348             this.importStmts = new HashSet<>();
349             for (final String stmt : importStmts) {
350                 if (stmt.endsWith("*")) {
351                     this.importStmts.add(stmt);
352                 }
353             }
354         }
355 
356         @Override
357         public Class<?> resolve(String name) throws ClassNotFoundException {
358             if (name == null) {
359                 throw new ClassNotFoundException();
360             }
361 
362             name = name.replace('.', '$');
363             for (String importStmt : importStmts) {
364                 final String fqClassName = new StringBuilder(importStmt.length() + name.length()).append(importStmt)
365                         .replace(importStmt.length() - 1, importStmt.length(), name).toString();
366                 if (pmdClassLoader.couldResolve(fqClassName)) {
367                     try {
368                         return pmdClassLoader.loadClass(fqClassName);
369                     } catch (ClassNotFoundException e) {
370                         // ignored
371                     }
372                 }
373             }
374 
375             throw new ClassNotFoundException("Type " + name + " not found");
376         }
377 
378         @Override
379         public boolean couldResolve(String name) {
380             name = name.replace('.', '$');
381             for (String importStmt : importStmts) {
382                 final String fqClassName = new StringBuilder(importStmt.length() + name.length()).append(importStmt)
383                         .replace(importStmt.length() - 1, importStmt.length(), name).toString();
384                 // can any class be resolved / was never attempted?
385                 if (pmdClassLoader.couldResolve(fqClassName)) {
386                     return true;
387                 }
388             }
389 
390             return false;
391         }
392     }

394     /**
395      * Resolver that resolves primitive types such as int or double.
396      */
397     public static class PrimitiveTypeResolver implements Resolver {
398         private static final Map<String, Class<?>> PRIMITIVE_TYPES;
399 
400         static {
401             final Map<String, Class<?>> types = new HashMap<>();
402             types.put("int", int.class);
403             types.put("float", float.class);
404             types.put("double", double.class);
405             types.put("long", long.class);
406             types.put("boolean", boolean.class);
407             types.put("byte", byte.class);
408             types.put("short", short.class);
409             types.put("char", char.class);
410             PRIMITIVE_TYPES = Collections.unmodifiableMap(types);
411         }
412 
413         @Override
414         public Class<?> resolve(String name) throws ClassNotFoundException {
415             if (!PRIMITIVE_TYPES.containsKey(name)) {
416                 throw new ClassNotFoundException(name);
417             }
418             return PRIMITIVE_TYPES.get(name);
419         }
420 
421         @Override
422         public boolean couldResolve(String name) {
423             return PRIMITIVE_TYPES.containsKey(name);
424         }
425     }

445     /**
446      * Resolver that simply loads the class by name. This only works if the
447      * class name is given as a fully qualified name.
448      */
449     public static class FullyQualifiedNameResolver extends AbstractResolver {
450         /**
451          * Creates a {@link FullyQualifiedNameResolver}
452          * 
453          * @param pmdClassLoader
454          *            the class loader to use
455          */
456         public FullyQualifiedNameResolver(PMDASMClassLoader pmdClassLoader) {
457             super(pmdClassLoader);
458         }
459 
460         @Override
461         public Class<?> resolve(String name) throws ClassNotFoundException {
462             if (name == null) {
463                 throw new ClassNotFoundException();
464             }
465 
466             final Class<?> c = resolveMaybeInner(name, name);
467 
468             if (c == null) {
469                 throw new ClassNotFoundException("Type " + name + " not found");
470             }
471 
472             return c;
473         }
474 
475         @Override
476         public boolean couldResolve(String name) {
477             /*
478              * We can always try!
479              * If a file used an explicit import on A.Inner, the class loader will register
480              * A.Inner can't be resolved even if A$Inner can.
481              * If a second file used A.Inner without an explicit import, we would end here,
482              * super.couldResolve("A.Inner") will return false, but we CAN resolve it as A$Inner.
483              */
484             return true;
485         }
486     }

514     /**
515      * Resolves a class by its name using all known resolvers.
516      * 
517      * @param name
518      *            the name of the class, can be a simple name or a fully
519      *            qualified name.
520      * @return the class or <code>null</code> if none found
521      */
522     public Class<?> findClass(String name) {
523         // we don't build the resolvers until now since we first want to get all
524         // the imports
525         if (resolvers.isEmpty()) {
526             buildResolvers();
527         }
528 
529         for (final Resolver resolver : resolvers) {
530             if (resolver.couldResolve(name)) {
531                 try {
532                     return resolver.resolve(name);
533                 } catch (ClassNotFoundException cnfe) {
534                     // ignored, maybe another resolver will find the class
535                 }
536             }
537         }
538 
539         return null;
540     }
541 
542     private void buildResolvers() {
543         resolvers.add(new PrimitiveTypeResolver());
544         resolvers.add(new VoidResolver());
545         resolvers.add(new ExplicitImportResolver(pmdClassLoader, imports));
546         resolvers.add(new CurrentPackageResolver(pmdClassLoader, pkg));
547         resolvers.add(new ImplicitImportResolver(pmdClassLoader));
548         resolvers.add(new ImportOnDemandResolver(pmdClassLoader, imports));
549         resolvers.add(new FullyQualifiedNameResolver(pmdClassLoader));
550     }
