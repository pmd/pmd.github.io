5   package net.sourceforge.pmd;

7   import java.util.List;
8   
9   import net.sourceforge.pmd.lang.DummyLanguageModule;
10  import net.sourceforge.pmd.lang.LanguageRegistry;
11  import net.sourceforge.pmd.lang.ast.Node;
12  import net.sourceforge.pmd.lang.rule.AbstractRule;
13  
14  /**
15   * Sample rule that detect any node with an image of "Foo". Used for testing.
16   */
17  public class FooRule extends AbstractRule {
18      public FooRule() {
19          setLanguage(LanguageRegistry.getLanguage(DummyLanguageModule.NAME));
20      }
21  
22      @Override
23      public String getMessage() {
24          return "blah";
25      }
26  
27      @Override
28      public String getName() {
29          return "Foo";
30      }
31  
32      @Override
33      public String getRuleSetName() {
34          return "RuleSet";
35      }
36  
37      @Override
38      public String getDescription() {
39          return "desc";
40      }
41  
42      @Override
43      public void apply(List<? extends Node> nodes, RuleContext ctx) {
44          for (Node node : nodes) {
45              apply(node, ctx);
46          }
47      }
48  
49      protected void apply(Node node, RuleContext ctx) {
50          for (int i = 0; i < node.jjtGetNumChildren(); i++) {
51              apply(node.jjtGetChild(i), ctx);
52          }
53          if ("Foo".equals(node.getImage())) {
54              addViolation(ctx, node);
55          }
56      }
57  }
