16  import net.sourceforge.pmd.PMD;

18  import net.sourceforge.pmd.lang.java.JavaLanguageModule;
19  import net.sourceforge.pmd.lang.java.ParserTst;
20  
21  public class ASTFormalParameterTest extends ParserTst {
22  
23      @Test
24      public void testVarargs() {
25          int nrOfVarArgs = 0;
26          int nrOfNoVarArgs = 0;
27  
28          Set<ASTFormalParameter> ops = getNodes(LanguageRegistry.getLanguage(JavaLanguageModule.NAME).getVersion("1.5"),
29                  ASTFormalParameter.class, TEST1);
30          for (Iterator<ASTFormalParameter> iter = ops.iterator(); iter.hasNext();) {
31              ASTFormalParameter b = iter.next();
32              ASTVariableDeclaratorId variableDeclId = b.getFirstDescendantOfType(ASTVariableDeclaratorId.class);
33              if (!"x".equals(variableDeclId.getImage())) {
34                  assertTrue(b.isVarargs());
35                  nrOfVarArgs++;
36              } else {
37                  assertFalse(b.isVarargs());
38                  nrOfNoVarArgs++;
39              }
40          }
41  
42          // Ensure that both possibilities are tested
43          assertEquals(1, nrOfVarArgs);
44          assertEquals(1, nrOfNoVarArgs);
45      }
46  
47      private static final String TEST1 = "class Foo {" + PMD.EOL + " void bar(int x, int... others) {}" + PMD.EOL + "}";
48  }
