{% highlight bash %} $ cd $HOME $ curl -OL https://github.com/pmd/pmd/releases/download/pmd_releases%2F{{ site.pmd.latestVersion }}/pmd-bin-{{ site.pmd.latestVersion }}.zip $ unzip pmd-bin-{{ site.pmd.latestVersion }}.zip $ alias pmd="$HOME/pmd-bin-{{ site.pmd.latestVersion }}/bin/run.sh pmd" $ pmd -d /usr/src -R rulesets/java/quickstart.xml -f text {% endhighlight %}
  1. Download pmd-bin-{{ site.pmd.latestVersion }}.zip
  2. Extract the zip-archive, e.g. to C:\pmd-bin-{{ site.pmd.latestVersion }}
  3. Add folder C:\pmd-bin-{{ site.pmd.latestVersion }}\bin to PATH, either
    1. Permanently: Using System Properties dialog > Environment variables > Append to PATH variable
    2. Temporarily, at command line: SET PATH=C:\pmd-bin-{{ site.pmd.latestVersion }};%PATH%
  4. Execute at command line: pmd.bat -d c:\src -R rulesets/java/quickstart.xml -f text
  1. Install Chocolatey, if you don't already have it!
  2. Install PMD: choco install -y pmd
  3. Run from the command-line! pmd -d c:\src -R rulesets/java/quickstart.xml -f text
