4   package net.sourceforge.pmd.cpd;

6   import java.io.StringReader;
7   
8   import net.sourceforge.pmd.lang.LanguageRegistry;
9   import net.sourceforge.pmd.lang.LanguageVersionHandler;
10  import net.sourceforge.pmd.lang.TokenManager;
11  import net.sourceforge.pmd.lang.ast.TokenMgrError;
12  import net.sourceforge.pmd.lang.cpp.CppLanguageModule;
13  import net.sourceforge.pmd.lang.cpp.ast.Token;
14  
15  import org.apache.commons.io.IOUtils;
16  
17  /**
18   * The C++ tokenizer.
19   */
20  public class CPPTokenizer implements Tokenizer {
21  
22      @Override
23      public void tokenize(SourceCode sourceCode, Tokens tokenEntries) {
24          StringBuilder buffer = sourceCode.getCodeBuffer();
25          StringReader reader = null;
26          try {
27              LanguageVersionHandler languageVersionHandler = LanguageRegistry.getLanguage(CppLanguageModule.NAME)
28                      .getDefaultVersion().getLanguageVersionHandler();
29              reader = new StringReader(buffer.toString());
30              TokenManager tokenManager = languageVersionHandler.getParser(
31                      languageVersionHandler.getDefaultParserOptions()).getTokenManager(sourceCode.getFileName(), reader);
32              Token currentToken = (Token) tokenManager.getNextToken();
33              while (currentToken.image.length() > 0) {
34                  tokenEntries.add(new TokenEntry(currentToken.image, sourceCode.getFileName(), currentToken.beginLine));
35                  currentToken = (Token) tokenManager.getNextToken();
36              }
37              tokenEntries.add(TokenEntry.getEOF());
38              System.err.println("Added " + sourceCode.getFileName());
39          } catch (TokenMgrError err) {
40              err.printStackTrace();
41              System.err.println("Skipping " + sourceCode.getFileName() + " due to parse error");
42              tokenEntries.add(TokenEntry.getEOF());
43          } finally {
44              IOUtils.closeQuietly(reader);
45          }
46      }
47  }
